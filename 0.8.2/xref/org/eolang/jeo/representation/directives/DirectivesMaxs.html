5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.xembly.Directive;
9   
10  /**
11   * Max stack and locals.
12   * @since 0.3
13   */
14  public final class DirectivesMaxs implements Iterable<Directive> {
15  
16      /**
17       * Undefined value.
18       * Need to be recomputed.
19       */
20      private static final int UNDEFINED = -1;
21  
22      /**
23       * Max stack size.
24       */
25      private final int stack;
26  
27      /**
28       * Max locals size.
29       */
30      private final int locals;
31  
32      /**
33       * Constructor.
34       */
35      public DirectivesMaxs() {
36          this(DirectivesMaxs.UNDEFINED, DirectivesMaxs.UNDEFINED);
37      }
38  
39      /**
40       * Constructor.
41       *
42       * @param stack Max stack size.
43       * @param locals Max locals size.
44       */
45      public DirectivesMaxs(final int stack, final int locals) {
46          this.stack = stack;
47          this.locals = locals;
48      }
49  
50      @Override
51      public Iterator<Directive> iterator() {
52          return new DirectivesJeoObject(
53              "maxs",
54              "maxs",
55              new DirectivesValue(this.stack),
56              new DirectivesValue(this.locals)
57          ).iterator();
58      }
59  }
