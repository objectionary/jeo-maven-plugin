23  /**
24   * Test case for {@link XmlField}.
25   *
26   * @since 0.3
27   */
28  final class XmlFieldTest {
29  
30      @Test
31      void parsesXmirFieldSuccessfully() throws ImpossibleModificationException {
32          final int access = Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
33          final String name = "serialVersionUID";
34          final String descriptor = "J";
35          final long value = 7_099_057_708_183_571_937L;
36          MatcherAssert.assertThat(
37              "Failed to parse XMIR field",
38              new XmlField(
39                  new NativeXmlNode(
40                      new Xembler(
41                          new DirectivesField(
42                              access,
43                              name,
44                              descriptor,
45                              "",
46                              value
47                          )
48                      ).xml()
49                  )
50              ).bytecode(),
51              Matchers.equalTo(
52                  new BytecodeField(
53                      name,
54                      descriptor,
55                      null,
56                      value,
57                      access,
58                      new BytecodeAnnotations()
59                  )
60              )
61          );
62      }
63  
64      @ParameterizedTest
65      @MethodSource("values")
66      void parsesFieldWithValues(final Object value) throws ImpossibleModificationException {
67          final int access = Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
68          final String name = "serialVersionUID";
69          final String descriptor = "Ljava/lang/String;";
70          MatcherAssert.assertThat(
71              "We expect the value to be the same",
72              new XmlField(
73                  new NativeXmlNode(
74                      new Xembler(
75                          new DirectivesField(access, name, descriptor, null, value)
76                      ).xml()
77                  )
78              ).bytecode(),
79              Matchers.equalTo(
80                  new BytecodeField(
81                      name, descriptor, null, value, access
82                  )
83              )
84          );
85      }
86  
87      @Test
88      void retrievesFieldAnnotations() throws ImpossibleModificationException {
89          final String override = "java/lang/Override";
90          final String safe = "java/lang/SafeVarargs";
91          MatcherAssert.assertThat(
92              "Annotations are not found",
93              new XmlField(
94                  new NativeXmlNode(
95                      new Xembler(
96                          new DirectivesField(
97                              new DirectivesAnnotation(override, true),
98                              new DirectivesAnnotation(safe, true)
99                          )
100                     ).xml()
101                 )
102             ).bytecode(),
103             Matchers.equalTo(
104                 new BytecodeField(
105                     "unknown",
106                     "I",
107                     null,
108                     0,
109                     1,
110                     new BytecodeAnnotations(
111                         new BytecodeAnnotation(override, true),
112                         new BytecodeAnnotation(safe, true)
113                     )
114                 )
115             )
116         );
117     }
118 
119     @ParameterizedTest
120     @ValueSource(strings = {
121         "Φ", "Ψ", "Ω", "Δ", "Σ", "Θ", "Λ", "Ξ", "Π", "Υ", "\u03A3", "\u03A6", "\u03A8", "\u03A9"
122     })
123     void parsesXmirFieldWithUnicodeCharacterInTheName(final String original)
124         throws ImpossibleModificationException {
125         final BytecodeField expected = new BytecodeField(
126             original,
127             "I",
128             null,
129             0,
130             Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL
131         );
132         MatcherAssert.assertThat(
133             "We expect the field name with unicode characters to be successfully parsed from directives",
134             new XmlField(new NativeXmlNode(new Xembler(expected.directives()).xml())).bytecode(),
135             Matchers.equalTo(expected)
136         );
137     }
