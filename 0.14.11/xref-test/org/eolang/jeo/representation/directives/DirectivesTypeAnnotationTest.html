5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import java.util.Collections;
9   import org.hamcrest.MatcherAssert;
10  import org.junit.jupiter.api.Test;
11  import org.objectweb.asm.TypeReference;
12  import org.xembly.ImpossibleModificationException;
13  import org.xembly.Xembler;
14  
15  /**
16   * Test case for {@link DirectivesTypeAnnotationTest}.
17   * @since 0.15.0
18   */
19  final class DirectivesTypeAnnotationTest {
20  
21      @Test
22      void translatesToCorrectXmir() throws ImpossibleModificationException {
23          final Format format = new Format();
24          MatcherAssert.assertThat(
25              "We expect the directives type annotation to be converted to correct XMIR",
26              new Xembler(
27                  new DirectivesTypeAnnotation(
28                      format,
29                      42,
30                      TypeReference.FIELD,
31                      "*",
32                      "LAnn;",
33                      true,
34                      Collections.singletonList(
35                          new DirectivesPlainAnnotationValue(
36                              0,
37                              format,
38                              "key",
39                              "value"
40                          )
41                      )
42                  )
43              ).xml(),
44              XhtmlMatchers.hasXPaths(
45                  "./o[@name='type-annotation-42']",
46                  new JeoBaseXpath("./o[@name='type-annotation-42']", "seq.of5").toXpath(),
47                  "./o/o[contains(@base,'number') and @name='v0']/o/o[ text()='40-33-00-00-00-00-00-00']",
48                  "./o/o[contains(@base,'string') and @name='v1']/o/o[ text()='2A-']",
49                  "./o/o[contains(@base,'string') and @name='v2']/o/o[ text()='4C-41-6E-6E-3B']",
50                  "./o/o[contains(@base,'true') and @name='v3']",
51                  "./o/o[contains(@name,'p0')]"
52              )
53          );
54      }
55  }
