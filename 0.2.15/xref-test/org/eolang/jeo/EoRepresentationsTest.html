24  package org.eolang.jeo;
25  
26  import java.io.IOException;
27  import java.nio.charset.StandardCharsets;
28  import java.nio.file.Files;
29  import java.nio.file.Path;
30  import org.eolang.jeo.representation.bytecode.BytecodeClass;
31  import org.hamcrest.MatcherAssert;
32  import org.hamcrest.Matchers;
33  import org.junit.jupiter.api.Assertions;
34  import org.junit.jupiter.api.Test;
35  import org.junit.jupiter.api.io.TempDir;
36  
37  /**
38   * Test case for {@link EoRepresentations}.
39   *
40   * @since 0.1.0
41   */
42  final class EoRepresentationsTest {
43  
44      @Test
45      void retrievesObjectsSuccessfully(@TempDir final Path temp) throws IOException {
46          final int expected = 2;
47          final Path directory = temp.resolve(new XmirDefaultDirectory().toPath());
48          Files.createDirectories(directory);
49          Files.write(
50              directory.resolve("first.xmir"),
51              new BytecodeClass("First").xml().toString().getBytes(StandardCharsets.UTF_8)
52          );
53          Files.write(
54              directory.resolve("second.xmir"),
55              new BytecodeClass("Second").xml().toString().getBytes(StandardCharsets.UTF_8)
56          );
57          MatcherAssert.assertThat(
58              String.format("Objects were not retrieved, we expected '%d' objects", expected),
59              new EoRepresentations(temp).objects(),
60              Matchers.hasSize(expected)
61          );
62      }
63  
64      @Test
65      void retrievesEmptyObjectsIfFolderIsEmpty(@TempDir final Path temp) throws IOException {
66          Files.createDirectories(temp.resolve(new XmirDefaultDirectory().toPath()));
67          MatcherAssert.assertThat(
68              "Objects were not retrieved, we expected empty list",
69              new EoRepresentations(temp).objects(),
70              Matchers.empty()
71          );
72      }
73  
74      @Test
75      void throwsExceptionIfFolderDoesNotExist(@TempDir final Path temp) {
76          Assertions.assertThrows(
77              IllegalStateException.class,
78              () -> new EoRepresentations(temp).objects(),
79              "Exception was not thrown when folder does not exist"
80          );
81      }
82  }
