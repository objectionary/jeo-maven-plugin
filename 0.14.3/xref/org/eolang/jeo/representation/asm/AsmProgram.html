5   package org.eolang.jeo.representation.asm;
6   
7   import org.eolang.jeo.representation.ClassName;
8   import org.eolang.jeo.representation.bytecode.BytecodeObject;
9   import org.objectweb.asm.ClassReader;
10  import org.objectweb.asm.tree.ClassNode;
11  
12  /**
13   * ASM bytecode parser.
14   * We used to use the Visitor pattern, but it's too verbose and not very readable.
15   * So, we decided to switch to a Tree-API-based approach.
16   * You can read more about different approaches right here:
17   * <a href="https://asm.ow2.io/asm4-guide.pdf">https://asm.ow2.io/asm4-guide.pdf</a>
18   * The recent version with the Visitor pattern is still available in the history:
19   * <a href="https://github.com/objectionary/jeo-maven-plugin/tree/29daa0a167b5c2ba4caaceafb6e6bafc381ac05c">github</a>
20   * @since 0.6
21   */
22  public final class AsmProgram {
23  
24      /**
25       * Bytecode as plain bytes.
26       */
27      private final byte[] bytes;
28  
29      /**
30       * Constructor.
31       * @param bytes Bytes.
32       */
33      public AsmProgram(final byte... bytes) {
34          this.bytes = bytes.clone();
35      }
36  
37      /**
38       * Convert to bytecode.
39       * @return Bytecode.
40       */
41      public BytecodeObject bytecode() {
42          return this.bytecode(ClassReader.SKIP_DEBUG);
43      }
44  
45      /**
46       * Convert to bytecode.
47       * @param flags Flags.
48       * @return Bytecode.
49       */
50      public BytecodeObject bytecode(final int flags) {
51          final ClassNode node = new ClassNode();
52          new ClassReader(this.bytes).accept(node, AsmUnknownAttributes.prototypes(), flags);
53          return new BytecodeObject(
54              new ClassName(node.name).pckg(),
55              new AsmClass(node).bytecode()
56          );
57      }
58  }
