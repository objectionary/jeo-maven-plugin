5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import java.util.Optional;
9   import org.xembly.Directive;
10  
11  /**
12   * An annotation value that is an enumeration.
13   * @since 0.6
14   */
15  public final class DirectivesEnumAnnotationValue implements Iterable<Directive> {
16  
17      /**
18       * Index of the annotation value among other annotation values.
19       */
20      private final int index;
21  
22      /**
23       * Format of the directives.
24       */
25      private final Format format;
26  
27      /**
28       * The name of the annotation property.
29       */
30      private final String name;
31  
32      /**
33       * The descriptor of the enumeration.
34       */
35      private final String descriptor;
36  
37      /**
38       * The actual enumeration value.
39       */
40      private final String value;
41  
42      /**
43       * Constructor.
44       * @param index Index of the annotation value among other annotation values.
45       * @param format Format of the directives.
46       * @param name The name of the annotation property.
47       * @param descriptor The descriptor of the enumeration.
48       * @param value The actual enumeration value.
49       * @checkstyle ParameterNumber (5 lines)
50       */
51      public DirectivesEnumAnnotationValue(
52          final int index,
53          final Format format,
54          final String name,
55          final String descriptor,
56          final String value
57      ) {
58          this.index = index;
59          this.format = format;
60          this.name = name;
61          this.descriptor = descriptor;
62          this.value = value;
63      }
64  
65      @Override
66      public Iterator<Directive> iterator() {
67          return new DirectivesJeoObject(
68              "annotation-property",
69              new NumName("e", this.index).toString(),
70              new DirectivesValue(0, this.format, "ENUM"),
71              new DirectivesValue(1, this.format, Optional.ofNullable(this.name).orElse("")),
72              new DirectivesValue(2, this.format, this.descriptor),
73              new DirectivesValue(3, this.format, this.value)
74          ).iterator();
75      }
76  }
