5   package org.eolang.jeo;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.stream.Stream;
12  import org.eolang.jeo.representation.directives.Format;
13  
14  /**
15   * Disassembler for bytecode classes.
16   *
17   * <p>This class disassembles the project's compiled Java bytecode classes into
18   * XMIR (EO XML representation). It processes all .class files from a specified
19   * directory and converts them into corresponding XMIR files, supporting different
20   * disassembly modes for various levels of detail.</p>
21   * @since 0.1.0
22   */
23  public final class Disassembler {
24  
25      /**
26       * Project compiled classes.
27       */
28      private final Classes classes;
29  
30      /**
31       * Where to save decompiled classes.
32       */
33      private final Path target;
34  
35      /**
36       * Disassemble params.
37       */
38      private final Format params;
39  
40      /**
41       * Constructor.
42       * @param classes Directory containing compiled class files
43       * @param target Target directory where XMIR files will be saved
44       */
45      public Disassembler(
46          final Path classes,
47          final Path target
48      ) {
49          this(classes, target, new Format());
50      }
51  
52      /**
53       * Constructor.
54       * @param classes Directory containing compiled class files
55       * @param target Target directory where XMIR files will be saved
56       * @param params Disassembling params.
57       */
58      public Disassembler(
59          final Path classes,
60          final Path target,
61          final Format params
62      ) {
63          this(new BytecodeClasses(classes), target, params);
64      }
65  
66      public Disassembler(
67          final Classes classes,
68          final Path target,
69          final Format params
70      ) {
71          this.classes = classes;
72          this.target = target;
73          this.params = params;
74      }
75  
76      /**
77       * Disassemble all bytecode files.
78       */
79      public void disassemble() {
80          final String process = "Disassembling";
81          final String disassembled = "disassembled";
82          final Stream<Path> stream = new Summary(
83              process,
84              disassembled,
85              this.classes.toString(),
86              this.target,
87              new ParallelTranslator(this::disassemble)
88          ).apply(this.classes.all());
89          stream.forEach(this::log);
90          stream.close();
91      }
92  
93      /**
94       * Disassemble a single bytecode file.
95       * @param path Path to the bytecode file to disassemble
96       * @return Path to the disassembled XMIR file
97       */
98      private Path disassemble(final Path path) {
99          final Transformation trans = new Logging(
100             "Disassembling",
101             "disassembled",
102             new Caching(new Informative(new Disassembling(this.target, path, this.params)))
103         );
104         trans.transform();
105         return trans.target();
106     }
107 
108     /**
109      * Log the disassembling process.
110      * @param disassembled Path to the disassembled XMIR file
111      */
112     private void log(final Path disassembled) {
113         try {
114             Logger.debug(
115                 this,
116                 "Disassembling of %[file]s (%[size]s) finished successfully",
117                 disassembled,
118                 Files.size(disassembled)
119             );
120         } catch (final IOException exception) {
121             throw new IllegalStateException(
122                 String.format(
123                     "Failed to get size of '%s'",
124                     disassembled
125                 ),
126                 exception
127             );
128         }
129     }
130 }
