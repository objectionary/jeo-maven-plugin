5   package org.eolang.jeo.representation.xmir;

7   import org.eolang.jeo.representation.bytecode.BytecodeFrame;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  import org.objectweb.asm.Opcodes;
12  import org.xembly.ImpossibleModificationException;
13  import org.xembly.Xembler;
14  
15  /**
16   * Test case for {@link XmlFrame}.
17   *
18   * @since 0.3
19   */
20  final class XmlFrameTest {
21  
22      @Test
23      void parsesRawXmirNode() throws ImpossibleModificationException {
24          final BytecodeFrame expected = new BytecodeFrame(
25              Opcodes.F_NEW,
26              2,
27              new Object[]{"java/lang/Object", Opcodes.LONG},
28              3,
29              new Object[]{"java/lang/String", Opcodes.DOUBLE}
30          );
31          MatcherAssert.assertThat(
32              "Parsed frame type is not correct.",
33              new XmlFrame(new Xembler(expected.directives()).xml()).bytecode(),
34              Matchers.equalTo(expected)
35          );
36      }
37  }
