5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.xembly.Directive;
9   import org.xembly.Directives;
10  
11  /**
12   * Directives for bytes.
13   * @since 0.6
14   */
15  public final class DirectivesBytes implements Iterable<Directive> {
16  
17      /**
18       * Hex representation of bytes.
19       * For example,
20       * "00 01 02 03 04 05 06 07 08 09 0A 0B 0C 0D 0E 0F"
21       */
22      private final String hex;
23  
24      /**
25       * Constructor.
26       * @param hex Hex representation of bytes.
27       */
28      public DirectivesBytes(final String hex) {
29          this.hex = hex;
30      }
31  
32      @Override
33      public Iterator<Directive> iterator() {
34          return new DirectivesClosedObject(
35              "Q.org.eolang.bytes",
36              new Directives().add("o").set(this.hex).up()
37          ).iterator();
38      }
39  }
