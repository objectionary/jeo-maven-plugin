24  package org.eolang.jeo.representation.bytecode;
25  
26  import org.eolang.jeo.JeoMojo;
27  import org.objectweb.asm.ClassWriter;
28  
29  /**
30   * Custom class writer.
31   * This class works in couple with {@link JeoMojo#initClassloader()} method that sets
32   * the maven classloader as the current thread classloader.
33   * Originally we faced with the problem that {@link ClassWriter} uses classes from ClassLoader
34   * to perform {@link org.objectweb.asm.MethodVisitor#visitMaxs(int, int)} method and if it can't
35   * find the class it throws {@link ClassNotFoundException}. To prevent this we override
36   * {@link ClassWriter#getClassLoader()} method and return the current thread classloader that
37   * knows about all classes that were compiled on the previous maven phases.
38   * You can read more about this problem here:
39   * - https://gitlab.ow2.org/asm/asm/-/issues/317918
40   * - https://stackoverflow.com/questions/11292701/error-while-instrumenting-class-files-asm-classwriter-getcommonsuperclass
41   *
42   * @since 0.1
43   */
44  public final class CustomClassWriter extends ClassWriter {
45  
46      /**
47       * Constructor.
48       */
49      CustomClassWriter() {
50          super(ClassWriter.COMPUTE_FRAMES);
51      }
52  
53      @Override
54      public ClassLoader getClassLoader() {
55          return Thread.currentThread().getContextClassLoader();
56      }
57  }
