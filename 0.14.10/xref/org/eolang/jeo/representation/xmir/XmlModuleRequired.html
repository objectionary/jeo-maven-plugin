5   package org.eolang.jeo.representation.xmir;
6   
7   import org.eolang.jeo.representation.bytecode.BytecodeModuleRequired;
8   
9   /**
10   * XML representation of a required module.
11   * <p>
12   *     Mirrors {@link org.eolang.jeo.representation.bytecode.BytecodeModuleRequired}
13   * </p>
14   * @since 0.15.0
15   */
16  public final class XmlModuleRequired {
17  
18      /**
19       * Required module node to parse.
20       */
21      private final XmlJeoObject node;
22  
23      /**
24       * Constructor.
25       * @param node Required module node.
26       */
27      public XmlModuleRequired(final XmlNode node) {
28          this(new XmlJeoObject(node));
29      }
30  
31      /**
32       * Constructor.
33       * @param node Required module node.
34       */
35      public XmlModuleRequired(final XmlJeoObject node) {
36          this.node = node;
37      }
38  
39      /**
40       * Parse required module to bytecode.
41       * @return Bytecode of the required module.
42       */
43      BytecodeModuleRequired bytecode() {
44          return new BytecodeModuleRequired(
45              this.module(),
46              this.access(),
47              this.version()
48          );
49      }
50  
51      /**
52       * Get module name.
53       * @return Module name.
54       */
55      private String module() {
56          return new XmlValue(this.byName("module")).string();
57      }
58  
59      /**
60       * Get access flags.
61       * @return Access flags.
62       */
63      private int access() {
64          return (int) new XmlValue(this.byName("access")).object();
65      }
66  
67      /**
68       * Get module version.
69       * @return Module version.
70       */
71      private String version() {
72          return new XmlValue(this.byName("version")).string();
73      }
74  
75      /**
76       * Find child node by name.
77       * @param name Name of the child node.
78       * @return Child node.
79       * @throws IllegalStateException When child node is missing.
80       */
81      private XmlNode byName(final String name) {
82          return this.node.children()
83              .filter(n -> n.attribute("name").map(name::equals).orElse(false))
84              .findFirst()
85              .orElseThrow(
86                  () -> new IllegalStateException(
87                      String.format("'%s' is missing in `%s`", name, this.node)
88                  )
89              );
90      }
91  }
