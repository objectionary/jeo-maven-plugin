5   package org.eolang.jeo.representation.bytecode;
6   
7   import java.util.Collections;
8   import org.eolang.jeo.representation.asm.AsmModule;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  import org.objectweb.asm.tree.ClassNode;
13  
14  /**
15   * Test case for {@link BytecodeModule}.
16   * @since 0.15.0
17   */
18  final class BytecodeModuleTest {
19  
20      @Test
21      void writesModuleToAsmClass() {
22          final BytecodeModule original = new BytecodeModule(
23              "name",
24              0,
25              "0.1.0",
26              "main",
27              Collections.singletonList("org.eolang.jeo"),
28              Collections.singletonList(new BytecodeModuleRequired("required", 0, "0.1.0")),
29              Collections.singletonList(
30                  new BytecodeModuleExported("exported", 1, Collections.singletonList("m1"))
31              ),
32              Collections.singletonList(
33                  new BytecodeModuleOpened("opened", 2, Collections.singletonList("m2"))
34              ),
35              Collections.singletonList(
36                  new BytecodeModuleProvided("provided", Collections.singletonList("impl"))
37              ),
38              Collections.singletonList("used")
39          );
40          final ClassNode node = new ClassNode();
41          original.write(node);
42          MatcherAssert.assertThat(
43              "We expect the module to be written to ASM class and read back correctly",
44              new AsmModule(node.module).bytecode(),
45              Matchers.equalTo(original)
46          );
47      }
48  }
