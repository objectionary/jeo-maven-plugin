5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.List;
8   import java.util.stream.Collectors;
9   import org.eolang.jeo.representation.bytecode.BytecodeMethodParameter;
10  import org.eolang.jeo.representation.bytecode.BytecodeMethodParameters;
11  import org.eolang.jeo.representation.bytecode.BytecodeParamAnnotations;
12  import org.eolang.jeo.representation.directives.JeoFqn;
13  
14  /**
15   * XML method params.
16   * @since 0.6
17   */
18  final class XmlMethodParams {
19  
20      /**
21       * Params fully qualified name.
22       */
23      private static final String PARAMS_BASE = new JeoFqn("params").fqn();
24  
25      /**
26       * Xml representation of a method params.
27       */
28      private final XmlJeoObject node;
29  
30      /**
31       * Constructor.
32       * @param node Xml representation of a method params.
33       */
34      XmlMethodParams(final XmlNode node) {
35          this(new XmlJeoObject(node));
36      }
37  
38      /**
39       * Constructor.
40       * @param node XML Jeo object node representing the method params.
41       */
42      private XmlMethodParams(final XmlJeoObject node) {
43          this.node = node;
44      }
45  
46      /**
47       * Is this node a method params?
48       * @return True if this node is a method params.
49       */
50      boolean isParams() {
51          return this.node.base()
52              .map(XmlMethodParams.PARAMS_BASE::equals)
53              .orElse(false);
54      }
55  
56      /**
57       * Get method params.
58       * @return Method params.
59       */
60      BytecodeMethodParameters params() {
61          return new BytecodeMethodParameters(
62              this.parameters(),
63              this.annotations()
64          );
65      }
66  
67      /**
68       * Method parameters.
69       * @return List of bytecode method parameters.
70       */
71      private List<BytecodeMethodParameter> parameters() {
72          return this.node.children()
73              .map(XmlMethodParam::new)
74              .filter(XmlMethodParam::isParam)
75              .map(XmlMethodParam::bytecode)
76              .collect(Collectors.toList());
77      }
78  
79      /**
80       * Annotations of the parameter.
81       * @return Annotations.
82       */
83      private List<BytecodeParamAnnotations> annotations() {
84          return this.node.children()
85              .map(XmlJeoObject::new)
86              .map(XmlParamAnnotations::new)
87              .filter(XmlParamAnnotations::isParamAnnotations)
88              .map(XmlParamAnnotations::bytecode)
89              .collect(Collectors.toList());
90      }
91  }
