5   package org.eolang.jeo;
6   
7   import java.nio.file.Path;
8   import java.nio.file.Paths;
9   import org.eolang.jeo.representation.PrefixedName;
10  import org.eolang.jeo.representation.XmirRepresentation;
11  
12  /**
13   * Assembling transformation.
14   * @since 0.6
15   */
16  public final class Assembling implements Transformation {
17  
18      /**
19       * Target folder where to save the assembled class.
20       */
21      private final Path folder;
22  
23      /**
24       * Representation to assemble.
25       */
26      private final Path from;
27  
28      /**
29       * XMIR representation.
30       */
31      private final XmirRepresentation repr;
32  
33      /**
34       * Constructor.
35       * @param target Target folder.
36       * @param representation Representation to assemble.
37       */
38      Assembling(final Path target, final Path representation) {
39          this.folder = target;
40          this.from = representation;
41          this.repr = new XmirRepresentation(this.from);
42      }
43  
44      @Override
45      public Path source() {
46          return this.from;
47      }
48  
49      @Override
50      public Path target() {
51          final String name = new PrefixedName(this.repr.name()).decode();
52          final String[] subpath = name.split("\\.");
53          subpath[subpath.length - 1] = String.format("%s.class", subpath[subpath.length - 1]);
54          return Paths.get(this.folder.toString(), subpath);
55      }
56  
57      @Override
58      public byte[] transform() {
59          return this.repr.toBytecode().bytes();
60      }
61  }
