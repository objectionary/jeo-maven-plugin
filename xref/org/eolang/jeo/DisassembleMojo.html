5   package org.eolang.jeo;
6   
7   import com.jcabi.log.Logger;
8   import java.io.File;
9   import java.util.Set;
10  import org.apache.maven.artifact.DependencyResolutionRequiredException;
11  import org.apache.maven.plugin.AbstractMojo;
12  import org.apache.maven.plugin.MojoExecutionException;
13  import org.apache.maven.plugins.annotations.LifecyclePhase;
14  import org.apache.maven.plugins.annotations.Mojo;
15  import org.apache.maven.plugins.annotations.Parameter;
16  import org.apache.maven.project.MavenProject;
17  import org.cactoos.set.SetOf;
18  import org.eolang.jeo.representation.directives.Format;
19  
20  /**
21   * Disassembles Java bytecode into XMIR representation.
22   *
23   * <p>This Maven plugin converts compiled Java class files into low-level EO representation
24   * (in XMIR format) that contains JVM opcodes and their operands. The resulting XMIR files
25   * preserve all bytecode instructions and can be assembled back into executable class files.</p>
26   *
27   * <p>The plugin supports different disassembly modes to control the level of detail in the
28   * output, including debug information such as line numbers and variable names.</p>
29   *
30   * @since 0.1.0
31   */
32  @Mojo(name = "disassemble", defaultPhase = LifecyclePhase.PROCESS_CLASSES, requiresProject = false)
33  public final class DisassembleMojo extends AbstractMojo {
34  
35      /**
36       * Maven project instance.
37       * <p>
38       * Provides access to project configuration and classpath dependencies required for
39       * bytecode analysis and disassembly.
40       * </p>
41       *
42       * @since 0.2.0
43       */
44      @Parameter(defaultValue = "${project}", readonly = true)
45      private MavenProject project;
46  
47      /**
48       * Source directory containing compiled Java class files.
49       * <p>
50       * This directory should contain {@code .class} files that will be disassembled into
51       * XMIR format. Typically points to the project's build output directory.
52       * </p>
53       *
54       * @since 0.2.0
55       * @checkstyle MemberNameCheck (6 lines)
56       */
57      @Parameter(
58          property = "jeo.disassemble.sourcesDir",
59          defaultValue = "${project.build.outputDirectory}"
60      )
61      private File sourcesDir;
62  
63      /**
64       * Target directory for generated XMIR files.
65       * <p>
66       * All disassembled XMIR files will be written to this directory, preserving the package
67       * structure of the original class files. Each class file will be converted to a corresponding
68       * XMIR file with {@code .xmir} extension.
69       * </p>
70       *
71       * @since 0.2.0
72       * @checkstyle MemberNameCheck (6 lines)
73       */
74      @Parameter(
75          property = "jeo.disassemble.outputDir",
76          defaultValue = "${project.build.directory}/generated-sources/jeo-xmir"
77      )
78      private File outputDir;
79  
80      /**
81       * Flag to disable the plugin execution.
82       * <p>
83       * When set to {@code true}, the plugin will skip all processing and exit immediately.
84       * This can be useful for conditional builds or troubleshooting.
85       * </p>
86       *
87       * @since 0.2.0
88       * @checkstyle MemberNameCheck (6 lines)
89       */
90      @Parameter(
91          property = "jeo.disassemble.disabled",
92          defaultValue = "false"
93      )
94      private boolean disabled;
95  
96      /**
97       * Disassembly mode controlling the level of detail in output.
98       * <p>
99       * Supported modes:
100      * <ul>
101      *   <li>{@code short} - Minimal output with bytecode instructions only</li>
102      *   <li>{@code debug} - Include debug information such as line numbers, local variables,
103      *       and source file references (default)</li>
104      * </ul>
105      * </p>
106      *
107      * @since 0.6.0
108      * @checkstyle MemberNameCheck (6 lines)
109      */
110     @Parameter(
111         property = "jeo.disassemble.mode",
112         defaultValue = "debug"
113     )
114     private String mode;
115 
116     /**
117      * Flag to omit detailed bytecode listings in generated XMIR.
118      * <p>
119      * When enabled, the {@code <listing>} element in XMIR files will not contain bytecode listing.
120      * This reduces file size and improves readability in production environments where detailed
121      * bytecode output is not needed. When disabled, full bytecode listings are included
122      * for debugging purposes.
123      * </p>
124      *
125      * @since 0.11.0
126      * @checkstyle MemberNameCheck (6 lines)
127      */
128     @Parameter(
129         property = "jeo.disassemble.omitListings",
130         defaultValue = "true"
131     )
132     private boolean omitListings;
133 
134     /**
135      * Flag to omit XML comments in generated XMIR files.
136      * <p>
137      * When enabled, no comments will be generated in the XMIR output, which can be
138      * useful for production builds where comments are not needed and may reduce file size.
139      * When disabled, XML comments will be included to provide debugging information.
140      * </p>
141      *
142      * @since 0.11.0
143      * @checkstyle MemberNameCheck (6 lines)
144      */
145     @Parameter(
146         property = "jeo.disassemble.omitComments",
147         defaultValue = "true"
148     )
149     private boolean omitComments;
150 
151     /**
152      * Flag to enable pretty-printing of XMIR files.
153      * <p>
154      *     When enabled, the generated XMIR files will be formatted with indentation (2 spaces) and
155      *     line breaks for better readability.
156      *     This is useful for development and debugging purposes.
157      *     By default, pretty-printing is enabled, but it's best to disable it for large
158      *     projects or production builds to reduce file size and improve performance.
159      * </p>
160      * @since 0.11.0
161      * @checkstyle MemberNameCheck (6 lines)
162      */
163     @Parameter(
164         property = "jeo.disassemble.prettyXmir",
165         defaultValue = "true"
166     )
167     private boolean prettyXmir;
168 
169     /**
170      * Flag to enable XMIR verification after disassembling.
171      * <p>
172      * When enabled, verifies all generated XMIR files for structural integrity and correctness
173      * after disassembly. If any XMIR file is invalid or corrupted, the build process will fail.
174      * This verification is disabled by default for performance.
175      * </p>
176      *
177      * @since 0.8.0
178      * @checkstyle MemberNameCheck (6 lines)
179      */
180     @Parameter(
181         property = "jeo.disassemble.xmir.verification",
182         defaultValue = "false"
183     )
184     private boolean xmirVerification;
185 
186     /**
187      * Should method modifiers be included in the output.
188      * <p>
189      * When true, method modifiers (e.g., public, private, static) will be
190      * included in the disassembled output.
191      * </p>
192      *
193      * @since 0.14.0
194      * @checkstyle MemberNameCheck (6 lines)
195      */
196     @Parameter(
197         property = "jeo.disassemble.xmir.modifiers",
198         defaultValue = "false"
199     )
200     private boolean modifiers;
201 
202     /**
203      * Set of inclusion GLOB filters for finding .class files
204      * in the {@link #sourcesDir} directory.
205      *
206      * @since 0.13.0
207      * @checkstyle MemberNameCheck (15 lines)
208      */
209     @Parameter(property = "jeo.disassemble.includes")
210     @SuppressWarnings("PMD.ImmutableField")
211     private Set<String> includes = new SetOf<>("**/*.class");
212 
213     /**
214      * Set of exclusion GLOB filters for finding .class files
215      * in the {@link #sourcesDir} directory.
216      *
217      * @since 0.13.0
218      * @checkstyle MemberNameCheck (7 lines)
219      */
220     @Parameter(property = "jeo.disassemble.excludes")
221     @SuppressWarnings("PMD.ImmutableField")
222     private Set<String> excludes = new SetOf<>();
223 
224     @Override
225     public void execute() throws MojoExecutionException {
226         try {
227             new PluginStartup(this.project, this.sourcesDir.toPath()).init();
228             if (this.disabled) {
229                 Logger.info(this, "Disassemble mojo is disabled, skipping");
230             } else {
231                 final boolean listings = !this.omitListings;
232                 final boolean comments = !this.omitComments;
233                 Logger.info(
234                     this,
235                     "Disassembling is started with mode '%s' (with listings = '%b', comments = '%b', modifiers = '%b', pretty = '%b')",
236                     this.mode,
237                     listings,
238                     comments,
239                     this.modifiers,
240                     this.prettyXmir
241                 );
242                 new Disassembler(
243                     new FilteredClasses(
244                         new BytecodeClasses(this.sourcesDir.toPath()),
245                         new GlobFilter(this.includes, this.excludes)
246                     ),
247                     this.outputDir.toPath(),
248                     new Format(
249                         Format.MODIFIERS, this.modifiers,
250                         Format.COMMENTS, comments,
251                         Format.WITH_LISTING, listings,
252                         Format.PRETTY, this.prettyXmir,
253                         Format.MODE, this.mode
254                     )
255                 ).disassemble();
256                 if (this.xmirVerification) {
257                     Logger.info(this, "Verifying all the XMIR files after disassembling");
258                     new XmirFiles(this.outputDir.toPath()).verify();
259                 } else {
260                     Logger.info(
261                         this, "XMIR verification after disassembling is disabled, skipping"
262                     );
263                 }
264             }
265         } catch (final DependencyResolutionRequiredException exception) {
266             throw new MojoExecutionException(
267                 String.format(
268                     "Failed to transpile bytecode to EO, from '%s' to '%s'",
269                     this.sourcesDir.toPath(),
270                     this.outputDir.toPath()
271                 ),
272                 exception
273             );
274         }
275     }
276 }
