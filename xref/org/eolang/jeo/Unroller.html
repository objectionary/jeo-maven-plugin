24  package org.eolang.jeo;
25  
26  import com.jcabi.log.Logger;
27  import java.io.FileNotFoundException;
28  import java.io.IOException;
29  import java.nio.charset.StandardCharsets;
30  import java.nio.file.Files;
31  import java.nio.file.Path;
32  import java.util.stream.Stream;
33  import org.eolang.jeo.representation.CanonicalXmir;
34  
35  /**
36   * Unroller.
37   * This class unrolls all the formatting changes made by `phi/unphi` transformations.
38   * In other words, it makes XMIR understandable by jeo-maven-plugin after `phi/unphi`
39   * transformations.
40   * @since 0.6
41   */
42  final class Unroller {
43  
44      /**
45       * Source directory with XMIR files that were changed by `phi/unphi` transformations.
46       */
47      private final Path source;
48  
49      /**
50       * Target directory where unrolled XMIR files will be saved.
51       */
52      private final Path target;
53  
54      /**
55       * Constructor.
56       * @param source Directory with XMIR files that were changed by `phi/unphi` transformations.
57       * @param target Target directory where unrolled XMIR files will be saved.
58       */
59      Unroller(final Path source, final Path target) {
60          this.source = source;
61          this.target = target;
62      }
63  
64      /**
65       * Unrolls all the XMIR files from the source directory to the target directory.
66       * @return The number of unrolled XMIR files.
67       */
68      long unroll() {
69          try (Stream<Path> xmirs = Files.walk(this.source)) {
70              return xmirs.filter(Unroller::isXmir)
71                  .parallel()
72                  .peek(this::unroll)
73                  .count();
74          } catch (final IOException exception) {
75              throw new IllegalStateException(
76                  String.format(
77                      "Failed to unroll XMIR files from '%s' directory to '%s'",
78                      this.source,
79                      this.target
80                  ),
81                  exception
82              );
83          }
84      }
85  
86      /**
87       * Unrolls the XMIR file.
88       * @param path The path to the XMIR file.
89       */
90      private void unroll(final Path path) {
91          try {
92              this.prepareThread();
93              final Path output = this.target.resolve(this.source.relativize(path));
94              Logger.info(this, "Unrolling XMIR file '%s' to '%s'", path, output);
95              Files.createDirectories(output.getParent());
96              Files.write(
97                  output,
98                  new CanonicalXmir(path)
99                      .plain()
100                     .toString()
101                     .getBytes(StandardCharsets.UTF_8)
102             );
103         } catch (final FileNotFoundException exception) {
104             throw new IllegalStateException(
105                 String.format(
106                     "Failed to unroll XMIR file '%s', most probably the file does not exist",
107                     path
108                 ),
109                 exception
110             );
111         } catch (final IOException exception) {
112             throw new IllegalStateException(
113                 String.format(
114                     "Failed to unroll XMIR file '%s', most probably an I/O error occurred",
115                     path
116                 ),
117                 exception
118             );
119         }
120     }
121 
122     /**
123      * Prepares the thread for unrolling.
124      * ATTENTION! DO NOT REMOVE THIS METHOD!
125      * This method was added to solve the problem with class loading during the unrolling process.
126      * Since we unroll files in parallel, we need to set the context class loader for each thread.
127      * Otherwise, the class loader won't be able to find `net.sf.saxon.TransformerFactoryImpl`.
128      * This method is a workaround for the problem with class loading during the unrolling process.
129      */
130     @SuppressWarnings("PMD.UseProperClassLoader")
131     private void prepareThread() {
132         System.setProperty(
133             "javax.xml.transform.TransformerFactory",
134             "net.sf.saxon.TransformerFactoryImpl"
135         );
136         Thread.currentThread().setContextClassLoader(this.getClass().getClassLoader());
137     }
138 
139     /**
140      * Checks if the file is an XMIR file.
141      * @param path The path to the file.
142      * @return True if the file is an XMIR file, false otherwise.
143      */
144     private static boolean isXmir(final Path path) {
145         return Files.isRegularFile(path) && path.toString().endsWith(".xmir");
146     }
147 }
