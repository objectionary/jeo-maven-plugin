5   package org.eolang.jeo.representation;
6   
7   /**
8    * Method name converter between Java bytecode and EO representations.
9    *
10   * <p>This class handles the conversion of Java method names, particularly
11   * special cases like constructors and static initializers. Constructor names
12   * are represented as '&lt;init&gt;' in bytecode which isn't allowed in EO,
13   * so they need to be converted to valid EO names and back.</p>
14   * @since 0.6.0
15   */
16  public final class MethodName {
17  
18      /**
19       * Constructor name in EO.
20       */
21      private static final String EO_CONSTRUCTOR = "object@init@";
22  
23      /**
24       * Constructor name in bytecode.
25       */
26      private static final String CONSTRUCTOR = "<init>";
27  
28      /**
29       * Static initializer name.
30       */
31      private static final String STATIC = "<clinit>";
32  
33      /**
34       * Static initializer name in EO.
35       */
36      private static final String EO_STATIC = "class@clinit@";
37  
38      /**
39       * Original name.
40       */
41      private final String original;
42  
43      /**
44       * Constructor.
45       * @param name The method name (in either bytecode or EO format)
46       */
47      public MethodName(final String name) {
48          this.original = name;
49      }
50  
51      /**
52       * Convert method name to bytecode format.
53       * @return The method name in bytecode format
54       */
55      public String bytecode() {
56          final String result;
57          switch (this.original) {
58              case MethodName.EO_CONSTRUCTOR:
59                  result = MethodName.CONSTRUCTOR;
60                  break;
61              case MethodName.EO_STATIC:
62                  result = MethodName.STATIC;
63                  break;
64              default:
65                  result = this.original;
66                  break;
67          }
68          return result;
69      }
70  
71      /**
72       * Convert method name to XMIR/EO format.
73       * @return The method name in XMIR/EO format
74       */
75      public String xmir() {
76          final String result;
77          switch (this.original) {
78              case MethodName.CONSTRUCTOR:
79                  result = MethodName.EO_CONSTRUCTOR;
80                  break;
81              case MethodName.STATIC:
82                  result = MethodName.EO_STATIC;
83                  break;
84              default:
85                  result = this.original;
86                  break;
87          }
88          return result;
89      }
90  }
