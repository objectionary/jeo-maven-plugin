27  import java.nio.file.Path;
28  import java.util.Arrays;
29  import lombok.ToString;
30  import org.cactoos.Input;
31  import org.cactoos.bytes.BytesOf;
32  import org.cactoos.bytes.UncheckedBytes;
33  import org.cactoos.io.InputOf;
34  import org.cactoos.scalar.Sticky;
35  import org.cactoos.scalar.Synced;
36  import org.cactoos.scalar.Unchecked;
37  import org.eolang.jeo.Details;
38  import org.eolang.jeo.Representation;
39  import org.eolang.jeo.representation.asm.AsmProgram;
40  import org.eolang.jeo.representation.bytecode.Bytecode;
41  import org.eolang.jeo.representation.directives.DirectivesProgram;
42  import org.objectweb.asm.ClassReader;
43  import org.xembly.ImpossibleModificationException;

45  /**
46   * Intermediate representation of a class files which can be optimized from bytecode.
47   * You can also use that site to check if bytecode is correct:
48   * <a href="https://godbolt.org">https://godbolt.org/</a>
49   *
50   * @since 0.1.0
51   */
52  @ToString
53  @SuppressWarnings("PMD.UseObjectForClearerAPI")
54  public final class BytecodeRepresentation implements Representation {
55  
56      /**
57       * Input source.
58       */
59      private final Unchecked<byte[]> input;
60  
61      /**
62       * The source of the input.
63       */
64      private final String source;
65  
66      /**
67       * Constructor.
68       *
69       * @param clazz Path to the class file
70       */
71      public BytecodeRepresentation(final Path clazz) {
72          this(BytecodeRepresentation.fromFile(clazz), String.valueOf(clazz));
73      }
74  
75      /**
76       * Constructor.
77       *
78       * @param bytecode Bytecode
79       */
80      public BytecodeRepresentation(final Bytecode bytecode) {
81          this(BytecodeRepresentation.fromBytes(bytecode.bytes()), "bytecode");
82      }
83  
84      /**
85       * Constructor.
86       *
87       * @param input Input source
88       */
89      BytecodeRepresentation(final Input input) {
90          this(BytecodeRepresentation.fromInput(input), "bytecode");
91      }
92  
93      /**
94       * Constructor.
95       * @param input Input.
96       * @param source The source of the input.
97       */
98      public BytecodeRepresentation(
99          final Unchecked<byte[]> input,
100         final String source
101     ) {
102         this.input = input;
103         this.source = source;
104     }
105 
106     @Override
107     public Details details() {
108         return new Details(this.className(), this.source);
109     }
110 
111     @Override
112     public XML toEO() {
113         return this.toEO(true);
114     }
115 
116     /**
117      * Converts bytecode into XML.
118      * @param count Do we add number to opcode name or not?
119      * @return XML representation of bytecode.
120      */
121     public XML toEO(final boolean count) {
122         final DirectivesProgram directives = new AsmProgram(this.input.value())
123             .bytecode()
124             .directives(new Base64Bytecode(this.input.value()).asString(), count);
125         try {
126             return new VerifiedEo(directives).asXml();
127         } catch (final IllegalStateException exception) {
128             throw new IllegalStateException(
129                 String.format(
130                     "Something went wrong during transformation %s into XML by using directives %n%s%n",
131                     this.className(),
132                     directives
133                 ),
134                 exception
135             );
136         } catch (final ImpossibleModificationException exception) {
137             throw new IllegalStateException(
138                 String.format(
139                     "Can't build XML from %s by using directives %s",
140                     Arrays.toString(this.input.value()),
141                     directives
142                 ),
143                 exception
144             );
145         }
146     }
147 
148     @Override
149     public Bytecode toBytecode() {
150         return new Bytecode(new UncheckedBytes(new BytesOf(this.input.value())).asBytes());
151     }
152 
153     /**
154      * Read class name from bytecode.
155      *
156      * @return Class name.
157      */
158     private String className() {
159         final ClassNameVisitor name = new ClassNameVisitor();
160         new ClassReader(this.input.value()).accept(name, 0);
161         return name.asString();
162     }
163 
164     /**
165      * Prestructor that converts a file to a byte source.
166      * @param path Path to the file.
167      * @return Byte source.
168      */
169     private static Unchecked<byte[]> fromFile(final Path path) {
170         return BytecodeRepresentation.fromInput(new InputOf(path));
171     }
172 
173     /**
174      * Prestructor that converts input to a byte source.
175      * @param input Input.
176      * @return Byte source.
177      */
178     private static Unchecked<byte[]> fromInput(final Input input) {
179         return new Unchecked<>(new Synced<>(new Sticky<>(() -> new BytesOf(input).asBytes())));
180     }
181 
182     /**
183      * Prestructor that converts bytes to a byte source.
184      * @param bytes Bytes.
185      * @return Byte source.
186      */
187     private static Unchecked<byte[]> fromBytes(final byte[] bytes) {
188         return new Unchecked<>(new Synced<>(new Sticky<>(() -> bytes)));
189     }
190 }
