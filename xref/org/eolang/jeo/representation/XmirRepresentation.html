5   package org.eolang.jeo.representation;
6   
7   import com.jcabi.xml.XML;
8   import java.nio.file.Path;
9   import org.eolang.jeo.representation.bytecode.Bytecode;
10  import org.eolang.jeo.representation.xmir.JcabiXmlDoc;
11  import org.eolang.jeo.representation.xmir.XmlDoc;
12  import org.eolang.jeo.representation.xmir.XmlNode;
13  import org.eolang.jeo.representation.xmir.XmlObject;
14  
15  /**
16   * Intermediate representation of class files from XMIR format.
17   *
18   * <p>This class provides a unified interface for working with XMIR (EO XML representation)
19   * files. It can read XMIR from files or XML objects and convert them to Java bytecode.</p>
20   * @since 0.1.0
21   */
22  public final class XmirRepresentation {
23  
24      /**
25       * XML.
26       */
27      private final XmlDoc xml;
28  
29      /**
30       * Source of the XML.
31       */
32      private final String source;
33  
34      /**
35       * Constructor.
36       * @param path The path to XMIR file to read
37       */
38      public XmirRepresentation(final Path path) {
39          this(new JcabiXmlDoc(path), path.toAbsolutePath().toString());
40      }
41  
42      /**
43       * Constructor.
44       * @param xml The XML object containing XMIR representation
45       */
46      public XmirRepresentation(final XML xml) {
47          this(new JcabiXmlDoc(xml), "Unknown");
48      }
49  
50      /**
51       * Constructor.
52       * @param xml The XML document wrapper
53       * @param source The source description for error reporting
54       */
55      private XmirRepresentation(final XmlDoc xml, final String source) {
56          this.xml = xml;
57          this.source = source;
58      }
59  
60      /**
61       * Retrieves class name from XMIR.
62       * <p>This method intentionally uses classes from {@code org.w3c.dom} instead of
63       * {@code com.jcabi.xml} for performance reasons.</p>
64       * @return Fully qualified class name
65       */
66      public String name() {
67          final XmlNode root = this.xml.root();
68          return new ClassName(
69              root.xpath("/object/metas/meta[head[text()]='package']/tail/text()")
70                  .stream()
71                  .findFirst()
72                  .orElse(""),
73              root.xpath("/object/o/@name").get(0)
74          ).full();
75      }
76  
77      /**
78       * Convert to bytecode.
79       * @return Bytecode object containing the compiled class
80       */
81      public Bytecode toBytecode() {
82          try {
83              return new XmlObject(this.xml.root()).bytecode().bytecode();
84          } catch (final IllegalArgumentException exception) {
85              throw new IllegalArgumentException(
86                  String.format("Can't transform '%s' to bytecode", this.xml),
87                  exception
88              );
89          } catch (final IllegalStateException exception) {
90              throw new IllegalStateException(
91                  String.format("Can't transform XMIR to bytecode from the '%s' source", this.source),
92                  exception
93              );
94          }
95      }
96  }
