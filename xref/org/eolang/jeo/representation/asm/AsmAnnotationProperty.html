5   package org.eolang.jeo.representation.asm;
6   
7   import java.util.Collection;
8   import java.util.List;
9   import java.util.Optional;
10  import java.util.stream.Collectors;
11  import java.util.stream.Stream;
12  import org.eolang.jeo.representation.bytecode.BytecodeAnnotationAnnotationValue;
13  import org.eolang.jeo.representation.bytecode.BytecodeAnnotationValue;
14  import org.eolang.jeo.representation.bytecode.BytecodeArrayAnnotationValue;
15  import org.eolang.jeo.representation.bytecode.BytecodeEnumAnnotationValue;
16  import org.eolang.jeo.representation.bytecode.BytecodePlainAnnotationValue;
17  import org.objectweb.asm.tree.AnnotationNode;
18  
19  /**
20   * Asm annotation property.
21   * @since 0.6
22   */
23  final class AsmAnnotationProperty {
24  
25      /**
26       * Property name.
27       */
28      private final String name;
29  
30      /**
31       * Property value.
32       */
33      private final Object value;
34  
35      /**
36       * Constructor.
37       * @param value Property value.
38       */
39      AsmAnnotationProperty(final Object value) {
40          this(null, value);
41      }
42  
43      /**
44       * Constructor.
45       * @param name Property name.
46       * @param value Property value.
47       */
48      AsmAnnotationProperty(final String name, final Object value) {
49          this.name = name;
50          this.value = value;
51      }
52  
53      /**
54       * Convert asm annotation property to domain annotation property.
55       * @return Domain annotation.
56       */
57      BytecodeAnnotationValue bytecode() {
58          return AsmAnnotationProperty.property(this.name, this.value);
59      }
60  
61      /**
62       * Convert asm annotation property to domain annotation property.
63       * @param name Property name.
64       * @param value Property value.
65       * @return Domain annotation.
66       */
67      private static BytecodeAnnotationValue property(final String name, final Object value) {
68          final BytecodeAnnotationValue result;
69          if (value instanceof String[]) {
70              final String[] params = (String[]) value;
71              result = new BytecodeEnumAnnotationValue(name, params[0], params[1]);
72          } else if (value instanceof AnnotationNode) {
73              final AnnotationNode cast = AnnotationNode.class.cast(value);
74              result = new BytecodeAnnotationAnnotationValue(
75                  name,
76                  cast.desc,
77                  Optional.ofNullable(cast.values)
78                      .map(Collection::stream)
79                      .orElseGet(Stream::empty)
80                      .map(val -> AsmAnnotationProperty.property("", val))
81                      .collect(Collectors.toList())
82              );
83          } else if (value instanceof List) {
84              result = new BytecodeArrayAnnotationValue(
85                  name,
86                  ((Collection<?>) value).stream()
87                      .map(val -> AsmAnnotationProperty.property("", val))
88                      .collect(Collectors.toList())
89              );
90          } else {
91              result = new BytecodePlainAnnotationValue(name, value);
92          }
93          return result;
94      }
95  
96  }
