5   package org.eolang.jeo.representation.asm;
6   
7   import java.util.Collections;
8   import java.util.List;
9   import java.util.Optional;
10  import java.util.stream.Collectors;
11  import org.eolang.jeo.representation.bytecode.BytecodeRecordComponent;
12  import org.objectweb.asm.tree.RecordComponentNode;
13  
14  /**
15   * Asm record components.
16   * @since 0.15.0
17   */
18  final class AsmRecordComponents {
19  
20      /**
21       * Nodes of record components.
22       */
23      private final List<RecordComponentNode> nodes;
24  
25      /**
26       * Constructor.
27       * @param components List of record component nodes
28       */
29      AsmRecordComponents(final List<RecordComponentNode> components) {
30          this.nodes = Optional.ofNullable(components).orElse(Collections.emptyList());
31      }
32  
33      /**
34       * Bytecode record components.
35       * @return List of bytecode record components
36       */
37      public List<BytecodeRecordComponent> bytecode() {
38          return this.nodes.stream()
39              .map(
40                  comp -> new BytecodeRecordComponent(
41                      comp.name,
42                      comp.descriptor,
43                      comp.signature,
44                      new AsmAnnotations(comp).bytecode(),
45                      new AsmTypeAnnotations(comp).bytecode()
46                  )
47              ).collect(Collectors.toList());
48      }
49  }
