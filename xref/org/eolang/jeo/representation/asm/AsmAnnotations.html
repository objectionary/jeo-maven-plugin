24  package org.eolang.jeo.representation.asm;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  import java.util.Optional;
29  import java.util.stream.Collectors;
30  import java.util.stream.Stream;
31  import org.eolang.jeo.representation.bytecode.BytecodeAnnotation;
32  import org.eolang.jeo.representation.bytecode.BytecodeAnnotationValue;
33  import org.eolang.jeo.representation.bytecode.BytecodeAnnotations;
34  import org.objectweb.asm.tree.AnnotationNode;
35  import org.objectweb.asm.tree.ClassNode;
36  import org.objectweb.asm.tree.FieldNode;
37  import org.objectweb.asm.tree.MethodNode;
38  
39  /**
40   * Asm annotations.
41   * Asm parser for annotations.
42   * @since 0.6
43   */
44  public final class AsmAnnotations {
45  
46      /**
47       * Visible annotations.
48       */
49      private final List<AnnotationNode> visible;
50  
51      /**
52       * Invisible annotations.
53       */
54      private final List<AnnotationNode> invisible;
55  
56      /**
57       * Constructor.
58       * @param node Class node.
59       */
60      AsmAnnotations(final ClassNode node) {
61          this(node.visibleAnnotations, node.invisibleAnnotations);
62      }
63  
64      /**
65       * Constructor.
66       * @param node Method node.
67       */
68      AsmAnnotations(final MethodNode node) {
69          this(node.visibleAnnotations, node.invisibleAnnotations);
70      }
71  
72      /**
73       * Constructor.
74       * @param node Field node.
75       */
76      AsmAnnotations(final FieldNode node) {
77          this(node.visibleAnnotations, node.invisibleAnnotations);
78      }
79  
80      /**
81       * Constructor.
82       * @param visible Visible annotations.
83       * @param invisible Invisible annotations.
84       */
85      AsmAnnotations(
86          final List<AnnotationNode> visible,
87          final List<AnnotationNode> invisible
88      ) {
89          this.visible = visible;
90          this.invisible = invisible;
91      }
92  
93      /**
94       * Convert asm annotations to domain annotations.
95       * @return Domain annotations.
96       */
97      public BytecodeAnnotations bytecode() {
98          return new BytecodeAnnotations(
99              Stream.concat(
100                 AsmAnnotations.safe(this.visible, true),
101                 AsmAnnotations.safe(this.invisible, false)
102             ).collect(Collectors.toList())
103         );
104     }
105 
106     /**
107      * Safe annotations.
108      * @param nodes Annotation nodes.
109      * @param visible Is it visible?
110      * @return Annotations.
111      */
112     private static Stream<BytecodeAnnotation> safe(
113         final List<AnnotationNode> nodes, final boolean visible
114     ) {
115         return Optional.ofNullable(nodes)
116             .orElse(new ArrayList<>(0))
117             .stream()
118             .map(ann -> AsmAnnotations.annotation(ann, visible));
119     }
120 
121     /**
122      * Convert asm annotation to domain annotation.
123      * @param node Asm annotation node.
124      * @param visible Is it visible?
125      * @return Domain annotation.
126      */
127     private static BytecodeAnnotation annotation(final AnnotationNode node, final boolean visible) {
128         final List<BytecodeAnnotationValue> properties = new ArrayList<>(0);
129         final List<Object> values = Optional.ofNullable(node.values).orElse(new ArrayList<>(0));
130         for (int index = 0; index < values.size(); index += 2) {
131             properties.add(
132                 new AsmAnnotationProperty(
133                     (String) values.get(index),
134                     values.get(index + 1)
135                 ).bytecode()
136             );
137         }
138         return new BytecodeAnnotation(node.desc, visible, properties);
139     }
140 }
