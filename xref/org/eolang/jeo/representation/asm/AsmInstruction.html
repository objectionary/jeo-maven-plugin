5   package org.eolang.jeo.representation.asm;
6   
7   import java.util.Arrays;
8   import java.util.stream.Stream;
9   import org.eolang.jeo.representation.bytecode.BytecodeEntry;
10  import org.eolang.jeo.representation.bytecode.BytecodeFrame;
11  import org.eolang.jeo.representation.bytecode.BytecodeInstruction;
12  import org.eolang.jeo.representation.bytecode.BytecodeLabel;
13  import org.eolang.jeo.representation.bytecode.BytecodeLine;
14  import org.objectweb.asm.Label;
15  import org.objectweb.asm.tree.AbstractInsnNode;
16  import org.objectweb.asm.tree.FieldInsnNode;
17  import org.objectweb.asm.tree.FrameNode;
18  import org.objectweb.asm.tree.IincInsnNode;
19  import org.objectweb.asm.tree.IntInsnNode;
20  import org.objectweb.asm.tree.InvokeDynamicInsnNode;
21  import org.objectweb.asm.tree.JumpInsnNode;
22  import org.objectweb.asm.tree.LabelNode;
23  import org.objectweb.asm.tree.LdcInsnNode;
24  import org.objectweb.asm.tree.LineNumberNode;
25  import org.objectweb.asm.tree.LookupSwitchInsnNode;
26  import org.objectweb.asm.tree.MethodInsnNode;
27  import org.objectweb.asm.tree.MultiANewArrayInsnNode;
28  import org.objectweb.asm.tree.TableSwitchInsnNode;
29  import org.objectweb.asm.tree.TypeInsnNode;
30  import org.objectweb.asm.tree.VarInsnNode;
31  
32  /**
33   * Asm instruction.
34   * @since 0.6
35   */
36  final class AsmInstruction {
37  
38      /**
39       * Instruction node.
40       */
41      private final AbstractInsnNode node;
42  
43      /**
44       * Constructor.
45       * @param node Instruction node.
46       */
47      AsmInstruction(final AbstractInsnNode node) {
48          this.node = node;
49      }
50  
51      /**
52       * Convert asm instruction to domain instruction.
53       * @return Domain instruction.
54       * @checkstyle CyclomaticComplexityCheck (100 lines)
55       * @checkstyle JavaNCSSCheck (100 lines)
56       * @checkstyle MethodLengthCheck (200 lines)
57       */
58      @SuppressWarnings({"PMD.NcssCount", "PMD.ExcessiveMethodLength"})
59      BytecodeEntry bytecode() {
60          final BytecodeEntry result;
61          switch (this.node.getType()) {
62              case AbstractInsnNode.INSN:
63                  result = new BytecodeInstruction(this.node.getOpcode());
64                  break;
65              case AbstractInsnNode.INT_INSN:
66                  final IntInsnNode instr = IntInsnNode.class.cast(this.node);
67                  result = new BytecodeInstruction(
68                      instr.getOpcode(),
69                      instr.operand
70                  );
71                  break;
72              case AbstractInsnNode.VAR_INSN:
73                  final VarInsnNode varinstr = VarInsnNode.class.cast(this.node);
74                  result = new BytecodeInstruction(
75                      varinstr.getOpcode(), varinstr.var
76                  );
77                  break;
78              case AbstractInsnNode.TYPE_INSN:
79                  final TypeInsnNode typeinstr = TypeInsnNode.class.cast(this.node);
80                  result = new BytecodeInstruction(
81                      typeinstr.getOpcode(),
82                      typeinstr.desc
83                  );
84                  break;
85              case AbstractInsnNode.FIELD_INSN:
86                  final FieldInsnNode field = FieldInsnNode.class.cast(this.node);
87                  result = new BytecodeInstruction(
88                      field.getOpcode(),
89                      field.owner,
90                      field.name,
91                      field.desc
92                  );
93                  break;
94              case AbstractInsnNode.METHOD_INSN:
95                  final MethodInsnNode method = MethodInsnNode.class.cast(this.node);
96                  result = new BytecodeInstruction(
97                      method.getOpcode(),
98                      method.owner,
99                      method.name,
100                     method.desc,
101                     method.itf
102                 );
103                 break;
104             case AbstractInsnNode.INVOKE_DYNAMIC_INSN:
105                 final InvokeDynamicInsnNode dynamic = InvokeDynamicInsnNode.class.cast(this.node);
106                 result = new BytecodeInstruction(
107                     dynamic.getOpcode(),
108                     Stream.concat(
109                         Stream.of(
110                             dynamic.name,
111                             dynamic.desc,
112                             dynamic.bsm
113                         ),
114                         Arrays.stream(dynamic.bsmArgs)
115                     ).toArray(Object[]::new)
116                 );
117                 break;
118             case AbstractInsnNode.JUMP_INSN:
119                 final JumpInsnNode jump = JumpInsnNode.class.cast(this.node);
120                 result = new BytecodeInstruction(
121                     jump.getOpcode(),
122                     new BytecodeLabel(jump.label.getLabel().toString())
123                 );
124                 break;
125             case AbstractInsnNode.LABEL:
126                 final LabelNode label = LabelNode.class.cast(this.node);
127                 result = new BytecodeLabel(label.getLabel().toString());
128                 break;
129             case AbstractInsnNode.LDC_INSN:
130                 final LdcInsnNode ldc = LdcInsnNode.class.cast(this.node);
131                 result = new BytecodeInstruction(
132                     ldc.getOpcode(),
133                     ldc.cst
134                 );
135                 break;
136             case AbstractInsnNode.IINC_INSN:
137                 final IincInsnNode iinc = IincInsnNode.class.cast(this.node);
138                 result = new BytecodeInstruction(
139                     iinc.getOpcode(),
140                     iinc.var,
141                     iinc.incr
142                 );
143                 break;
144             case AbstractInsnNode.TABLESWITCH_INSN:
145                 final TableSwitchInsnNode table = TableSwitchInsnNode.class.cast(this.node);
146                 result = new BytecodeInstruction(
147                     table.getOpcode(),
148                     Stream.concat(
149                         Stream.of(
150                             table.min,
151                             table.max,
152                             new BytecodeLabel(table.dflt.getLabel().toString())
153                         ),
154                         table.labels
155                             .stream()
156                             .map(LabelNode::getLabel)
157                             .map(Label::toString)
158                             .map(BytecodeLabel::new)
159                     ).toArray(Object[]::new)
160                 );
161                 break;
162             case AbstractInsnNode.LOOKUPSWITCH_INSN:
163                 final LookupSwitchInsnNode lookup = LookupSwitchInsnNode.class.cast(this.node);
164                 result = new BytecodeInstruction(
165                     lookup.getOpcode(),
166                     Stream.concat(
167                         Stream.of(new BytecodeLabel(lookup.dflt.getLabel().toString())),
168                         Stream.concat(
169                             lookup.keys.stream(),
170                             lookup.labels.stream()
171                                 .map(LabelNode::getLabel)
172                                 .map(Label::toString)
173                                 .map(BytecodeLabel::new)
174                         )
175                     ).toArray(Object[]::new)
176                 );
177                 break;
178             case AbstractInsnNode.MULTIANEWARRAY_INSN:
179                 final MultiANewArrayInsnNode multiarr = MultiANewArrayInsnNode.class.cast(
180                     this.node
181                 );
182                 result = new BytecodeInstruction(
183                     multiarr.getOpcode(),
184                     multiarr.desc,
185                     multiarr.dims
186                 );
187                 break;
188             case AbstractInsnNode.FRAME:
189                 final FrameNode frame = FrameNode.class.cast(this.node);
190                 result = new BytecodeFrame(
191                     frame.type,
192                     frame.local,
193                     frame.stack
194                 );
195                 break;
196             case AbstractInsnNode.LINE:
197                 final LineNumberNode line = LineNumberNode.class.cast(this.node);
198                 result = new BytecodeLine(
199                     line.line,
200                     new BytecodeLabel(LabelNode.class.cast(line.start).getLabel().toString())
201                 );
202                 break;
203             default:
204                 throw new IllegalStateException(
205                     String.format("Unknown instruction type: %s", this.node)
206                 );
207         }
208         return result;
209     }
210 }
