5   package org.eolang.jeo.representation.asm;
6   
7   import java.util.List;
8   import java.util.Optional;
9   import java.util.stream.Collectors;
10  import java.util.stream.Stream;
11  import org.eolang.jeo.representation.ClassName;
12  import org.eolang.jeo.representation.bytecode.BytecodeAttribute;
13  import org.eolang.jeo.representation.bytecode.BytecodeAttributes;
14  import org.eolang.jeo.representation.bytecode.BytecodeClass;
15  import org.eolang.jeo.representation.bytecode.BytecodeClassProperties;
16  import org.eolang.jeo.representation.bytecode.BytecodeField;
17  import org.eolang.jeo.representation.bytecode.BytecodeMethod;
18  import org.eolang.jeo.representation.bytecode.InnerClass;
19  import org.objectweb.asm.tree.ClassNode;
20  import org.objectweb.asm.tree.ModuleNode;
21  
22  /**
23   * ASM-based bytecode parser for Java classes.
24   *
25   * <p>This class provides functionality to parse ASM ClassNode objects and convert
26   * them into domain-specific bytecode representations. It handles extraction of
27   * class properties, methods, fields, and annotations from ASM's internal structure.</p>
28   * @since 0.6.0
29   */
30  @SuppressWarnings("PMD.TooManyMethods")
31  public final class AsmClass {
32  
33      /**
34       * Class node.
35       */
36      private final ClassNode node;
37  
38      /**
39       * Constructor.
40       * @param node The ASM class node to parse
41       */
42      AsmClass(final ClassNode node) {
43          this.node = node;
44      }
45  
46      /**
47       * Convert ASM class to domain bytecode class.
48       * @return The domain bytecode class representation
49       */
50      public BytecodeClass bytecode() {
51          final ClassName full = new ClassName(this.node.name);
52          return new BytecodeClass(
53              full,
54              this.methods(),
55              this.fields(),
56              new AsmAnnotations(this.node).bytecode(),
57              this.attributes(),
58              new BytecodeClassProperties(
59                  this.node.version,
60                  this.node.access,
61                  this.node.signature,
62                  this.node.superName,
63                  this.node.interfaces.toArray(new String[0])
64              )
65          );
66      }
67  
68      /**
69       * Convert ASM fields to domain fields.
70       * @return The list of domain field representations
71       */
72      private List<BytecodeField> fields() {
73          return this.node.fields.stream()
74              .map(AsmField::new)
75              .map(AsmField::bytecode)
76              .collect(Collectors.toList());
77      }
78  
79      /**
80       * Convert ASM methods to domain methods.
81       * @return The list of domain method representations
82       */
83      private List<BytecodeMethod> methods() {
84          return this.node.methods.stream()
85              .map(AsmMethod::new)
86              .map(AsmMethod::bytecode)
87              .collect(Collectors.toList());
88      }
89  
90      /**
91       * Retrieve domain attributes from ASM class.
92       * @return The domain attributes representation
93       */
94      private BytecodeAttributes attributes() {
95          return new BytecodeAttributes(
96              Stream.concat(
97                  Stream.of(
98                      this.source(),
99                      this.module(),
100                     this.enclosing(),
101                     this.nesthost(),
102                     this.nestmembers(),
103                     this.permitted(),
104                     this.records()
105                 ).filter(Optional::isPresent).map(Optional::get),
106                 Stream.concat(
107                     this.inners(),
108                     new AsmUnknownAttributes(this.node).bytecode().stream()
109                 )
110             ).collect(Collectors.toList())
111         );
112     }
113 
114     /**
115      * Source attribute.
116      * @return Source attribute if present
117      */
118     private Optional<BytecodeAttribute> source() {
119         final Optional<BytecodeAttribute> result;
120         if (this.node.sourceFile == null && this.node.sourceDebug == null) {
121             result = Optional.empty();
122         } else {
123             result = Optional.of(
124                 new BytecodeAttribute.SourceFile(
125                     this.node.sourceFile,
126                     this.node.sourceDebug
127                 )
128             );
129         }
130         return result;
131     }
132 
133     /**
134      * Module attribute.
135      * @return Module attribute if present
136      */
137     private Optional<BytecodeAttribute> module() {
138         final Optional<BytecodeAttribute> result;
139         final ModuleNode module = this.node.module;
140         if (module == null) {
141             result = Optional.empty();
142         } else {
143             result = Optional.of(new AsmModule(module).bytecode());
144         }
145         return result;
146     }
147 
148     /**
149      * Enclosing method attribute.
150      * @return Enclosing method attribute if present
151      */
152     private Optional<BytecodeAttribute> enclosing() {
153         final Optional<BytecodeAttribute> result;
154         if (this.node.outerClass == null
155             && this.node.outerMethod == null
156             && this.node.outerMethodDesc == null
157         ) {
158             result = Optional.empty();
159         } else {
160             result = Optional.of(
161                 new BytecodeAttribute.EnclosingMethod(
162                     this.node.outerClass,
163                     this.node.outerMethod,
164                     this.node.outerMethodDesc
165                 )
166             );
167         }
168         return result;
169     }
170 
171     /**
172      * Nest host attribute.
173      * @return Nest host attribute if present
174      */
175     private Optional<BytecodeAttribute> nesthost() {
176         final Optional<BytecodeAttribute> result;
177         if (this.node.nestHostClass == null) {
178             result = Optional.empty();
179         } else {
180             result = Optional.of(
181                 new BytecodeAttribute.NestHost(this.node.nestHostClass)
182             );
183         }
184         return result;
185     }
186 
187     /**
188      * Nest members attribute.
189      * @return Nest members attribute if present
190      */
191     private Optional<BytecodeAttribute> nestmembers() {
192         final Optional<BytecodeAttribute> result;
193         if (this.node.nestMembers == null) {
194             result = Optional.empty();
195         } else {
196             result = Optional.of(
197                 new BytecodeAttribute.NestMembers(this.node.nestMembers)
198             );
199         }
200         return result;
201     }
202 
203     /**
204      * Permitted subclasses attribute.
205      * @return Permitted subclasses attribute if present
206      */
207     private Optional<BytecodeAttribute> permitted() {
208         final Optional<BytecodeAttribute> result;
209         if (this.node.permittedSubclasses == null) {
210             result = Optional.empty();
211         } else {
212             result = Optional.of(
213                 new BytecodeAttribute.PermittedSubclasses(this.node.permittedSubclasses)
214             );
215         }
216         return result;
217     }
218 
219     /**
220      * Record components attribute.
221      * @return Record components attribute if present
222      */
223     private Optional<BytecodeAttribute> records() {
224         final Optional<BytecodeAttribute> result;
225         if (this.node.recordComponents == null) {
226             result = Optional.empty();
227         } else {
228             result = Optional.of(
229                 new BytecodeAttribute.RecordComponents(
230                     new AsmRecordComponents(this.node.recordComponents).bytecode()
231                 )
232             );
233         }
234         return result;
235     }
236 
237     /**
238      * Inner classes attribute.
239      * @return Stream of inner class attributes.
240      */
241     private Stream<InnerClass> inners() {
242         final Stream<InnerClass> result;
243         if (this.node.innerClasses == null) {
244             result = Stream.empty();
245         } else {
246             result = this.node.innerClasses.stream().map(
247                 clazz -> new InnerClass(
248                     clazz.name,
249                     clazz.outerName,
250                     clazz.innerName,
251                     clazz.access
252                 )
253             );
254         }
255         return result;
256     }
257 }
