5   package org.eolang.jeo.representation;
6   
7   import java.util.Objects;
8   import java.util.concurrent.atomic.AtomicReference;
9   import org.objectweb.asm.ClassVisitor;
10  
11  /**
12   * ASM class visitor for extracting class names from bytecode.
13   *
14   * <p>This visitor extracts the class name from bytecode using ASM's visitor pattern.
15   * It captures the class name when the {@link #visit} method is called during
16   * bytecode traversal.</p>
17   * @since 0.1.0
18   */
19  @SuppressWarnings("PMD.UseObjectForClearerAPI")
20  public final class ClassNameVisitor extends ClassVisitor {
21  
22      /**
23       * Atomic reference to store class name.
24       */
25      private final AtomicReference<String> bag;
26  
27      /**
28       * Constructor.
29       */
30      public ClassNameVisitor() {
31          this(new AtomicReference<>());
32      }
33  
34      /**
35       * Constructor.
36       * @param bag The atomic reference to store class name
37       */
38      private ClassNameVisitor(final AtomicReference<String> bag) {
39          this(new DefaultVersion().api(), bag);
40      }
41  
42      /**
43       * Constructor.
44       * @param api The ASM API version to use
45       * @param bag The atomic reference to store class name
46       */
47      private ClassNameVisitor(final int api, final AtomicReference<String> bag) {
48          super(api);
49          this.bag = bag;
50      }
51  
52      @Override
53      public void visit(
54          final int version,
55          final int access,
56          final String name,
57          final String signature,
58          final String supername,
59          final String[] interfaces
60      ) {
61          this.bag.set(name);
62          super.visit(version, access, name, signature, supername, interfaces);
63      }
64  
65      /**
66       * Get class name.
67       * @return The extracted class name
68       */
69      public String asString() {
70          final String last = this.bag.get();
71          if (Objects.isNull(last)) {
72              throw new IllegalStateException(
73                  "Class name is not set, bag is empty. Use #visit() method to set it."
74              );
75          }
76          return last;
77      }
78  }
