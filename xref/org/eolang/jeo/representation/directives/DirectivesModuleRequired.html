5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.xembly.Directive;
9   
10  /**
11   * Directives for module required.
12   * Mirrors {@link org.eolang.jeo.representation.bytecode.BytecodeModuleRequired}.
13   * JVM Specification:
14   * {@code
15   *     {   u2 requires_index;
16   *         u2 requires_flags; {@link #access}
17   *         u2 requires_version_index; {@link #version}
18   *     }}
19   * @since 0.15.0
20   */
21  public final class DirectivesModuleRequired implements Iterable<Directive> {
22  
23      /**
24       * Directives format.
25       */
26      private final Format format;
27  
28      /**
29       * The access flag of the dependence.
30       * Among {@code ACC_TRANSITIVE}, {@code ACC_STATIC_PHASE},
31       * {@code ACC_SYNTHETIC} and {@code ACC_MANDATED}.
32       */
33      private final int access;
34  
35      /**
36       * The module version at compile time.
37       */
38      private final String version;
39  
40      /**
41       * The fully qualified name (using dots) of the dependence.
42       * */
43      private final String module;
44  
45      /**
46       * Constructor.
47       * @param format Directive format
48       * @param module The fully qualified name (using dots) of the dependence
49       * @param access The access flag of the dependence
50       * @param version The module version at compile time
51       * @checkstyle ParameterNumberCheck (5 lines)
52       */
53      public DirectivesModuleRequired(
54          final Format format, final String module, final int access, final String version) {
55          this.format = format;
56          this.module = module;
57          this.access = access;
58          this.version = version;
59      }
60  
61      @Override
62      public Iterator<Directive> iterator() {
63          return new DirectivesJeoObject(
64              "required",
65              "required",
66              new DirectivesValue(this.format, "access", this.access),
67              new DirectivesValue(this.format, "version", this.version),
68              new DirectivesValue(this.format, "module", this.module)
69          ).iterator();
70      }
71  }
