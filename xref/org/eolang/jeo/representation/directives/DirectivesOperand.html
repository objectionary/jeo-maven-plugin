5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.eolang.jeo.representation.bytecode.BytecodeEntry;
9   import org.eolang.jeo.representation.bytecode.BytecodeLabel;
10  import org.objectweb.asm.Handle;
11  import org.objectweb.asm.Label;
12  import org.objectweb.asm.Type;
13  import org.xembly.Directive;
14  
15  /**
16   * Operand XML directives.
17   * @since 0.1
18   */
19  public final class DirectivesOperand implements Iterable<Directive> {
20  
21      /**
22       * Raw operand.
23       */
24      private final Object raw;
25  
26      /**
27       * Constructor.
28       * @param operand Raw operand.
29       */
30      DirectivesOperand(final Object operand) {
31          this.raw = operand;
32      }
33  
34      @Override
35      public Iterator<Directive> iterator() {
36          final Iterator<Directive> result;
37          if (this.raw instanceof Label) {
38              result = new DirectivesLabel(this.raw.toString()).iterator();
39          } else if (this.raw instanceof BytecodeLabel) {
40              result = ((BytecodeEntry) this.raw).directives().iterator();
41          } else if (this.raw instanceof Handle) {
42              result = new DirectivesHandle((Handle) this.raw).iterator();
43          } else if (this.raw instanceof Type) {
44              result = new DirectivesType((Type) this.raw).iterator();
45          } else {
46              result = new DirectivesValue(this.raw).iterator();
47          }
48          return result;
49      }
50  
51  }
