5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import java.util.List;
9   import java.util.concurrent.atomic.AtomicInteger;
10  import java.util.stream.Collectors;
11  import org.xembly.Directive;
12  
13  /**
14   * Directives for module opened.
15   * JVM Specification:
16   * {@code
17   * {   u2 opens_index; {@link #pckg}
18   *     u2 opens_flags; {@link #access}
19   *     u2 opens_to_count; {@link #modules.size()}
20   *     u2 opens_to_index[opens_to_count]; }} {@link #modules}
21   * @since 0.15.0
22   */
23  public final class DirectivesModuleOpened implements Iterable<Directive> {
24  
25      /**
26       * Directives format.
27       */
28      private final Format format;
29  
30      /**
31       * The internal name of the opened package.
32       */
33      private final String pckg;
34  
35      /**
36       * The access flag of the opened package.
37       * Valid values are among {@code ACC_SYNTHETIC} and {@code ACC_MANDATED}.
38       */
39      private final int access;
40  
41      /**
42       * The fully qualified names (using dots) of the modules.
43       */
44      private final List<String> modules;
45  
46      /**
47       * Constructor.
48       * @param format Directive format
49       * @param pckg The internal name of the opened package
50       * @param access The access flag of the opened package
51       * @param modules The fully qualified names (using dots) of the modules
52       * @checkstyle ParameterNumberCheck (5 lines)
53       */
54      public DirectivesModuleOpened(
55          final Format format,
56          final String pckg,
57          final int access,
58          final List<String> modules
59      ) {
60          this.format = format;
61          this.pckg = pckg;
62          this.access = access;
63          this.modules = modules;
64      }
65  
66      @Override
67      public Iterator<Directive> iterator() {
68          final AtomicInteger counter = new AtomicInteger(0);
69          return new DirectivesJeoObject(
70              "opened",
71              "opened",
72              new DirectivesValue(this.format, "package", this.pckg),
73              new DirectivesValue(this.format, "access", this.access),
74              new DirectivesSeq(
75                  "modules",
76                  this.modules.stream()
77                      .map(
78                          module -> new DirectivesValue(
79                              this.format,
80                              String.format("m%d", counter.getAndIncrement()),
81                              module
82                          )
83                      ).collect(Collectors.toList())
84              )
85          ).iterator();
86      }
87  }
