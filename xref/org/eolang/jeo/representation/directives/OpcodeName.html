5   package org.eolang.jeo.representation.directives;
6   
7   import java.lang.reflect.Field;
8   import java.util.HashMap;
9   import java.util.Locale;
10  import java.util.Map;
11  import java.util.concurrent.atomic.AtomicInteger;
12  import org.objectweb.asm.Opcodes;
13  
14  /**
15   * Opcode name.
16   * The name of bytecode instruction. The name combined with unique number in order to
17   * avoid name collisions.
18   * @since 0.1.0
19   */
20  public final class OpcodeName {
21  
22      /**
23       * Opcode names.
24       */
25      private static final Map<Integer, String> NAMES = OpcodeName.init();
26  
27      /**
28       * Default counter.
29       */
30      private static final AtomicInteger DEFAULT = new AtomicInteger(0);
31  
32      /**
33       * Unknown opcode name.
34       */
35      private static final String UNKNOWN = "unknown";
36  
37      /**
38       * Bytecode operation code.
39       */
40      private final int opcode;
41  
42      /**
43       * Opcode counter.
44       */
45      private final AtomicInteger counter;
46  
47      /**
48       * Constructor.
49       * @param opcode Bytecode operation code.
50       */
51      public OpcodeName(final int opcode) {
52          this(opcode, OpcodeName.DEFAULT);
53      }
54  
55      /**
56       * Constructor.
57       * @param opcode Bytecode operation code.
58       * @param counter Opcode counter.
59       */
60      OpcodeName(final int opcode, final AtomicInteger counter) {
61          this.opcode = opcode;
62          this.counter = counter;
63      }
64  
65      /**
66       * Get simplified opcode name without counter.
67       * @return Simplified opcode name.
68       */
69      public String simplified() {
70          return OpcodeName.NAMES.getOrDefault(this.opcode, OpcodeName.UNKNOWN);
71      }
72  
73      /**
74       * Opcode.
75       * @return Opcode
76       * @todo #770:35min Get rid of code() method.
77       *  This method looks like a getter. For now this method should be used by
78       *  <a href="https://github.com/objectionary/opeo-maven-plugin">opeo</a>
79       *  in it/AgentsIT.java. Let's remove code() method, so this class will be
80       *  getter-free.
81       */
82      public int code() {
83          return this.opcode;
84      }
85  
86      /**
87       * Get string representation of a bytecode.
88       * @return String representation of a bytecode.
89       */
90      public String asString() {
91          final String opc = OpcodeName.NAMES.getOrDefault(this.opcode, OpcodeName.UNKNOWN);
92          return String.format("%s-%X", opc, this.counter.incrementAndGet());
93      }
94  
95      /**
96       * Initialize opcode names.
97       * @return Opcode names.
98       */
99      private static Map<Integer, String> init() {
100         try {
101             final Map<Integer, String> res = new HashMap<>();
102             for (final Field field : Opcodes.class.getFields()) {
103                 if (field.getType() == int.class) {
104                     res.put(field.getInt(Opcodes.class), field.getName().toLowerCase(Locale.ROOT));
105                 }
106             }
107             return res;
108         } catch (final IllegalAccessException exception) {
109             throw new IllegalStateException(
110                 String.format("Can't retrieve opcode names from '%s'", Opcodes.class),
111                 exception
112             );
113         }
114     }
115 }
