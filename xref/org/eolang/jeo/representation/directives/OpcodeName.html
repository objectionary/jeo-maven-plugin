5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.concurrent.atomic.AtomicInteger;
8   import org.eolang.jeo.representation.OpcodeDictionary;
9   
10  /**
11   * Opcode name.
12   * The name of bytecode instruction. The name combined with unique number in order to
13   * avoid name collisions.
14   * @since 0.1.0
15   */
16  public final class OpcodeName {
17  
18      /**
19       * Opcode names.
20       */
21      private static final OpcodeDictionary NAMES = new OpcodeDictionary();
22  
23      /**
24       * Default counter.
25       */
26      private static final AtomicInteger DEFAULT = new AtomicInteger(0);
27  
28      /**
29       * Bytecode operation code.
30       */
31      private final int opcode;
32  
33      /**
34       * Opcode counter.
35       */
36      private final AtomicInteger counter;
37  
38      /**
39       * Constructor.
40       * @param opcode Bytecode operation code.
41       */
42      public OpcodeName(final int opcode) {
43          this(opcode, OpcodeName.DEFAULT);
44      }
45  
46      /**
47       * Constructor.
48       * @param opcode Bytecode operation code.
49       * @param counter Opcode counter.
50       */
51      OpcodeName(final int opcode, final AtomicInteger counter) {
52          this.opcode = opcode;
53          this.counter = counter;
54      }
55  
56      /**
57       * Get simplified opcode name without counter.
58       * @return Simplified opcode name.
59       */
60      public String simplified() {
61          return OpcodeName.NAMES.name(this.opcode);
62      }
63  
64      /**
65       * Get string representation of a bytecode.
66       * @return String representation of a bytecode.
67       */
68      String asString() {
69          return String.format(
70              "%s-%X", OpcodeName.NAMES.name(this.opcode), this.counter.incrementAndGet()
71          );
72      }
73  }
