24  package org.eolang.jeo.representation.directives;
25  
26  import java.util.Iterator;
27  import java.util.List;
28  import java.util.stream.Collectors;
29  import java.util.stream.Stream;
30  import org.xembly.Directive;
31  import org.xembly.Directives;
32  
33  /**
34   * An annotation value that is an array.
35   * @since 0.6
36   */
37  public final class DirectivesArrayAnnotationValue implements Iterable<Directive> {
38  
39      /**
40       * The name of the annotation property.
41       */
42      private final String name;
43  
44      /**
45       * The actual values.
46       */
47      private final List<Iterable<Directive>> values;
48  
49      /**
50       * Constructor.
51       * @param name The name of the annotation property.
52       * @param children The actual values.
53       */
54      public DirectivesArrayAnnotationValue(
55          final String name, final List<Iterable<Directive>> children
56      ) {
57          this.name = name;
58          this.values = children;
59      }
60  
61      @Override
62      public Iterator<Directive> iterator() {
63          return new DirectivesJeoObject(
64              "annotation-property",
65              Stream.concat(
66                  Stream.of(
67                      new DirectivesValue("ARRAY"),
68                      new DirectivesValue(this.name)
69                  ),
70                  this.values.stream()
71              ).map(Directives::new).collect(Collectors.toList())
72          ).iterator();
73      }
74  }
