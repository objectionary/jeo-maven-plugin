5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Arrays;
8   import java.util.Iterator;
9   import java.util.Locale;
10  import java.util.UUID;
11  import java.util.concurrent.atomic.AtomicInteger;
12  import java.util.regex.Pattern;
13  import java.util.stream.Collectors;
14  import org.eolang.jeo.representation.bytecode.BytecodeEntry;
15  import org.eolang.jeo.representation.bytecode.BytecodeLabel;
16  import org.xembly.Directive;
17  
18  /**
19   * Array representation of some values as Xembly directives.
20   * We used to use "tuple" for this, but it leads to some confusion.
21   * You can read more about this problem
22   * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/707">here</a>
23   * @since 0.6
24   */
25  public final class DirectivesValues implements Iterable<Directive> {
26  
27      /**
28       * Pattern to remove all non-digits from the beginning of a string.
29       */
30      private static final Pattern DIGITS = Pattern.compile("^[0-9]");
31  
32      /**
33       * The format of the directives.
34       */
35      private final Format format;
36  
37      /**
38       * Tuple name.
39       */
40      private final String name;
41  
42      /**
43       * Tuple values.
44       */
45      private final Object[] values;
46  
47      /**
48       * Constructor.
49       * @param format The format of the directives.
50       * @param name Group of values name.
51       * @param vals Values themselves.
52       * @param <T> Values type.
53       */
54      @SafeVarargs
55      public <T> DirectivesValues(final Format format, final String name, final T... vals) {
56          this.format = format;
57          this.name = name;
58          this.values = vals.clone();
59      }
60  
61      @Override
62      public Iterator<Directive> iterator() {
63          final AtomicInteger index = new AtomicInteger(0);
64          return new DirectivesSeq(
65              this.nonEmptyName(),
66              Arrays.stream(this.values)
67                  .map(
68                      value -> {
69                          final Iterable<Directive> result;
70                          if (value instanceof BytecodeLabel) {
71                              result = ((BytecodeEntry) value).directives(
72                                  index.getAndIncrement(),
73                                  this.format
74                              );
75                          } else {
76                              result = new DirectivesValue(
77                                  this.format,
78                                  String.format("x%d", index.getAndIncrement()),
79                                  value
80                              );
81                          }
82                          return result;
83                      }
84                  )
85                  .collect(Collectors.toList())
86          ).iterator();
87      }
88  
89      /**
90       * Name of the group of values.
91       * @return Name of the group of values.
92       */
93      private String nonEmptyName() {
94          final String result;
95          if (this.name.isEmpty()) {
96              result = DirectivesValues.randomName();
97          } else {
98              result = this.name;
99          }
100         return result;
101     }
102 
103     /**
104      * Generate random name.
105      * @return Random name.
106      */
107     private static String randomName() {
108         return DirectivesValues.DIGITS.matcher(
109             UUID.randomUUID().toString().toLowerCase(Locale.getDefault())
110         ).replaceAll("a");
111     }
112 }
