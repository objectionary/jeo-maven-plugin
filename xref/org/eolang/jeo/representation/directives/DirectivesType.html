5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.objectweb.asm.Type;
9   import org.xembly.Directive;
10  
11  /**
12   * Directives for Type object.
13   * @since 0.11.0
14   */
15  public final class DirectivesType implements Iterable<Directive> {
16  
17      /**
18       * ASM Type object.
19       */
20      private final Type type;
21  
22      /**
23       * Constructor.
24       * @param type ASM Type object
25       */
26      public DirectivesType(final Type type) {
27          this.type = type;
28      }
29  
30      @Override
31      public Iterator<Directive> iterator() {
32          return new DirectivesJeoObject(
33              "type",
34              new RandName("type").toString(),
35              new DirectivesValue(this.type.getDescriptor())
36          ).iterator();
37      }
38  }
