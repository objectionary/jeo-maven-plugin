5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.ArrayList;
8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.stream.Collectors;
11  import java.util.stream.Stream;
12  import org.eolang.jeo.representation.MethodName;
13  import org.eolang.jeo.representation.NumberedName;
14  import org.eolang.jeo.representation.PrefixedName;
15  import org.xembly.Directive;
16  import org.xembly.Directives;
17  
18  /**
19   * Directives Method.
20   * @since 0.1
21   * @todo #534:90min Refactor DirectivesMethod class to simplify the code.
22   *  Currently, the class has a lot of methods and fields. It's kind of hard to understand
23   *  what this class does. We need to refactor it to make it more readable and understandable.
24   *  Moreover, in many places this class uses null checks, which is not good.
25   */
26  public final class DirectivesMethod implements Iterable<Directive> {
27  
28      /**
29       * Directives format.
30       */
31      private final Format format;
32  
33      /**
34       * Method name.
35       */
36      private final NumberedName name;
37  
38      /**
39       * Method properties.
40       */
41      private final DirectivesMethodProperties properties;
42  
43      /**
44       * Method instructions.
45       */
46      private final List<Iterable<Directive>> instructions;
47  
48      /**
49       * Method exceptions.
50       */
51      private final List<Iterable<Directive>> exceptions;
52  
53      /**
54       * Method annotations.
55       */
56      private final DirectivesAnnotations annotations;
57  
58      /**
59       * Default value.
60       */
61      private final List<Iterable<Directive>> dvalue;
62  
63      /**
64       * Method attributes.
65       */
66      private final DirectivesAttributes attributes;
67  
68      /**
69       * Constructor.
70       * @param name Method name
71       */
72      public DirectivesMethod(final String name) {
73          this(name, new DirectivesMethodProperties());
74      }
75  
76      /**
77       * Constructor.
78       * @param name Method name
79       * @param properties Method properties
80       */
81      public DirectivesMethod(
82          final String name,
83          final DirectivesMethodProperties properties
84      ) {
85          this(
86              new Format(),
87              new NumberedName(1, new MethodName(name).xmir()),
88              properties,
89              new ArrayList<>(0),
90              new ArrayList<>(0),
91              new DirectivesAnnotations(),
92              new ArrayList<>(0),
93              new DirectivesAttributes()
94          );
95      }
96  
97      /**
98       * Constructor.
99       * @param format Directives format
100      * @param name Method name
101      * @param properties Method properties
102      * @param instructions Method instructions
103      * @param exceptions Method exceptions
104      * @param annotations Method annotations
105      * @param dvalue Annotation default value
106      * @param attributes Method attributes
107      * @checkstyle ParameterNumberCheck (10 lines)
108      */
109     public DirectivesMethod(
110         final Format format,
111         final NumberedName name,
112         final DirectivesMethodProperties properties,
113         final List<Iterable<Directive>> instructions,
114         final List<Iterable<Directive>> exceptions,
115         final DirectivesAnnotations annotations,
116         final List<Iterable<Directive>> dvalue,
117         final DirectivesAttributes attributes
118     ) {
119         this.format = format;
120         this.name = name;
121         this.properties = properties;
122         this.instructions = instructions;
123         this.exceptions = exceptions;
124         this.annotations = annotations;
125         this.dvalue = dvalue;
126         this.attributes = attributes;
127     }
128 
129     /**
130      * Add opcode to the directives.
131      * @param index Instruction index
132      * @param opcode Opcode
133      * @param operands Operands
134      * @return This object
135      */
136     public DirectivesMethod withOpcode(
137         final int index, final int opcode, final Object... operands
138     ) {
139         this.instructions.add(new DirectivesInstruction(index, this.format, opcode, operands));
140         return this;
141     }
142 
143     /**
144      * Add annotation to the directives.
145      * @param annotation Annotation directives.
146      * @return This object.
147      */
148     public DirectivesMethod withAnnotation(final DirectivesAnnotation annotation) {
149         this.annotations.add(annotation);
150         return this;
151     }
152 
153     @Override
154     public Iterator<Directive> iterator() {
155         final String mname = this.name.toString();
156         return new DirectivesJeoObject(
157             "method",
158             new PrefixedName(mname).encode(),
159             Stream.concat(
160                 Stream.of(
161                     this.properties,
162                     this.annotations,
163                     new DirectivesOptionalSeq("body", this.instructions),
164                     new DirectivesOptionalSeq(
165                         String.format("trycatchblocks-%s", this.name.plain()),
166                         this.exceptions
167                     )
168                 ),
169                 Stream.concat(
170                     this.dvalue.stream(),
171                     Stream.of(
172                         this.attributes,
173                         new DirectivesValue(this.format, "name", mname)
174                     )
175                 )
176             ).map(Directives::new).collect(Collectors.toList())
177         ).iterator();
178     }
179 }
