5   package org.eolang.jeo.representation.bytecode;
6   
7   import lombok.EqualsAndHashCode;
8   import lombok.ToString;
9   import org.eolang.jeo.representation.directives.DirectivesDefaultValue;
10  import org.objectweb.asm.AnnotationVisitor;
11  import org.objectweb.asm.MethodVisitor;
12  import org.xembly.Directive;
13  
14  /**
15   * Bytecode default value.
16   *
17   * @since 0.3
18   */
19  @ToString
20  @EqualsAndHashCode
21  public final class BytecodeDefaultValue {
22  
23      /**
24       * Annotation property as a value.
25       */
26      private final BytecodeAnnotationValue property;
27  
28      /**
29       * Constructor.
30       * @param property Annotation property as a value.
31       */
32      public BytecodeDefaultValue(final BytecodeAnnotationValue property) {
33          this.property = property;
34      }
35  
36      /**
37       * Write the default value to the given visitor.
38       * @param mvisitor Visitor.
39       */
40      public void writeTo(final MethodVisitor mvisitor) {
41          final AnnotationVisitor visitor = mvisitor.visitAnnotationDefault();
42          this.property.writeTo(visitor);
43          visitor.visitEnd();
44      }
45  
46      public Iterable<Directive> directives() {
47          return new DirectivesDefaultValue(this.property.directives());
48      }
49  }
