5   package org.eolang.jeo.representation.bytecode;
6   
7   import java.util.List;
8   import org.eolang.jeo.representation.asm.AsmLabels;
9   import org.objectweb.asm.MethodVisitor;
10  import org.xembly.Directive;
11  
12  /**
13   * Bytecode instruction or a label.
14   * Might be a label, a jump, a method call, etc.
15   * @since 0.1
16   */
17  public interface BytecodeEntry {
18      /**
19       * Write instruction to the method visitor.
20       * @param visitor Method visitor.
21       * @param labels Method labels.
22       */
23      void writeTo(MethodVisitor visitor, AsmLabels labels);
24  
25      /**
26       * Convert entry to directives.
27       * @return Directives.
28       */
29      Iterable<Directive> directives();
30  
31      /**
32       * Is this instruction a label?
33       * @return True if it is.
34       */
35      boolean isLabel();
36  
37      /**
38       * Is this instruction a switch?
39       * @return True if it is.
40       */
41      boolean isSwitch();
42  
43      /**
44       * Is this instruction a goto?
45       * Is it a goto or jsr?
46       * @return True if it is.
47       */
48      boolean isJump();
49  
50      /**
51       * Is this instruction a conditional branch?
52       * @return True if it is.
53       */
54      boolean isIf();
55  
56      /**
57       * Is this instruction a return statement?
58       * @return True if it is.
59       */
60      boolean isReturn();
61  
62      /**
63       * Is this instruction a throw statement?
64       * @return True if it is.
65       */
66      boolean isThrow();
67  
68      /**
69       * Is this instruction a regular opcode?
70       * @return True if it is.
71       */
72      boolean isOpcode();
73  
74      /**
75       * Impact of the instruction on the stack.
76       * @return Stack impact.
77       */
78      int impact();
79  
80      /**
81       * Jump to a label.
82       * Where to jump.
83       * @return Jumps.
84       */
85      List<BytecodeLabel> jumps();
86  
87      /**
88       * Human-readable representation.
89       * @return Text.
90       */
91      String view();
92  }
