5   package org.eolang.jeo.representation.bytecode;
6   
7   import java.util.Collections;
8   import java.util.List;
9   import java.util.Optional;
10  import lombok.EqualsAndHashCode;
11  import lombok.ToString;
12  import org.eolang.jeo.representation.directives.DirectivesModuleOpened;
13  import org.eolang.jeo.representation.directives.Format;
14  import org.objectweb.asm.ModuleVisitor;
15  
16  /**
17   * A node that represents an opened package with its name and the module that can access it.
18   * @since 0.15.0
19   */
20  @ToString
21  @EqualsAndHashCode
22  public final class BytecodeModuleOpened {
23  
24      /**
25       * The internal name of the opened package.
26       */
27      private final String pckg;
28  
29      /**
30       * The access flag of the opened package.
31       * Valid values are among {@code ACC_SYNTHETIC} and {@code ACC_MANDATED}.
32       */
33      private final int access;
34  
35      /**
36       * The fully qualified names (using dots) of the modules.
37       */
38      private final List<String> modules;
39  
40      /**
41       * Constructor.
42       * @param pckg The internal name of the opened package
43       * @param access The access flag of the opened package
44       * @param modules The fully qualified names (using dots) of the modules
45       */
46      public BytecodeModuleOpened(final String pckg, final int access, final List<String> modules) {
47          this.pckg = pckg;
48          this.access = access;
49          this.modules = Optional.ofNullable(modules).orElse(Collections.emptyList());
50      }
51  
52      /**
53       * Writes this opened package to the given module visitor.
54       * @param module The module visitor
55       */
56      public void write(final ModuleVisitor module) {
57          module.visitOpen(this.pckg, this.access, this.modules.toArray(new String[0]));
58      }
59  
60      /**
61       * Converts this opened package to directives.
62       * @param format Directive format
63       * @return Directives
64       */
65      public DirectivesModuleOpened directives(final Format format) {
66          return new DirectivesModuleOpened(
67              format,
68              this.pckg,
69              this.access,
70              this.modules
71          );
72      }
73  }
