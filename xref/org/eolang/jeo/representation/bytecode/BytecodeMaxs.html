5   package org.eolang.jeo.representation.bytecode;
6   
7   import lombok.EqualsAndHashCode;
8   import lombok.ToString;
9   import org.eolang.jeo.representation.directives.DirectivesMaxs;
10  import org.eolang.jeo.representation.directives.Format;
11  
12  /**
13   * Bytecode maxs.
14   * @since 0.6
15   */
16  @ToString
17  @EqualsAndHashCode
18  @SuppressWarnings("PMD.AvoidFieldNameMatchingMethodName")
19  public final class BytecodeMaxs {
20  
21      /**
22       * Undefined value.
23       * Need to be recomputed.
24       */
25      private static final int UNDEFINED = -1;
26  
27      /**
28       * Stack max size.
29       */
30      private final int stack;
31  
32      /**
33       * Locals variables size.
34       */
35      private final int locals;
36  
37      /**
38       * Constructor.
39       */
40      public BytecodeMaxs() {
41          this(BytecodeMaxs.UNDEFINED, BytecodeMaxs.UNDEFINED);
42      }
43  
44      /**
45       * Constructor.
46       * @param stack Stack size.
47       * @param locals Locals size.
48       */
49      public BytecodeMaxs(final int stack, final int locals) {
50          this.stack = stack;
51          this.locals = locals;
52      }
53  
54      /**
55       * Stack size.
56       * @return Stack size.
57       */
58      public int stack() {
59          return this.stack;
60      }
61  
62      /**
63       * Locals size.
64       * @return Locals size.
65       */
66      public int locals() {
67          return this.locals;
68      }
69  
70      /**
71       * Convert to directives.
72       * @param format Format of the directives.
73       * @return Directives.
74       */
75      public DirectivesMaxs directives(final Format format) {
76          return new DirectivesMaxs(format, this.stack, this.locals);
77      }
78  
79      /**
80       * Is maxs stack and locals are zero?
81       * @return True if both are zero.
82       */
83      boolean compute() {
84          return this.stack == BytecodeMaxs.UNDEFINED && this.locals == BytecodeMaxs.UNDEFINED;
85      }
86  }
