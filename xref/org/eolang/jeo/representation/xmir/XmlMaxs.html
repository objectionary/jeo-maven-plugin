5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.Objects;
8   import java.util.stream.Collectors;
9   import lombok.EqualsAndHashCode;
10  import lombok.ToString;
11  import org.eolang.jeo.representation.bytecode.BytecodeMaxs;
12  
13  /**
14   * Xmir representation of max stack and max locals of a method.
15   * @since 0.3
16   */
17  @EqualsAndHashCode
18  @ToString
19  public final class XmlMaxs {
20  
21      /**
22       * XML node.
23       */
24      @EqualsAndHashCode.Exclude
25      private final XmlJeoObject node;
26  
27      /**
28       * Constructor.
29       * @param node XML Jeo object node representing the maxs.
30       */
31      XmlMaxs(final XmlJeoObject node) {
32          this.node = node;
33      }
34  
35      /**
36       * Convert to bytecode maxs.
37       * @return Bytecode maxs.
38       */
39      public BytecodeMaxs bytecode() {
40          return new BytecodeMaxs(this.stack(), this.locals());
41      }
42  
43      /**
44       * Stack max size.
45       * @return Stack size.
46       */
47      @EqualsAndHashCode.Include
48      private int stack() {
49          return this.ichild(0);
50      }
51  
52      /**
53       * Locals max size.
54       * @return Locals size.
55       */
56      @EqualsAndHashCode.Include
57      private int locals() {
58          return this.ichild(1);
59      }
60  
61      /**
62       * Retrieve integer child.
63       * @param position Position.
64       * @return Integer value.
65       */
66      private int ichild(final int position) {
67          return (int) Objects.requireNonNull(
68              new XmlOperand(
69                  this.node.children().collect(Collectors.toList()).get(position)
70              ).asObject(),
71              String.format(
72                  "The XML node representing Maxs '%s' doesn't contain a valid integer at '%d' position",
73                  this.node,
74                  position
75              )
76          );
77      }
78  }
