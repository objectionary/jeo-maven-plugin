24  package org.eolang.jeo.representation.xmir;
25  
26  import java.io.FileNotFoundException;
27  import java.nio.file.Path;
28  import javax.xml.parsers.DocumentBuilderFactory;
29  import org.cactoos.scalar.Sticky;
30  import org.cactoos.scalar.Synced;
31  import org.cactoos.scalar.Unchecked;
32  
33  /**
34   * Native XML document.
35   * @since 0.7
36   */
37  public final class NativeXmlDoc implements XmlDoc {
38  
39      /**
40       * XML document factory.
41       */
42      private static final DocumentBuilderFactory DOC_FACTORY = DocumentBuilderFactory.newInstance();
43  
44      /**
45       * XML.
46       */
47      private final Unchecked<XmlNode> xml;
48  
49      /**
50       * Constructor.
51       * @param path Path to XML file.
52       */
53      public NativeXmlDoc(final Path path) {
54          this(NativeXmlDoc.fromFile(path));
55      }
56  
57      private NativeXmlDoc(final Unchecked<XmlNode> xml) {
58          this.xml = xml;
59      }
60  
61      @Override
62      public XmlNode root() {
63          return this.xml.value();
64      }
65  
66      @Override
67      public void validate() {
68          this.xml.value().validate();
69      }
70  
71      /**
72       * Prestructor that converts a path to a lazy XML.
73       * @param path Path to an XML file.
74       * @return Lazy XML.
75       */
76      private static Unchecked<XmlNode> fromFile(final Path path) {
77          return new Unchecked<>(new Synced<>(new Sticky<>(() -> NativeXmlDoc.open(path))));
78      }
79  
80      /**
81       * Convert a path to XML.
82       * @param path Path to XML file.
83       * @return XML.
84       * @checkstyle IllegalCatchCheck (20 lines)
85       */
86      @SuppressWarnings("PMD.AvoidCatchingGenericException")
87      private static XmlNode open(final Path path) {
88          try {
89              return new NativeXmlNode(
90                  NativeXmlDoc.DOC_FACTORY
91                      .newDocumentBuilder()
92                      .parse(path.toFile())
93                      .getDocumentElement()
94              );
95          } catch (final FileNotFoundException exception) {
96              throw new IllegalStateException(
97                  String.format("Can't find file '%s'", path),
98                  exception
99              );
100         } catch (final Exception broken) {
101             throw new IllegalStateException(
102                 String.format(
103                     "Can't parse XML from the file '%s'",
104                     path
105                 ),
106                 broken
107             );
108         }
109     }
110 }
