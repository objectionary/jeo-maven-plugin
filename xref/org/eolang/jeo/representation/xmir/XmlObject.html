5   package org.eolang.jeo.representation.xmir;
6   
7   import com.jcabi.xml.XML;
8   import org.eolang.jeo.representation.PrefixedName;
9   import org.eolang.jeo.representation.bytecode.BytecodeObject;
10  
11  /**
12   * XMIR Program.
13   * @since 0.1
14   */
15  public final class XmlObject {
16  
17      /**
18       * Root node.
19       */
20      private final XmlNode root;
21  
22      /**
23       * Constructor.
24       * @param lines Xmir lines.
25       */
26      public XmlObject(final String... lines) {
27          this(new JcabiXmlNode(lines));
28      }
29  
30      /**
31       * Constructor.
32       * @param xml Raw XMIR.
33       */
34      public XmlObject(final XML xml) {
35          this(new JcabiXmlDoc(xml).root());
36      }
37  
38      /**
39       * Constructor.
40       * @param root Root node.
41       */
42      public XmlObject(final XmlNode root) {
43          this.root = root;
44      }
45  
46      @Override
47      public String toString() {
48          return this.root.toString();
49      }
50  
51      /**
52       * Convert to bytecode.
53       * @return Bytecode program.
54       */
55      public BytecodeObject bytecode() {
56          try {
57              return new BytecodeObject(this.pckg(), this.top().bytecode());
58          } catch (final IllegalStateException exception) {
59              throw new ParsingException(
60                  String.format(
61                      "Unexpected exception during parsing the program in package '%s'",
62                      this.pckg()
63                  ),
64                  exception
65              );
66          }
67      }
68  
69      /**
70       * Find top-level class.
71       *
72       * @return Class.
73       */
74      private XmlClass top() {
75          return new XmlClass(this.pckg(), this.root.child("o"));
76      }
77  
78      /**
79       * Retrieve program package.
80       * In case if metas are empty, or there is no package meta, or there is no tail, return empty
81       * string.
82       *
83       * @return Package.
84       */
85      private String pckg() {
86          return this.root
87              .xpath("/object/metas/meta[head='package']/tail/text()")
88              .stream()
89              .findFirst()
90              .map(PrefixedName::new)
91              .map(PrefixedName::decode)
92              .orElse("");
93      }
94  }
