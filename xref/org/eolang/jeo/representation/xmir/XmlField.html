24  package org.eolang.jeo.representation.xmir;
25  
26  import java.util.Optional;
27  import java.util.stream.Collectors;
28  import lombok.ToString;
29  import org.eolang.jeo.representation.PrefixedName;
30  import org.eolang.jeo.representation.bytecode.BytecodeAnnotations;
31  import org.eolang.jeo.representation.bytecode.BytecodeField;
32  
33  /**
34   * XML field.
35   * @since 0.1
36   */
37  @ToString
38  public class XmlField {
39  
40      /**
41       * Field node.
42       */
43      private final XmlNode node;
44  
45      /**
46       * Constructor.
47       * @param xmlnode Field node.
48       */
49      XmlField(final XmlNode xmlnode) {
50          this.node = xmlnode;
51      }
52  
53      /**
54       * Convert to bytecode.
55       * @return Bytecode field.
56       */
57      public BytecodeField bytecode() {
58          return new BytecodeField(
59              this.name(),
60              this.descriptor(),
61              this.signature(),
62              this.value(),
63              this.access(),
64              this.annotations().map(XmlAnnotations::bytecode).orElse(new BytecodeAnnotations())
65          );
66      }
67  
68      /**
69       * Field name.
70       * @return Name.
71       */
72      private String name() {
73          return new PrefixedName(
74              this.node.attribute("name").orElseThrow(
75                  () -> new IllegalStateException(
76                      String.format("Can't find field name in '%s'", this.node)
77                  )
78              )
79          ).decode();
80      }
81  
82      /**
83       * Field access modifiers.
84       * @return Access modifiers.
85       */
86      private int access() {
87          return this.find(Attribute.ACCESS).map(XmlValue::integer).orElse(0);
88      }
89  
90      /**
91       * Field descriptor.
92       * @return Descriptor.
93       */
94      private String descriptor() {
95          return this.find(Attribute.DESCRIPTOR).map(XmlValue::string)
96              .filter(descriptor -> !descriptor.isEmpty())
97              .orElse(null);
98      }
99  
100     /**
101      * Field signature.
102      * @return Signature.
103      */
104     private String signature() {
105         return this.find(Attribute.SIGNATURE).map(XmlValue::string)
106             .filter(signature -> !signature.isEmpty())
107             .orElse(null);
108     }
109 
110     /**
111      * Field value.
112      * @return Value.
113      */
114     private Object value() {
115         return new XmlOperand(
116             this.node.children()
117                 .collect(Collectors.toList())
118                 .get(Attribute.VALUE.ordinal())
119         ).asObject();
120     }
121 
122     /**
123      * Field annotations.
124      * @return Annotations.
125      */
126     private Optional<XmlAnnotations> annotations() {
127         return this.node.optchild("name", String.format("annotations-%s", this.name()))
128             .map(XmlAnnotations::new);
129     }
130 
131     /**
132      * Find node text by attribute.
133      * @param attribute Attribute.
134      * @return Text.
135      */
136     private Optional<XmlValue> find(final Attribute attribute) {
137         return Optional.of(
138             new XmlValue(
139                 this.node.children().collect(Collectors.toList()).get(attribute.ordinal())
140             )
141         );
142     }
143 
144     /**
145      * Field attribute.
146      * Pay attention that the order of the attributes is important.
147      * They should be in the same order as in the XML representation.
148      * @since 0.1
149      */
150     private enum Attribute {
151 
152         /**
153          * Access modifier.
154          * It's a number that represents sum of access modifiers.
155          * For example, if the field is public and static, then the value will be 9.
156          * See {@link org.objectweb.asm.Opcodes} for more details.
157          */
158         ACCESS,
159 
160         /**
161          * Field descriptor.
162          * For example, for field of type int the descriptor will be "I".
163          */
164         DESCRIPTOR,
165 
166         /**
167          * Field signature.
168          * The field signature is used for defining generic types.
169          * Signature grammar:
170          * <p>
171          * {@code
172          * TypeSignature: Z|C|B|S|I|F|J|D|FieldTypeSignature
173          * FieldTypeSignature: ClassTypeSignature | [ TypeSignature | TypeVar
174          * ClassTypeSignature: L Id ( / Id )* TypeArgs? ( . Id TypeArgs? )* ;
175          * TypeArgs: < TypeArg+ >
176          * TypeArg: * | ( + | - )? FieldTypeSignature
177          * TypeVar: T Id ;
178          * }
179          * </p>
180          * Examples:
181          * <p>
182          * {@code
183          * List<E> -> Ljava/util/List<TE;>;
184          * List<?> -> Ljava/util/List<*>;
185          * List<? extends Number> -> Ljava/util/List<+Ljava/lang/Number;>;
186          * List<? super Integer> -> Ljava/util/List<-Ljava/lang/Integer;>;
187          * List<List<String>[]> -> Ljava/util/List<[Ljava/util/List<Ljava/lang/String;>;>;
188          * HashMap<K, V>.HashIterator<K> -> Ljava/util/HashMap<TK;TV;>.HashIterator<TK;>;
189          * }
190          * </p>
191          * You can read more in 4.1.1 section of the ASM
192          * <a href="https://asm.ow2.io/asm4-guide.pdf">manual </a>
193          */
194         SIGNATURE,
195 
196         /**
197          * Initial field value.
198          * For example, for field of type int with value 19, the value will be "13".
199          * Any data type will be represented as a hex string.
200          * May not be set.
201          */
202         VALUE;
203     }
204 }
