5   package org.eolang.jeo.representation.xmir;
6   
7   import java.nio.charset.StandardCharsets;
8   import org.eolang.jeo.representation.bytecode.BytecodeLabel;
9   
10  /**
11   * XML representation of bytecode label.
12   * @since 0.1
13   */
14  public final class XmlLabel implements XmlBytecodeEntry {
15  
16      /**
17       * Label node.
18       */
19      private final XmlJeoObject node;
20  
21      /**
22       * Constructor.
23       * @param node Label node.
24       */
25      XmlLabel(final XmlNode node) {
26          this.node = new XmlJeoObject(node);
27      }
28  
29      /**
30       * Converts label to bytecode.
31       * @return Bytecode label.
32       */
33      public BytecodeLabel bytecode() {
34          return new BytecodeLabel(
35              new String(
36                  (byte[]) new XmlValue(
37                      this.node.children()
38                          .findFirst()
39                          .orElseThrow(
40                              () -> new IllegalStateException(
41                                  "Label node must have at least one child"
42                              )
43                          )
44                  ).object(),
45                  StandardCharsets.UTF_8
46              )
47          );
48      }
49  }
