5   package org.eolang.jeo;
6   
7   import java.nio.file.Path;
8   import java.util.function.Function;
9   import java.util.stream.Stream;
10  
11  /**
12   * Translator that applies a translation to a batch of representations in parallel.
13   *
14   * <p>This class implements parallel processing of transformations to improve performance.
15   * It ensures that each parallel thread has the correct class loader context to avoid
16   * class loading issues during concurrent execution.</p>
17   * @since 0.2.0
18   */
19  public final class ParallelTranslator implements Translator {
20  
21      /**
22       * Original translation.
23       */
24      private final Function<? super Path, ? extends Path> translation;
25  
26      /**
27       * Class loader.
28       */
29      private final ClassLoader loader;
30  
31      /**
32       * Constructor.
33       * @param translation Function to apply to each path representation
34       */
35      ParallelTranslator(final Function<? super Path, ? extends Path> translation) {
36          this.translation = translation;
37          this.loader = Thread.currentThread().getContextClassLoader();
38      }
39  
40      @Override
41      public Stream<Path> apply(final Stream<Path> representations) {
42          return representations.parallel().map(this::translate);
43      }
44  
45      /**
46       * Translate a representation.
47       * <p>This method is run in parallel. Pay attention to the class loader;
48       * it's set for each sub-thread to avoid class loading issues.</p>
49       * @param rep Path representation to translate
50       * @return Translated path representation
51       */
52      private Path translate(final Path rep) {
53          Thread.currentThread().setContextClassLoader(this.loader);
54          return this.translation.apply(rep);
55      }
56  
57  }
