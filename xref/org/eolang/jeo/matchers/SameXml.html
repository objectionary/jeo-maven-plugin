5   package org.eolang.jeo.matchers;
6   
7   import com.jcabi.xml.XML;
8   import com.jcabi.xml.XMLDocument;
9   import org.hamcrest.Description;
10  import org.hamcrest.TypeSafeMatcher;
11  
12  /**
13   * Matcher to check if the received XML document is the same as the expected one.
14   * Smart comparison of XML documents that ignores 'line' attributes.
15   * @since 0.6
16   */
17  public final class SameXml extends TypeSafeMatcher<String> {
18  
19      /**
20       * Expected XML document.
21       */
22      private final String expected;
23  
24      /**
25       * Constructor.
26       * @param xml Expected XML document.
27       */
28      public SameXml(final XML xml) {
29          this(xml.toString());
30      }
31  
32      /**
33       * Constructor.
34       * @param expected Expected XML document.
35       */
36      public SameXml(final String expected) {
37          this.expected = expected;
38      }
39  
40      @Override
41      public boolean matchesSafely(final String item) {
42          return new WithoutLines(new XMLDocument(this.expected)).value()
43              .equals(new WithoutLines(new XMLDocument(item)).value());
44      }
45  
46      @Override
47      public void describeTo(final Description description) {
48          description.appendText("XML documents is not the same.")
49              .appendText("Expected:\n")
50              .appendText(this.expected);
51      }
52  }
