5   package org.eolang.jeo.matchers;
6   
7   import com.jcabi.xml.XML;
8   import com.jcabi.xml.XMLDocument;
9   import org.cactoos.Scalar;
10  import org.xembly.Directives;
11  import org.xembly.ImpossibleModificationException;
12  import org.xembly.Xembler;
13  
14  /**
15   * The same XML, but without lines and comments.
16   *
17   * @since 0.6
18   */
19  public final class WithoutLines implements Scalar<XML> {
20  
21      /**
22       * The original.
23       */
24      private final XML original;
25  
26      /**
27       * Constructor.
28       * @param orgnl Original XML
29       */
30      public WithoutLines(final XML orgnl) {
31          this.original = orgnl;
32      }
33  
34      @Override
35      public XML value() {
36          try {
37              return new XMLDocument(
38                  new Xembler(
39                      new Directives()
40                          .xpath(".//o[@line]/@line").remove()
41                          .xpath("//comment()").remove()
42                  ).apply(this.original.inner())
43              );
44          } catch (final ImpossibleModificationException exception) {
45              throw new IllegalStateException(
46                  "Failed to remove 'line' attributes", exception
47              );
48          }
49      }
50  }
