5   package org.eolang.jeo;
6   
7   import java.io.IOException;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.util.stream.Stream;
11  import org.eolang.jeo.representation.xmir.JcabiXmlDoc;
12  
13  /**
14   * EO objects.
15   * Reads all EO objects from the folder.
16   *
17   * @since 0.1.0
18   */
19  final class XmirFiles {
20  
21      /**
22       * Where to read objects from.
23       * Usually it's a folder with the name "generated-sources".
24       * See <a href="https://maven.apache.org/guides/mini/guide-generating-sources.html">generated-sources</a>.
25       */
26      private final Path root;
27  
28      /**
29       * Constructor.
30       * @param xmirs Where to read objects from.
31       */
32      XmirFiles(final Path xmirs) {
33          this.root = xmirs;
34      }
35  
36      /**
37       * All representations.
38       * @return All representations.
39       */
40      public Stream<Path> all() {
41          final Path path = this.root;
42          try {
43              return Files.walk(path).filter(Files::isRegularFile);
44          } catch (final IOException exception) {
45              throw new IllegalStateException(
46                  String.format("Can't read folder '%s'", path),
47                  exception
48              );
49          }
50      }
51  
52      /**
53       * Verify all the xmir files.
54       */
55      public void verify() {
56          this.all().map(JcabiXmlDoc::new).forEach(JcabiXmlDoc::validate);
57      }
58  }
