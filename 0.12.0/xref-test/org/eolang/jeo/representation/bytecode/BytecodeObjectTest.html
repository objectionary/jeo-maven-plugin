5   package org.eolang.jeo.representation.bytecode;
6   
7   import com.jcabi.xml.XMLDocument;
8   import org.eolang.jeo.representation.directives.DirectivesObject;
9   import org.eolang.jeo.representation.directives.HasClass;
10  import org.eolang.jeo.representation.directives.HasMethod;
11  import org.hamcrest.MatcherAssert;
12  import org.junit.jupiter.api.Disabled;
13  import org.junit.jupiter.api.Test;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Xembler;
16  
17  /**
18   * Test case for {@link BytecodeObject}.
19   * @since 0.6
20   * @todo #1130:30min Find a way to test {@link BytecodeObject#directives(String)}.
21   *  When you find a way to test it, remove the @Disabled annotation from
22   *  {@link #convertsSimpleClassWithMethodToXmir()} test.
23   *  Most probably it will require to remove such classes as {@link HasClass}.
24   */
25  final class BytecodeObjectTest {
26  
27      @Test
28      void convertsSimpleClassWithoutConstructorToXmir() throws ImpossibleModificationException {
29          MatcherAssert.assertThat(
30              "Can't parse simple class without constructor",
31              new Xembler(new BytecodeObject(new BytecodeClass()).directives("")).xml(),
32              new HasClass("Simple")
33          );
34      }
35  
36      @Test
37      @Disabled
38      void convertsSimpleClassWithMethodToXmir() throws ImpossibleModificationException {
39          final String clazz = "WithMethod";
40          final String xml = new Xembler(
41              new BytecodeObject(
42                  new BytecodeClass(clazz)
43                      .helloWorldMethod()
44              ).directives("")
45          ).xml();
46          MatcherAssert.assertThat(
47              String.format(
48                  "Can't parse simple class with method, result is: '%s'",
49                  new XMLDocument(xml)
50              ),
51              xml,
52              new HasMethod("main").inside(clazz)
53          );
54      }
55  
56      @Test
57      void convertsClassWithPackageToXmir() throws ImpossibleModificationException {
58          final String name = "ClassInPackage";
59          final DirectivesObject directives = new BytecodeObject(
60              "some/package",
61              new BytecodeClass(String.format("some/package/%s", name)).helloWorldMethod()
62          ).directives("");
63          final String xml = new Xembler(directives).xml();
64          final String pckg = "some.package";
65          MatcherAssert.assertThat(
66              String.format(
67                  "Can't parse '%s' class that placed under package '%s', result is: %n%s%n",
68                  name,
69                  pckg,
70                  new XMLDocument(xml)
71              ),
72              xml,
73              new HasClass(name).inside(pckg)
74          );
75      }
76  }
