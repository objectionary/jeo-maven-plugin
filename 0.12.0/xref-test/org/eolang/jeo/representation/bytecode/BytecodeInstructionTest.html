7   import org.eolang.jeo.representation.directives.DirectivesInstruction;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  import org.objectweb.asm.Opcodes;
12  import org.objectweb.asm.Type;
13  import org.xembly.Directive;

15  import org.xembly.ImpossibleModificationException;
16  import org.xembly.Xembler;

18  /**
19   * Test cases for {@link BytecodeInstruction}.
20   * @since 0.11.0
21   */
22  final class BytecodeInstructionTest {
23  
24      @Test
25      void covertsInstructionWithTypeToDirectives() throws ImpossibleModificationException {
26          MatcherAssert.assertThat(
27              "We expect that the bytecode instruction argument with type 'Type' will be wrapped in sting, see https://github.com/objectionary/jeo-maven-plugin/issues/1125",
28              new Xembler(
29                  BytecodeInstructionTest.withoutNames(
30                      new BytecodeInstruction(
31                          Opcodes.LDC,
32                          Type.getType(Integer.class)
33                      ).directives()
34                  )
35              ).xml(),
36              Matchers.equalTo(
37                  new Xembler(
38                      BytecodeInstructionTest.withoutNames(
39                          new DirectivesInstruction(
40                              Opcodes.LDC,
41                              Type.getType(Integer.class)
42                          )
43                      )
44                  ).xml()
45              )
46          );
47      }
48  
49      /**
50       * Set the name attribute to empty for all elements in the directives.
51       * This was added to compare generated XMIR independently of the names, generated
52       * randomly by {@link org.eolang.jeo.representation.directives.RandName}.
53       * <p>
54       *     <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1156">Issue #1156</a>
55       * </p>
56       * @param directives Directives to process
57       * @return Directives with empty names
58       */
59      private static Directives withoutNames(final Iterable<Directive> directives) {
60          return new Directives(directives).xpath("//o").attr("name", "");
61      }
62  }
