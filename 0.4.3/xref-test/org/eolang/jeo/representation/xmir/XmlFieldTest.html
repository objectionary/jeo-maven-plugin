24  package org.eolang.jeo.representation.xmir;
25  
26  import org.eolang.jeo.representation.directives.DirectivesField;
27  import org.hamcrest.MatcherAssert;
28  import org.hamcrest.Matchers;
29  import org.junit.jupiter.api.Test;
30  import org.objectweb.asm.Opcodes;
31  import org.xembly.ImpossibleModificationException;
32  import org.xembly.Xembler;
33  
34  /**
35   * Test case for {@link XmlField}.
36   *
37   * @since 0.3
38   */
39  final class XmlFieldTest {
40  
41      @Test
42      void parsesXmirFieldSuccessfully() throws ImpossibleModificationException {
43          final int access = Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
44          final String name = "serialVersionUID";
45          final String descriptor = "J";
46          final long value = 7_099_057_708_183_571_937L;
47          final XmlField field = new XmlField(
48              new XmlNode(
49                  new Xembler(
50                      new DirectivesField(
51                          access,
52                          name,
53                          descriptor,
54                          "",
55                          value
56                      )
57                  ).xml()
58              )
59          );
60          MatcherAssert.assertThat(
61              String.format(
62                  "Failed to parse XMIR field. Expected name '%s', actual '%s'. Expected descriptor '%s', actual '%s'. Expected value '%s', actual '%s'",
63                  name,
64                  field.name(),
65                  descriptor,
66                  field.descriptor(),
67                  value,
68                  field.value()
69              ),
70              name.equals(field.name())
71                  && descriptor.equals(field.descriptor())
72                  && field.value().equals(value),
73              Matchers.is(true)
74          );
75      }
76  
77      @Test
78      void parsesFieldWithStringValue() throws ImpossibleModificationException {
79          final String expected = "7099057708183571937";
80          MatcherAssert.assertThat(
81              "Failed to parse XMIR field",
82              new XmlField(
83                  new XmlNode(
84                      new Xembler(
85                          new DirectivesField(
86                              Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL,
87                              "serialVersionUID",
88                              "Ljava/lang/String;",
89                              "",
90                              expected
91                          )
92                      ).xml()
93                  )
94              ).value(),
95              Matchers.equalTo(expected)
96          );
97      }
98  
99      @Test
100     void parsesFieldWithEmptyStringValue() throws ImpossibleModificationException {
101         final String expected = "";
102         MatcherAssert.assertThat(
103             "Failed to parse XMIR field with empty initial value",
104             new XmlField(
105                 new XmlNode(
106                     new Xembler(
107                         new DirectivesField(
108                             Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL,
109                             "serialVersionUID",
110                             "Ljava/lang/String;",
111                             "",
112                             expected
113                         )
114                     ).xml()
115                 )
116             ).value(),
117             Matchers.equalTo(expected)
118         );
119     }
120 
121     @Test
122     void parsesFieldWithNullableStringValue() throws ImpossibleModificationException {
123         MatcherAssert.assertThat(
124             "Failed to parse XMIR field with null initial value",
125             new XmlField(
126                 new XmlNode(
127                     new Xembler(
128                         new DirectivesField(
129                             Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL,
130                             "serialVersionUID",
131                             "Ljava/lang/String;",
132                             "",
133                             null
134                         )
135                     ).xml()
136                 )
137             ).value(),
138             Matchers.nullValue()
139         );
140     }
