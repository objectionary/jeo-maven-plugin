5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.List;
8   import java.util.stream.Collectors;
9   import lombok.EqualsAndHashCode;
10  import lombok.ToString;
11  import org.eolang.jeo.representation.OpcodeDictionary;
12  import org.eolang.jeo.representation.bytecode.BytecodeInstruction;
13  import org.eolang.jeo.representation.directives.DirectivesInstruction;
14  import org.eolang.jeo.representation.directives.Format;
15  import org.xembly.Transformers;
16  import org.xembly.Xembler;
17  
18  /**
19   * Bytecode instruction from XML.
20   * @since 0.1
21   */
22  @ToString
23  @EqualsAndHashCode
24  public final class XmlInstruction implements XmlBytecodeEntry {
25  
26      /**
27       * Opcode dictionary.
28       */
29      private static final OpcodeDictionary DICTIONARY = new OpcodeDictionary();
30  
31      /**
32       * Instruction node.
33       */
34      @EqualsAndHashCode.Exclude
35      private final XmlJeoObject node;
36  
37      /**
38       * Constructor.
39       * @param index Index of the instruction among other instructions.
40       * @param opcode Opcode.
41       * @param args Arguments.
42       */
43      XmlInstruction(final int index, final int opcode, final Object... args) {
44          this(
45              new NativeXmlNode(
46                  new Xembler(
47                      new DirectivesInstruction(index, new Format(), opcode, args),
48                      new Transformers.Node()
49                  ).xmlQuietly()
50              )
51          );
52      }
53  
54      /**
55       * Constructor.
56       * @param xmlnode Instruction node.
57       */
58      XmlInstruction(final XmlNode xmlnode) {
59          this(new XmlJeoObject(xmlnode));
60      }
61  
62      /**
63       * Constructor.
64       * @param node XML Jeo object node representing the instruction.
65       */
66      private XmlInstruction(final XmlJeoObject node) {
67          this.node = node;
68      }
69  
70      /**
71       * Convert to bytecode.
72       * @return Bytecode instruction.
73       */
74      public BytecodeInstruction bytecode() {
75          return new BytecodeInstruction(
76              this.opcode(),
77              this.operands().stream().map(XmlOperand::asObject).toArray()
78          );
79      }
80  
81      /**
82       * Instruction code.
83       * @return Code.
84       */
85      @EqualsAndHashCode.Include
86      private int opcode() {
87          return this.node.base()
88              .map(s -> s.substring(s.lastIndexOf('.') + 1))
89              .map(XmlInstruction.DICTIONARY::code)
90              .orElseThrow(
91                  () -> new IllegalStateException(
92                      String.format("base is not found in node %s", this.node)
93                  )
94              );
95      }
96  
97      /**
98       * Instruction arguments.
99       * @return Arguments.
100      */
101     @EqualsAndHashCode.Include
102     private List<XmlOperand> operands() {
103         return this.node
104             .children()
105             .map(XmlOperand::new)
106             .collect(Collectors.toList());
107     }
108 }
