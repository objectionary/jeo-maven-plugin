5   package org.eolang.jeo;
6   
7   import java.nio.file.Path;
8   import java.nio.file.Paths;
9   import java.util.regex.Pattern;
10  import org.eolang.jeo.representation.XmirRepresentation;
11  
12  /**
13   * Assembling transformation.
14   *
15   * <p>This class implements the transformation process that converts XMIR
16   * representations into Java bytecode. It reads XMIR files and produces
17   * corresponding .class files in the specified target directory.</p>
18   * @since 0.6.0
19   */
20  public final class Assembling implements Transformation {
21  
22      /**
23       * XMIR file extension pattern.
24       */
25      private static final Pattern XMIR = Pattern.compile(".xmir", Pattern.LITERAL);
26  
27      /**
28       * Source folder where all the XMIR representations are located.
29       */
30      private final Path from;
31  
32      /**
33       * Target folder where to save the assembled class.
34       */
35      private final Path tgt;
36  
37      /**
38       * Representation to assemble.
39       */
40      private final Path xmir;
41  
42      /**
43       * Constructor.
44       * @param source Source folder where all the XMIR representations are located
45       * @param target Target folder where the assembled class will be saved
46       * @param representation Path to the XMIR representation to assemble
47       */
48      Assembling(final Path source, final Path target, final Path representation) {
49          this.from = source;
50          this.tgt = target;
51          this.xmir = representation;
52      }
53  
54      @Override
55      public Path source() {
56          return this.xmir;
57      }
58  
59      @Override
60      public Path target() {
61          return Paths.get(
62              this.tgt.toString(),
63              Assembling.XMIR.matcher(this.from.relativize(this.xmir).toString()).replaceAll(".class")
64          );
65      }
66  
67      @Override
68      public byte[] transform() {
69          return new XmirRepresentation(this.xmir).toBytecode().bytes();
70      }
71  }
