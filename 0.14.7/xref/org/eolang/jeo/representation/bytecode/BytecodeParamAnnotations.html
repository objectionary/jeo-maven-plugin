5   package org.eolang.jeo.representation.bytecode;
6   
7   import lombok.EqualsAndHashCode;
8   import lombok.ToString;
9   import org.eolang.jeo.representation.directives.Format;
10  import org.objectweb.asm.MethodVisitor;
11  import org.xembly.Directive;
12  
13  /**
14   * Bytecode parameter annotations.
15   * @since 0.15.0
16   */
17  @ToString
18  @EqualsAndHashCode
19  public final class BytecodeParamAnnotations {
20  
21      /**
22       * Parameter index.
23       */
24      private final int index;
25  
26      /**
27       * Parameter annotations.
28       */
29      private final BytecodeAnnotations annotations;
30  
31      /**
32       * Constructor.
33       * @param index Parameter index.
34       * @param annotations Parameter annotations.
35       */
36      public BytecodeParamAnnotations(final int index, final BytecodeAnnotations annotations) {
37          this.index = index;
38          this.annotations = annotations;
39      }
40  
41      /**
42       * Write all parameter annotations to bytecode.
43       * @param visitor Method to write in.
44       */
45      public void write(final MethodVisitor visitor) {
46          this.annotations.write(this.index, visitor);
47      }
48  
49      /**
50       * Convert to directives.
51       * @param format Directive format.
52       * @return Xmir Directives.
53       */
54      Iterable<Directive> directives(final Format format) {
55          return this.annotations.directives(
56              format, String.format("param-annotations-%d", this.index)
57          );
58      }
59  }
