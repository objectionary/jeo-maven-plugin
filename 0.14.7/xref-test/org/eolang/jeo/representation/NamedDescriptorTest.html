5   package org.eolang.jeo.representation;
6   
7   import java.util.stream.Stream;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.params.ParameterizedTest;
11  import org.junit.jupiter.params.provider.Arguments;
12  import org.junit.jupiter.params.provider.MethodSource;
13  
14  /**
15   * Test for {@link NamedDescriptor}.
16   * @since 0.5
17   */
18  final class NamedDescriptorTest {
19  
20      @ParameterizedTest(name = "Encodes \"{0}\" name with \"{1}\" descriptor, should be \"{2}\"")
21      @MethodSource("namesDescriptorsAndEncoded")
22      void encodesNamesWithDescriptors(final String name, final String descr, final String encoded) {
23          MatcherAssert.assertThat(
24              "Encoded method name and descriptor should be correct",
25              new NamedDescriptor(name, descr).encoded(),
26              Matchers.equalTo(encoded)
27          );
28      }
29  
30      @ParameterizedTest(name = "Decodes \"{2}\" with \"{1}\" descriptor, should be \"{0}\"")
31      @MethodSource("namesDescriptorsAndEncoded")
32      void decodesNames(final String name, final String descr, final String encoded) {
33          MatcherAssert.assertThat(
34              "Decoded method name should be correct",
35              new NamedDescriptor(encoded).name(),
36              Matchers.equalTo(name)
37          );
38      }
39  
40      @ParameterizedTest(name = "Decodes \"{2}\" with \"{1}\" descriptor")
41      @MethodSource("namesDescriptorsAndEncoded")
42      void decodesDescriptors(final String name, final String descr, final String encoded) {
43          MatcherAssert.assertThat(
44              "Decoded method descriptor should be correct",
45              new NamedDescriptor(encoded).descriptor(),
46              Matchers.equalTo(descr)
47          );
48      }
49  
50      /**
51       * Provides names, descriptors and encoded values.
52       * This method is used by {@link #encodesNamesWithDescriptors(String, String, String)},
53       * {@link #decodesNames(String, String, String)}
54       * and {@link #decodesDescriptors(String, String, String)}.
55       * @return Arguments for tests.
56       */
57      @SuppressWarnings("PMD.UnusedPrivateMethod")
58      private static Stream<Arguments> namesDescriptorsAndEncoded() {
59          return Stream.of(
60              Arguments.of("foo", "()I", "foo-%28%29I"),
61              Arguments.of("@init@", "()V", "@init@-%28%29V"),
62              Arguments.of("bar", "(Ljava/lang/String;)V", "bar-%28Ljava%2Flang%2FString%3B%29V"),
63              Arguments.of(
64                  "baz",
65                  "(Ljava/lang/String;Ljava/lang/String;)V",
66                  "baz-%28Ljava%2Flang%2FString%3BLjava%2Flang%2FString%3B%29V"
67              )
68          );
69      }
70  }
