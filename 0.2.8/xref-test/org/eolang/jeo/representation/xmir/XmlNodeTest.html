24  package org.eolang.jeo.representation.xmir;
25  
26  import java.util.List;
27  import java.util.Optional;
28  import java.util.stream.Collectors;
29  import org.hamcrest.MatcherAssert;
30  import org.hamcrest.Matchers;
31  import org.junit.jupiter.api.Test;
32  
33  /**
34   * Test case for {@link XmlNode}.
35   * @since 0.1
36   */
37  class XmlNodeTest {
38  
39      @Test
40      void retrievesTheFirstChild() {
41          MatcherAssert.assertThat(
42              "Can't retrieve the first child, or the first child is not the expected one",
43              new XmlNode("<o><o name='inner'/></o>").firstChild(),
44              Matchers.equalTo(new XmlNode("<o name='inner'/>"))
45          );
46      }
47  
48      @Test
49      void retrievesChildren() {
50          final List<XmlNode> children = new XmlNode("<o><o name='inner1'/><o name='inner2'/></o>")
51              .children().collect(Collectors.toList());
52          MatcherAssert.assertThat(
53              "Size of children is not as expected",
54              children,
55              Matchers.hasSize(2)
56          );
57          MatcherAssert.assertThat(
58              "Can't retrieve the children, or the children are not the expected ones",
59              children,
60              Matchers.contains(
61                  new XmlNode("<o name='inner1'/>"),
62                  new XmlNode("<o name='inner2'/>")
63              )
64          );
65      }
66  
67      @Test
68      void retrievesAttribute() {
69          final Optional<String> attribute = new XmlNode("<o name='some'/>").attribute("name");
70          MatcherAssert.assertThat(
71              "Can't retrieve the attribute",
72              attribute.isPresent(),
73              Matchers.is(true)
74          );
75          MatcherAssert.assertThat(
76              "he attribute is not the expected one",
77              attribute.get(),
78              Matchers.equalTo("some")
79          );
80      }
81  
82      @Test
83      void retrievesText() {
84          MatcherAssert.assertThat(
85              "Can't retrieve the text, or the text is not the expected one",
86              new XmlNode("<o>text</o>").text(),
87              Matchers.equalTo("text")
88          );
89      }
90  }
