24  package org.eolang.jeo.representation.directives;
25  
26  import com.jcabi.matchers.XhtmlMatchers;
27  import org.eolang.jeo.representation.HexData;
28  import org.hamcrest.MatcherAssert;
29  import org.hamcrest.Matchers;
30  import org.junit.jupiter.api.Test;
31  import org.xembly.ImpossibleModificationException;
32  import org.xembly.Xembler;
33  
34  /**
35   * Test case for {@link DirectivesAnnotation}.
36   * @since 0.1
37   */
38  class DirectivesAnnotationsTest {
39  
40      @Test
41      void returnsEmptyDirectviesIfNoAnnotations() {
42          MatcherAssert.assertThat(
43              "Must return empty directives if no annotations",
44              new DirectivesAnnotations().iterator().hasNext(),
45              Matchers.is(false)
46          );
47      }
48  
49      @Test
50      void returnsSingleAnnotation() throws ImpossibleModificationException {
51          final String annotation = "Ljava/lang/Override;";
52          final String xml = new Xembler(
53              new DirectivesAnnotations()
54                  .add(new DirectivesAnnotation(annotation, true))
55          ).xml();
56          MatcherAssert.assertThat(
57              String.format(
58                  "Must return single annotation with correct descriptor and visibility, but was: %n%s",
59                  xml
60              ),
61              xml,
62              Matchers.allOf(
63                  XhtmlMatchers.hasXPath(
64                      "/o[@base='tuple' and @name='annotations']/o"
65                  ),
66                  XhtmlMatchers.hasXPath(
67                      String.format(
68                          "/o[@base='tuple' and @name='annotations']/o/o[@base='string' and @name='descriptor' and text()='%s']",
69                          new HexData(annotation).value()
70                      )
71                  ),
72                  XhtmlMatchers.hasXPath(
73                      String.format(
74                          "/o[@base='tuple' and @name='annotations']/o/o[@base='bool' and @name='visible' and text()='%s']",
75                          new HexData(true).value()
76                      )
77                  )
78              )
79          );
80      }
81  }
