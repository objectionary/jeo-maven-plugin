5   package org.eolang.jeo.representation.bytecode;
6   
7   import java.nio.charset.StandardCharsets;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.params.ParameterizedTest;
11  import org.junit.jupiter.params.provider.MethodSource;
12  
13  /**
14   * Test cases for {@link PlainLongCodec}.
15   * @since 0.8
16   */
17  final class PlainLongCodecTest {
18  
19      @ParameterizedTest
20      @MethodSource("mapping")
21      void encodesSuccessfully(final Object value, final DataType type, final byte[] bytes) {
22          MatcherAssert.assertThat(
23              "Can't encode value to the correct byte array using PlainLongCodec",
24              new PlainLongCodec(new EoCodec()).encode(value, type),
25              Matchers.equalTo(bytes)
26          );
27      }
28  
29      @ParameterizedTest
30      @MethodSource("mapping")
31      void decodesSuccessfully(final Object value, final DataType type, final byte[] bytes) {
32          MatcherAssert.assertThat(
33              "Can't decode byte array to the correct value using PlainLongCodec",
34              new PlainLongCodec(new EoCodec()).decode(bytes, type),
35              Matchers.equalTo(value)
36          );
37      }
38  
39      /**
40       * Test cases.
41       * @return Arguments.
42       */
43      private static Object[] mapping() {
44          return new Object[]{
45              new Object[]{0L, DataType.LONG, new byte[]{0, 0, 0, 0, 0, 0, 0, 0}},
46              new Object[]{1L, DataType.LONG, new byte[]{0, 0, 0, 0, 0, 0, 0, 1}},
47              new Object[]{-1L, DataType.LONG, new byte[]{-1, -1, -1, -1, -1, -1, -1, -1}},
48              new Object[]{42L, DataType.LONG, new byte[]{0, 0, 0, 0, 0, 0, 0, 42}},
49              new Object[]{-42L, DataType.LONG, new byte[]{-1, -1, -1, -1, -1, -1, -1, -42}},
50              new Object[]{0.0d, DataType.DOUBLE, new byte[]{0, 0, 0, 0, 0, 0, 0, 0}},
51              new Object[]{1.0d, DataType.DOUBLE, new byte[]{63, -16, 0, 0, 0, 0, 0, 0}},
52              new Object[]{-1.0d, DataType.DOUBLE, new byte[]{-65, -16, 0, 0, 0, 0, 0, 0}},
53              new Object[]{42.0d, DataType.DOUBLE, new byte[]{64, 69, 0, 0, 0, 0, 0, 0}},
54              new Object[]{-42.0d, DataType.DOUBLE, new byte[]{-64, 69, 0, 0, 0, 0, 0, 0}},
55              new Object[]{"0", DataType.STRING, "0".getBytes(StandardCharsets.UTF_8)},
56              new Object[]{"1", DataType.STRING, "1".getBytes(StandardCharsets.UTF_8)},
57              new Object[]{"-1", DataType.STRING, "-1".getBytes(StandardCharsets.UTF_8)},
58              new Object[]{"42", DataType.STRING, "42".getBytes(StandardCharsets.UTF_8)},
59          };
60      }
