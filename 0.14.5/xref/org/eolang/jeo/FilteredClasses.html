5   package org.eolang.jeo;
6   
7   import com.jcabi.log.Logger;
8   import java.nio.file.Path;
9   import java.util.List;
10  import java.util.function.Consumer;
11  import java.util.stream.Collectors;
12  import java.util.stream.Stream;
13  
14  /**
15   * Filtered classes.
16   * @since 0.14.0
17   */
18  final class FilteredClasses implements Classes {
19  
20      /**
21       * Original classes.
22       */
23      private final Classes original;
24  
25      /**
26       * Glob filter to apply.
27       */
28      private final GlobFilter filter;
29  
30      /**
31       * Logger to use for logging messages.
32       */
33      private final Consumer<String> logger;
34  
35      /**
36       * Constructor.
37       * @param original Original classes to filter
38       * @param filter Glob filter to apply
39       */
40      FilteredClasses(final Classes original, final GlobFilter filter) {
41          this(original, filter, s -> Logger.info(FilteredClasses.class, s));
42      }
43  
44      /**
45       * Constructor with custom logger.
46       * @param original Original classes to filter
47       * @param filter Glob filter to apply
48       * @param logger Logger to use for logging messages
49       */
50      FilteredClasses(
51          final Classes original,
52          final GlobFilter filter,
53          final Consumer<String> logger
54      ) {
55          this.original = original;
56          this.filter = filter;
57          this.logger = logger;
58      }
59  
60      @Override
61      public Path root() {
62          return this.original.root();
63      }
64  
65      @Override
66      public Stream<Path> all() {
67          final List<Path> res = this.original.all().filter(this.filter).collect(Collectors.toList());
68          final int size = res.size();
69          this.logger.accept(
70              String.format("Found %d files in %s using %s", size, this.original, this.filter)
71          );
72          return res.stream();
73      }
74  
75      @Override
76      public String toString() {
77          return this.original.toString();
78      }
79  }
