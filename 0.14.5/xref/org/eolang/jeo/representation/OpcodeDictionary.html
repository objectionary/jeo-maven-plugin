13  /**
14   * Opcode dictionary.
15   * This class provides a mapping between opcode numbers and their corresponding names,
16   * allowing for easy retrieval of opcode names by their numbers and vice versa.
17   * @since 0.12.0
18   */
19  public final class OpcodeDictionary {
20  
21      /**
22       * Default opcode names and codes.
23       */
24      private static final Map<Integer, String> NAMES = OpcodeDictionary.opcodeNames();
25  
26      /**
27       * Default opcode codes.
28       */
29      private static final Map<String, Integer> CODES = OpcodeDictionary.nameOpcodes();
30  
31      /**
32       * Dictionary of opcode names.
33       */
34      private final Map<Integer, String> names;
35  
36      /**
37       * Dictionary of opcode codes.
38       */
39      private final Map<String, Integer> codes;
40  
41      /**
42       * Constructor for creating an instance with default opcode names and codes.
43       */
44      public OpcodeDictionary() {
45          this(OpcodeDictionary.NAMES, OpcodeDictionary.CODES);
46      }
47  
48      /**
49       * Constructor for creating an instance with custom maps.
50       * @param names Map of opcode numbers to their names
51       * @param codes Map of opcode names to their corresponding numbers
52       */
53      private OpcodeDictionary(final Map<Integer, String> names, final Map<String, Integer> codes) {
54          this.names = names;
55          this.codes = codes;
56      }
57  
58      /**
59       * Retrieves the opcode name for a given opcode number.
60       * @param opcode Opcode number
61       * @return Name of the opcode
62       */
63      public String name(final int opcode) {
64          final String result;
65          if (this.names.containsKey(opcode)) {
66              result = this.names.get(opcode);
67          } else {
68              throw new IllegalStateException(
69                  String.format("Unknown opcode: %d", opcode)
70              );
71          }
72          return result;
73      }
74  
75      /**
76       * Retrieves the opcode name for a given opcode number.
77       * @param name Name of the opcode
78       * @return Opcode number corresponding to the name
79       */
80      public int code(final String name) {
81          final int result;
82          final String key = name.toLowerCase(Locale.ROOT);
83          if (this.codes.containsKey(key)) {
84              result = this.codes.get(key);
85          } else {
86              throw new IllegalStateException(String.format("Unknown opcode name: %s", name));
87          }
88          return result;
89      }
90  
91      /**
92       * Returns a map of opcode numbers to their corresponding names.
93       * @return Map of opcode numbers to names
94       */
95      private static Map<Integer, String> opcodeNames() {
96          return Collections.unmodifiableMap(
97              OpcodeDictionary.entries().collect(
98                  Collectors.toMap(
99                      Map.Entry::getKey,
100                     Map.Entry::getValue,
101                     (old, fresh) -> fresh
102                 )
103             )
104         );
105     }
106 
107     /**
108      * Returns a map of opcode names to their corresponding integer values.
109      * @return Map of opcode names to their integer values
110      */
111     private static Map<String, Integer> nameOpcodes() {
112         return Collections.unmodifiableMap(
113             OpcodeDictionary.entries().collect(
114                 Collectors.toMap(
115                     Map.Entry::getValue,
116                     Map.Entry::getKey,
117                     (old, fresh) -> fresh
118                 )
119             )
120         );
121     }

290     /**
291      * Pair of opcode and its name.
292      * @since 0.12.0
293      */
294     private static class Pair implements Map.Entry<Integer, String> {
295         /**
296          * Opcode key number.
297          */
298         private final int key;
299 
300         /**
301          * Opcode name.
302          */
303         private final String name;
304 
305         /**
306          * Constructor for creating a pair of opcode and its name.
307          * @param key Opcode key number
308          * @param value Opcode name
309          */
310         Pair(final int key, final String value) {
311             this.key = key;
312             this.name = value;
313         }
314 
315         @Override
316         public Integer getKey() {
317             return this.key;
318         }
319 
320         @Override
321         public String getValue() {
322             return this.name;
323         }
324 
325         @Override
326         public String setValue(final String value) {
327             throw new UnsupportedOperationException("setValue is not supported for Pair entry");
328         }
329     }
