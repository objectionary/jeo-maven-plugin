5   package org.eolang.jeo.representation;

7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.cactoos.bytes.BytesOf;
9   import org.cactoos.io.ResourceOf;
10  import org.eolang.jeo.representation.bytecode.Bytecode;
11  import org.eolang.jeo.representation.bytecode.EoCodec;
12  import org.eolang.jeo.representation.directives.DirectivesValue;
13  import org.eolang.jeo.representation.directives.Format;
14  import org.hamcrest.MatcherAssert;
15  import org.hamcrest.Matchers;
16  import org.junit.jupiter.api.Assertions;
17  import org.junit.jupiter.api.Test;
18  
19  /**
20   * Test cases for {@link BytecodeRepresentation}.
21   * This class verifies the parsing and conversion of Java bytecode
22   * to EO representation, including name extraction and XML generation.
23   *
24   * @since 0.1.0
25   */
26  final class BytecodeRepresentationTest {
27  
28      /**
29       * Bytecode example that contains a double value in the stack.
30       */
31      private static final String EXAMPLE_BYTECODE = "Example.class";
32  
33      /**
34       * The name of the resource with the simplest class.
35       */
36      private static final String METHOD_BYTE = "MethodByte.class";
37  
38      /**
39       * The example of bytecode with a nullable param name.
40       * You can read more about it here:
41       * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1233">#1233</a>
42       */
43      private static final String NULL_NAME = "LtIncorrectUnlint.class";
44  
45      @Test
46      void parsesBytecode() {
47          MatcherAssert.assertThat(
48              "The simplest class should contain the object with MethodByte name",
49              new BytecodeRepresentation(new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE))
50                  .toXmir().xpath("/object/o/@name").get(0),
51              Matchers.equalTo(new PrefixedName("MethodByte").encode())
52          );
53      }
54  
55      @Test
56      void generatesXmlWithCorrectObjectAttributes() {
57          MatcherAssert.assertThat(
58              "XMIR objects should have either 'name' or 'base' attribute, or both",
59              new BytecodeRepresentation(
60                  new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE)
61              ).toXmir().toString(),
62              Matchers.not(
63                  XhtmlMatchers.hasXPath(
64                      "//o[not(@base) and not(@name) and not(parent::*[@base='Q.org.eolang.bytes'])]"
65                  )
66              )
67          );
68      }
69  
70      /**
71       * This test was added to mitigate the issue:
72       * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1160">Issue #1160</a>.
73       */
74      @Test
75      void generatesXmlWithFormationsThatCIncludeObjectsWithNames() {
76          MatcherAssert.assertThat(
77              "A formation must only include objects with names.",
78              new BytecodeRepresentation(
79                  new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE)
80              ).toXmir().toString(),
81              Matchers.not(XhtmlMatchers.hasXPath("//o[not(@base)]/o[not(@name)]"))
82          );
83      }
84  
85      @Test
86      void retrievesName() {
87          final ResourceOf input = new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE);
88          final String actual = new BytecodeRepresentation(input).name();
89          final String expected = "org/eolang/jeo/MethodByte";
90          MatcherAssert.assertThat(
91              String.format(
92                  "The name should be retrieved without exceptions and equal to the expected '%s'",
93                  expected
94              ),
95              actual,
96              Matchers.equalTo(expected)
97          );
98      }
99  
100     /**
101      * This test checks that the bytecode representation have a stack with a double value.
102      * The test was added to mitigate the issue:
103      * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1211:">Issue #1211</a>.
104      */
105     @Test
106     void parsesBytecodeWithDoubleStack() {
107         MatcherAssert.assertThat(
108             "we expect to find a frame with a double stack",
109             new BytecodeRepresentation(
110                 new ResourceOf(BytecodeRepresentationTest.EXAMPLE_BYTECODE)
111             ).toXmir(),
112             XhtmlMatchers.hasXPath(
113                 String.format(
114                     "//o[contains(@name, 'stack')]/o[@name='x0']/o/o[text()='%s']",
115                     new DirectivesValue("double").hex(new EoCodec())
116                 )
117             )
118         );
119     }
120 
121     /**
122      * This test was added to mitigate the issue with stack map frames mapping.
123      * You can read more about it here:
124      * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1215">Issue</a>
125      * @throws Exception if something goes wrong
126      */
127     @Test
128     void parsesBytecodeLoggerStackFrames() throws Exception {
129         final ResourceOf input = new ResourceOf("LoggerFactory$DelegatingLogger.class");
130         final String actual = new Bytecode(
131             new XmirRepresentation(
132                 new BytecodeRepresentation(input).toXmir(new Format(Format.MODE, "debug"))
133             ).toBytecode().bytes()
134         ).toString();
135         final String expected = new Bytecode(new BytesOf(input).asBytes()).toString();
136         MatcherAssert.assertThat(
137             "The disassembled/assembled bytecode representation should match the original bytecode",
138             actual,
139             Matchers.equalTo(expected)
140         );
141     }
142 
143     @Test
144     void parsesBytecodeWithNullParameterName() {
145         Assertions.assertDoesNotThrow(
146             () -> new BytecodeRepresentation(
147                 new ResourceOf(BytecodeRepresentationTest.NULL_NAME)
148             ).toXmir(),
149             "We expect to parse a bytecode with a null name without exceptions"
150         );
151     }
152 }
