5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import java.util.Collections;
9   import org.hamcrest.MatcherAssert;
10  import org.junit.jupiter.api.Test;
11  import org.xembly.ImpossibleModificationException;
12  import org.xembly.Xembler;
13  
14  /**
15   * Test cases for {@link DirectivesArrayAnnotationValue}.
16   * This class verifies the generation of array annotation value directives,
17   * ensuring proper handling of array properties and conversions.
18   *
19   * @since 0.6.0
20   */
21  final class DirectivesArrayAnnotationValueTest {
22  
23      @Test
24      void createsArrayProperty() throws ImpossibleModificationException {
25          MatcherAssert.assertThat(
26              "Can't create an array property with a name",
27              new Xembler(
28                  new DirectivesArrayAnnotationValue(
29                      0,
30                      new Format(),
31                      "name",
32                      Collections.singletonList(new DirectivesPlainAnnotationValue())
33                  )
34              ).xml(),
35              XhtmlMatchers.hasXPaths(
36                  new JeoBaseXpath("./o", "annotation-property").toXpath(),
37                  "./o/o[2]/o[1]/o[text()='41-52-52-41-59']"
38              )
39          );
40      }
41  
42  }
