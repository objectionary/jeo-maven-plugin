5   package org.eolang.jeo.representation.bytecode;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Test;
10  
11  /**
12   * Test case for {@link org.eolang.jeo.representation.bytecode.Bytecode}.
13   *
14   * @since 0.1.0
15   */
16  final class BytecodeTest {
17  
18      @Test
19      void retrievesTheSameBytes() {
20          final byte[] expected = {1, 2, 3};
21          MatcherAssert.assertThat(
22              "Bytecode should remain the same as we pass to the constructor",
23              new Bytecode(expected).bytes(),
24              Matchers.equalTo(expected)
25          );
26      }
27  
28      @Test
29      void printsHumanReadableBytecodeDescription() {
30          MatcherAssert.assertThat(
31              "We expect correct and human-readable class description",
32              new BytecodeObject(new BytecodeClass("Classname")).bytecode().toString(),
33              Matchers.allOf(
34                  Matchers.containsString("class version"),
35                  Matchers.containsString("public class Classname")
36              )
37          );
38      }
39  }
