5   package org.eolang.jeo.representation.directives;
6   
7   /**
8    * Modifiers that can be applied to classes, methods, or fields.
9    * Like public, private, static, final, etc.
10   * @since 0.15.0
11   */
12  final class Modifiers {
13  
14      /**
15       * Modifiers value as integer.
16       */
17      private final int all;
18  
19      /**
20       * Constructor.
21       * @param modifier Modifier value as integer.
22       */
23      Modifiers(final int modifier) {
24          this.all = modifier;
25      }
26  
27      /**
28       * Checks if the given flag is set.
29       * @param flag Flag to check.
30       * @return True if the flag is set, false otherwise.
31       * @checkstyle MethodNameCheck (3 lines)
32       */
33      @SuppressWarnings("PMD.ShortMethodName")
34      public boolean is(final int flag) {
35          return (this.all & flag) != 0;
36      }
37  }
