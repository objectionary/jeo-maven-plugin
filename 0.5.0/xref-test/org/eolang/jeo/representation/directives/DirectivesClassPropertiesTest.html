27  import org.hamcrest.MatcherAssert;
28  import org.junit.jupiter.api.Test;
29  import org.xembly.Directives;
30  import org.xembly.ImpossibleModificationException;
31  import org.xembly.Xembler;
32  
33  /**
34   * Test case for {@link org.eolang.jeo.representation.directives.DirectivesClassProperties}.
35   *
36   * @since 0.1.0
37   */
38  final class DirectivesClassPropertiesTest {
39  
40      @Test
41      void createsDirectives() throws ImpossibleModificationException {
42          MatcherAssert.assertThat(
43              "Can't create proper xml",
44              new Xembler(
45                  new Directives()
46                      .add("o")
47                      .append(
48                          new DirectivesClassProperties(
49                              1,
50                              "org/eolang/SomeClass",
51                              "java/lang/Object",
52                              "org/eolang/SomeInterface"
53                          )
54                      ).up()
55              ).xml(),
56              new SameXml(
57                  String.join(
58                      "",
59                      "<?xml version='1.0' encoding='UTF-8'?>\n",
60                      "<o>\n",
61                      "   <o base='int' data='bytes' name='version'>00 00 00 00 00 00 00 34</o>\n",
62                      "   <o base='int' data='bytes' name='access'>00 00 00 00 00 00 00 01</o>\n",
63                      "   <o base='string' data='bytes' name='signature'>6F 72 67 2F 65 6F 6C 61 6E 67 2F 53 6F 6D 65 43 6C 61 73 73</o>\n",
64                      "   <o base='string' data='bytes' name='supername'>6A 61 76 61 2F 6C 61 6E 67 2F 4F 62 6A 65 63 74</o>\n",
65                      "   <o base='tuple' name='interfaces' star=''>\n",
66                      "      <o base='string' data='bytes'>6F 72 67 2F 65 6F 6C 61 6E 67 2F 53 6F 6D 65 49 6E 74 65 72 66 61 63 65</o>\n",
67                      "   </o>\n",
68                      "</o>\n"
69                  )
70              )
71          );
72      }
