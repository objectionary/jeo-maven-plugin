24  package org.eolang.jeo.representation.xmir;
25  
26  import com.jcabi.xml.XMLDocument;
27  import java.util.ArrayList;
28  import java.util.List;
29  import java.util.Objects;
30  import java.util.Optional;
31  import java.util.stream.Stream;
32  import org.eolang.jeo.representation.directives.JeoFqn;
33  import org.w3c.dom.NamedNodeMap;
34  import org.w3c.dom.Node;
35  import org.w3c.dom.NodeList;
36  
37  /**
38   * XML smart element.
39   * Utility class that simplifies work with XML.
40   * @since 0.1
41   */
42  @SuppressWarnings("PMD.TooManyMethods")
43  public final class XmlNode {
44  
45      /**
46       * Parent node.
47       */
48      private final Node node;
49  
50      /**
51       * Constructor.
52       * @param xml XML string.
53       */
54      public XmlNode(final String xml) {
55          this(new XMLDocument(xml).node().getFirstChild());
56      }
57  
58      /**
59       * Constructor.
60       * @param parent Parent node.
61       */
62      public XmlNode(final Node parent) {
63          this.node = parent;
64      }
65  
66      @Override
67      public boolean equals(final Object obj) {
68          final boolean res;
69          if (obj instanceof XmlNode) {
70              res = new XMLDocument(this.node).equals(new XMLDocument(((XmlNode) obj).node));
71          } else {
72              res = false;
73          }
74          return res;
75      }
76  
77      @Override
78      public int hashCode() {
79          return Objects.hashCode(this.node);
80      }
81  
82      @Override
83      public String toString() {
84          return new XMLDocument(this.node).toString();
85      }
86  
87      /**
88       * Get all child nodes.
89       * @return Child nodes.
90       */
91      public Stream<XmlNode> children() {
92          return this.objects().map(XmlNode::new);
93      }
94  
95      /**
96       * Retrieve node text content.
97       * @return Text content.
98       */
99      public String text() {
100         return this.node.getTextContent();
101     }
102 
103     /**
104      * Get attribute.
105      * @param name Attribute name.
106      * @return Attribute.
107      */
108     public Optional<String> attribute(final String name) {
109         final Optional<String> result;
110         final NamedNodeMap attrs = this.node.getAttributes();
111         if (attrs == null) {
112             result = Optional.empty();
113         } else {
114             result = Optional.ofNullable(attrs.getNamedItem(name)).map(Node::getTextContent);
115         }
116         return result;
117     }
118 
119     /**
120      * Get child node.
121      * @param name Child node name.
122      * @return Child node.
123      */
124     XmlNode child(final String name) {
125         return this.optchild(name).orElseThrow(() -> this.notFound(name));
126     }
127 
128     /**
129      * Find elements by xpath.
130      * @param xpath XPath.
131      * @return List of elements.
132      */
133     List<String> xpath(final String xpath) {
134         return new XMLDocument(this.node).xpath(xpath);
135     }
136 
137     /**
138      * Get child node by attribute.
139      * @param attribute Attribute name.
140      * @param value Attribute value.
141      * @return Child node.
142      */
143     XmlNode child(final String attribute, final String value) {
144         return this.children()
145             .filter(xmlnode -> xmlnode.hasAttribute(attribute, value))
146             .findFirst()
147             .orElseThrow(
148                 () -> this.notFound(
149                     String.format("object with attribute %s='%s'", attribute, value)
150                 )
151             );
152     }
153 
154     /**
155      * Get optional child node by attribute.
156      * @param attribute Attribute name.
157      * @param value Attribute value.
158      * @return Child node.
159      */
160     Optional<XmlNode> optchild(final String attribute, final String value) {
161         return this.children()
162             .filter(xmlnode -> xmlnode.hasAttribute(attribute, value))
163             .findFirst();
164     }
165 
166     /**
167      * Get first child.
168      * @return First child node.
169      */
170     XmlNode firstChild() {
171         return this.children().findFirst()
172             .orElseThrow(
173                 () -> new IllegalStateException(
174                     String.format(
175                         "Can't find any child nodes in '%s'",
176                         new XMLDocument(this.node)
177                     )
178                 )
179             );
180     }
181 
182     /**
183      * Check if an attribute exists.
184      * @param name Attribute name.
185      * @param value Attribute value.
186      * @return True if an attribute with specified value exists.
187      */
188     boolean hasAttribute(final String name, final String value) {
189         return this.attribute(name)
190             .map(String::valueOf)
191             .map(val -> val.equals(value))
192             .orElse(false);
193     }
194 
195     /**
196      * Convert to class.
197      * @return Class.
198      */
199     XmlClass toClass() {
200         return new XmlClass(this.node);
201     }
202 
203     /**
204      * Convert to an entry.
205      * @return Bytecode entry.
206      */
207     XmlBytecodeEntry toEntry() {
208         final XmlBytecodeEntry result;
209         final Optional<String> base = this.attribute("base");
210         if (base.isPresent() && new JeoFqn("label").fqn().equals(base.get())) {
211             result = new XmlLabel(this);
212         } else if (base.isPresent() && new JeoFqn("frame").fqn().equals(base.get())) {
213             result = new XmlFrame(this);
214         } else {
215             result = new XmlInstruction(this);
216         }
217         return result;
218     }
219 
220     /**
221      * Get optional child node.
222      * @param name Child node name.
223      * @return Child node.
224      */
225     private Optional<XmlNode> optchild(final String name) {
226         Optional<XmlNode> result = Optional.empty();
227         final NodeList children = this.node.getChildNodes();
228         for (int index = 0; index < children.getLength(); ++index) {
229             final Node current = children.item(index);
230             if (current.getNodeName().equals(name)) {
231                 result = Optional.of(new XmlNode(current));
232                 break;
233             }
234         }
235         return result;
236     }
237 
238     /**
239      * Generate exception if element not found.
240      * @param name Element name.
241      * @return Exception.
242      */
243     private IllegalStateException notFound(final String name) {
244         return new IllegalStateException(
245             String.format(
246                 "Can't find %s in '%s'",
247                 name,
248                 new XMLDocument(this.node)
249             )
250         );
251     }
252 
253     /**
254      * Objects.
255      * @return Stream of class objects.
256      */
257     private Stream<Node> objects() {
258         final NodeList children = this.node.getChildNodes();
259         final List<Node> res = new ArrayList<>(children.getLength());
260         for (int index = 0; index < children.getLength(); ++index) {
261             final Node child = children.item(index);
262             if ("o".equals(child.getNodeName())) {
263                 res.add(child);
264             }
265         }
266         return res.stream();
267     }
268 }
