5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.eolang.jeo.representation.xmir.NativeXmlNode;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.RepeatedTest;
12  import org.junit.jupiter.api.Test;
13  import org.xembly.ImpossibleModificationException;
14  import org.xembly.Xembler;
15  
16  /**
17   * Test cases for {@link DirectivesValues}.
18   * This class verifies the conversion of value arrays to XMIR directives,
19   * ensuring proper handling of multiple values and their serialization.
20   *
21   * @since 0.6.0
22   */
23  final class DirectivesValuesTest {
24  
25      @Test
26      void convertsToXmir() throws ImpossibleModificationException {
27          MatcherAssert.assertThat(
28              "We expect that array of values will be converted to correct Xembly directives",
29              new Xembler(new DirectivesValues("name", "value")).xml(),
30              XhtmlMatchers.hasXPaths(
31                  new JeoBaseXpath("./o", "seq.of1").toXpath(),
32                  "./o[contains(@name,'name')]",
33                  "./o[contains(@name,'name')]/o/o/o[text()='76-61-6C-75-65']"
34              )
35          );
36      }
37  
38      @RepeatedTest(100)
39      void generatesRandomNameWithoutFirstDigit() {
40          MatcherAssert.assertThat(
41              "We expect that the name of the sequence will be generated randomly and will not start with a digit",
42              new NativeXmlNode(
43                  new Xembler(
44                      new DirectivesValues("", "some-value")
45                  ).xmlQuietly()
46              ).attribute("name").orElseThrow(
47                  () -> new IllegalStateException("Name attribute is absent")
48              ),
49              Matchers.matchesRegex("^[^0-9].*")
50          );
51      }
52  }
