5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.stream.Collectors;
8   import org.eolang.jeo.representation.bytecode.BytecodeAttribute;
9   
10  /**
11   * Xml representation of record components.
12   * Maps to {@link BytecodeAttribute.RecordComponents}.
13   * Mirror of {@link org.eolang.jeo.representation.directives.DirectivesRecordComponents}.
14   *
15   * @since 0.15.0
16   */
17  final class XmlRecordComponents {
18  
19      /**
20       * Xml node of record components to parse.
21       */
22      private final XmlJeoObject node;
23  
24      /**
25       * Constructor.
26       * @param xmlnode XML node.
27       */
28      XmlRecordComponents(final XmlNode xmlnode) {
29          this(new XmlJeoObject(xmlnode));
30      }
31  
32      /**
33       * Constructor.
34       * @param node Node to parse.
35       */
36      XmlRecordComponents(final XmlJeoObject node) {
37          this.node = node;
38      }
39  
40      /**
41       * Convert to bytecode attribute.
42       * @return Bytecode attribute.
43       */
44      public BytecodeAttribute bytecode() {
45          final XmlNode seq = this.node.child(0).orElseThrow(
46              () -> new IllegalStateException(
47                  "Record components must contain a sequence as the first child"
48              )
49          );
50          return new BytecodeAttribute.RecordComponents(
51              new XmlSeq(seq).children()
52                  .map(XmlRecordComponent::new)
53                  .map(XmlRecordComponent::bytecode)
54                  .collect(Collectors.toList())
55          );
56      }
57  }
