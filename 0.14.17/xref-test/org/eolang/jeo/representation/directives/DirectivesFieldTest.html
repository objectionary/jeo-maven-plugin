5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.eolang.jeo.representation.bytecode.BytecodeField;
9   import org.hamcrest.MatcherAssert;
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.params.ParameterizedTest;
12  import org.junit.jupiter.params.provider.ValueSource;
13  import org.objectweb.asm.Opcodes;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Xembler;
16  
17  /**
18   * Test cases for {@link DirectivesField}.
19   * This class verifies the generation of field directives for Java class fields,
20   * including various field properties and access modifiers.
21   *
22   * @since 0.3.0
23   */
24  final class DirectivesFieldTest {
25  
26      @Test
27      void convertsDefaultFieldToDirectives() throws ImpossibleModificationException {
28          final String xml = new Xembler(new DirectivesField()).xml();
29          MatcherAssert.assertThat(
30              String.format(
31                  "Incorrect transformation of default field to directives, received invalid XMIR: %n%s",
32                  xml
33              ),
34              xml,
35              XhtmlMatchers.hasXPaths(
36                  new JeoBaseXpath("/o", "field").toXpath(),
37                  "/o[contains(@name,'unknown')]",
38                  "/o/o[@name='access']",
39                  "/o/o[@name='descriptor']",
40                  "/o/o[@name='signature']",
41                  "/o/o[contains(@base,'number') and @name='value']"
42              )
43          );
44      }
45  
46      @Test
47      void convertsLongFieldToDirectives() throws ImpossibleModificationException {
48          final String xml = new Xembler(
49              new DirectivesField(
50                  Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL,
51                  "serialVersionUID",
52                  "J",
53                  "",
54                  7_099_057_708_183_571_937L
55              )
56          ).xml();
57          MatcherAssert.assertThat(
58              String.format(
59                  "Incorrect transformation of long field to directives, received invalid XMIR: %n%s",
60                  xml
61              ),
62              xml,
63              XhtmlMatchers.hasXPaths(
64                  new JeoBaseXpath("/o", "field").toXpath(),
65                  "/o[contains(@name,'serialVersionUID')]",
66                  "/o/o[@name='access']",
67                  "/o/o[@name='descriptor']",
68                  "/o/o[@name='signature']",
69                  "/o/o[@name='value']"
70              )
71          );
72      }
73  
74      @ParameterizedTest
75      @ValueSource(strings = {
76          "Φ", "Ψ", "Ω", "Δ", "Σ", "Θ", "Λ", "Ξ", "Π", "Υ", "\u03A3", "\u03A6", "\u03A8", "\u03A9"
77      })
78      void convertsDirectivesFieldWithUnicodeName(
79          final String original
80      ) throws ImpossibleModificationException {
81          MatcherAssert.assertThat(
82              "We expect the field name with unicode characters to be successfully converted to directives",
83              new Xembler(
84                  new BytecodeField(
85                      original,
86                      "I",
87                      "",
88                      0,
89                      Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL
90                  ).directives(new Format())
91              ).xml(),
92              XhtmlMatchers.hasXPaths(
93                  new JeoBaseXpath("/o", "field").toXpath(),
94                  String.format("/o[contains(@name,'%s')]", original)
95              )
96          );
97      }
98  }
