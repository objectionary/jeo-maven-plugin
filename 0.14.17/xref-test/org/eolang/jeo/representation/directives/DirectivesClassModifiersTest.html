5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.objectweb.asm.Opcodes;
11  import org.xembly.ImpossibleModificationException;
12  import org.xembly.Xembler;
13  
14  /**
15   * Tests for {@link Modifiers}.
16   * @since 0.15.0
17   */
18  final class DirectivesClassModifiersTest {
19  
20      @Test
21      void convertsToXmir() throws ImpossibleModificationException {
22          final DirectivesClassModifiers dirs = new DirectivesClassModifiers(
23              new Format(), Opcodes.ACC_PUBLIC | Opcodes.ACC_FINAL | Opcodes.ACC_ENUM
24          );
25          final String xml = new Xembler(dirs).xml();
26          MatcherAssert.assertThat(
27              "We expect class access modifiers in the directives with correct values",
28              xml,
29              XhtmlMatchers.hasXPaths(
30                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'public') and contains(@base, 'true')]",
31                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'private') and contains(@base, 'false')]",
32                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'protected') and contains(@base, 'false')]",
33                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'final') and contains(@base, 'true')]",
34                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'super') and contains(@base, 'false')]",
35                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'interface') and contains(@base, 'false')]",
36                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'abstract') and contains(@base, 'false')]",
37                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'synthetic') and contains(@base, 'false')]",
38                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'annotation') and contains(@base, 'false')]",
39                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'enum') and contains(@base, 'true')]",
40                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'mandated') and contains(@base, 'false')]",
41                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'module') and contains(@base, 'false')]",
42                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'record') and contains(@base, 'false')]",
43                  "/o[contains(@name, 'modifiers')]/o[contains(@name, 'deprecated') and contains(@base, 'false')]"
44              )
45          );
46      }
47  }
