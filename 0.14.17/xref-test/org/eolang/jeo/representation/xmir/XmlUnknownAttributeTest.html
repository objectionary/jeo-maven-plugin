5   package org.eolang.jeo.representation.xmir;
6   
7   import org.eolang.jeo.representation.bytecode.BytecodeUnknownAttribute;
8   import org.eolang.jeo.representation.directives.Format;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  import org.xembly.ImpossibleModificationException;
13  import org.xembly.Xembler;
14  
15  /**
16   * Test case for {@link XmlUnknownAttribute}.
17   * @since 0.15.0
18   */
19  final class XmlUnknownAttributeTest {
20  
21      @Test
22      void parsesTypeAndData() throws ImpossibleModificationException {
23          final BytecodeUnknownAttribute origin = new BytecodeUnknownAttribute(
24              "Some-Type",
25              new byte[]{0, 1, 2, 3, 4, 5}
26          );
27          MatcherAssert.assertThat(
28              "We expect to receive the same bytecode attribute",
29              new XmlUnknownAttribute(
30                  new JcabiXmlNode(new Xembler(origin.directives(0, new Format())).xml())
31              ).bytecode(),
32              Matchers.equalTo(origin)
33          );
34      }
35  }
