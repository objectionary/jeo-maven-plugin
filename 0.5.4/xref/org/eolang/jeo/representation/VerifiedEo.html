24  package org.eolang.jeo.representation;
25  
26  import com.jcabi.xml.XML;
27  import com.jcabi.xml.XMLDocument;
28  import org.eolang.jeo.representation.directives.DirectivesClassVisitor;
29  import org.eolang.parser.Schema;
30  import org.xembly.Directives;
31  import org.xembly.ImpossibleModificationException;
32  import org.xembly.Xembler;
33  
34  /**
35   * Class that verifies EO representation and fixes some small syntax bugs.
36   *
37   * @since 5.1
38   */
39  final class VerifiedEo {
40  
41      /**
42       * Directives to verify.
43       */
44      private final DirectivesClassVisitor directives;
45  
46      /**
47       * Constructor.
48       * @param dirs Directives to verify.
49       */
50      VerifiedEo(final DirectivesClassVisitor dirs) {
51          this.directives = dirs;
52      }
53  
54      /**
55       * XML representation of the EO.
56       * We also check if the EO is correct.
57       * @return XML representation.
58       * @throws ImpossibleModificationException If something goes wrong.
59       */
60      XML asXml() throws ImpossibleModificationException {
61          final XML res = VerifiedEo.cleanAliases(
62              new XMLDocument(new Xembler(this.directives).xml())
63          );
64          new Schema(res).check();
65          return res;
66      }
67  
68      /**
69       * Clean aliases from the EO if they are not needed.
70       * @param original Original EO with aliases.
71       * @return Clean EO.
72       * @throws ImpossibleModificationException If something goes wrong.
73       * @todo #640:60min Remove {@link #cleanAliases(XML)} method.
74       *  This method is some sort of crutch to remove aliases from the EO.
75       *  Actually, it's better just not to add them if they are not needed.
76       *  So, we need to remove this method and refactor the code to not add aliases.
77       */
78      private static XML cleanAliases(final XML original) throws ImpossibleModificationException {
79          final XML result;
80          if (VerifiedEo.dontHaveInstructions(original)) {
81              result = new XMLDocument(
82                  new Xembler(
83                      new Directives()
84                          .xpath("./program/metas/meta[head[text()='alias']]")
85                          .remove()
86                  ).apply(original.node())
87              );
88          } else {
89              result = original;
90          }
91          return result;
92      }
93  
94      /**
95       * Check if EO has methods with instructions.
96       * @param original Original EO.
97       * @return True if EO has instructions.
98       */
99      private static boolean dontHaveInstructions(final XML original) {
100         return original.xpath(
101             ".//o[@base='seq']/o[@base='tuple']/o[@base='opcode' or @base='label']/text()"
102         ).isEmpty();
103     }
104 }
