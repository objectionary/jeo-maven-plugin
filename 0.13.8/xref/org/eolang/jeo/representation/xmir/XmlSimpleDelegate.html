5   package org.eolang.jeo.representation.xmir;
6   
7   /**
8    * Xml representation of a simple EO delegate object.
9    * <p>
10   *     Mirrors {@link org.eolang.jeo.representation.directives.DirectivesSimpleDelegate}
11   * </p>
12   * @since 0.12.0
13   */
14  public final class XmlSimpleDelegate {
15  
16      /**
17       * Inner XML node representing the delegate object.
18       */
19      private final XmlNode inner;
20  
21      /**
22       * Constructor.
23       * @param node XML node representing the delegate object
24       */
25      XmlSimpleDelegate(final XmlNode node) {
26          this.inner = node;
27      }
28  
29      /**
30       * Base of the delegate object.
31       * @return Base of the delegate object.
32       */
33      public String base() {
34          return this.inner.attribute("base").orElseThrow(
35              () -> new IllegalStateException(
36                  String.format(
37                      "Base attribute is not set in the XML node %s",
38                      this.inner
39                  )
40              )
41          );
42      }
43  }
