5   package org.eolang.jeo;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  
12  /**
13   * Logging transformation decorator.
14   *
15   * <p>This class wraps a transformation and adds logging functionality. It logs
16   * messages before and after applying the original transformation, including file
17   * sizes and processing time for performance monitoring.</p>
18   * @since 0.6.0
19   */
20  public final class Logging implements Transformation {
21  
22      /**
23       * Process name.
24       * <p>Usually it is a gerund form, like:
25       * "Disassembling", "Assembling", etc.</p>
26       */
27      private final String process;
28  
29      /**
30       * Participle of the process.
31       * <p>Past participle form, like:
32       * "disassembled", "assembled", etc.</p>
33       */
34      private final String participle;
35  
36      /**
37       * Original transformation.
38       */
39      private final Transformation origin;
40  
41      /**
42       * Constructor.
43       * @param process Process name (gerund form)
44       * @param participle Past participle of the process
45       * @param origin Original transformation to wrap with logging
46       */
47      public Logging(final String process, final String participle, final Transformation origin) {
48          this.process = process;
49          this.participle = participle;
50          this.origin = origin;
51      }
52  
53      @Override
54      public Path source() {
55          return this.origin.source();
56      }
57  
58      @Override
59      public Path target() {
60          return this.origin.target();
61      }
62  
63      @Override
64      public byte[] transform() {
65          final byte[] result;
66          this.logStartWithSize(this.source());
67          final long start = System.currentTimeMillis();
68          result = this.origin.transform();
69          final long time = System.currentTimeMillis() - start;
70          this.logEndWithSize(this.source(), this.target(), time);
71          return result;
72      }
73  
74      /**
75       * Log the start of the process.
76       * @param source Initial path of the file being processed
77       */
78      private void logStartWithSize(final Path source) {
79          Logger.debug(
80              this,
81              "%s of %[file]s (%[size]s) started",
82              this.process,
83              source,
84              Logging.size(source)
85          );
86      }
87  
88      /**
89       * Log the end of the process.
90       * @param source Initial path of the source file
91       * @param after Path to the resulting file after processing
92       * @param time Time spent in milliseconds
93       */
94      private void logEndWithSize(final Path source, final Path after, final long time) {
95          Logger.info(
96              this,
97              "%[file]s %s to %[file]s (%[size]s) in %[ms]s",
98              source,
99              this.participle,
100             after,
101             Logging.size(after),
102             time
103         );
104     }
105 
106     /**
107      * Size of the file.
108      * @param path Path to the file
109      * @return Size of the file in bytes, or 0 if file doesn't exist
110      */
111     private static long size(final Path path) {
112         try {
113             final long result;
114             if (Files.exists(path)) {
115                 result = Files.size(path);
116             } else {
117                 result = 0L;
118             }
119             return result;
120         } catch (final IOException exception) {
121             throw new IllegalStateException(
122                 String.format("Can't determine the size of '%s'", path),
123                 exception
124             );
125         }
126     }
127 }
