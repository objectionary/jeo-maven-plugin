24  package org.eolang.jeo.representation;

26  import java.util.stream.Stream;
27  import org.hamcrest.MatcherAssert;
28  import org.hamcrest.Matchers;
29  import org.junit.jupiter.params.ParameterizedTest;
30  import org.junit.jupiter.params.provider.Arguments;
31  import org.junit.jupiter.params.provider.MethodSource;
32  
33  /**
34   * Test cases for {@link org.eolang.jeo.representation.HexData}.
35   * @since 0.1
36   */
37  class HexDataTest {
38  
39      @MethodSource("types")
40      @ParameterizedTest
41      void determinesTypeCorrectly(final Object data, final String type) {
42          MatcherAssert.assertThat(
43              String.format(
44                  "Expected and actual types differ, the type for '%s' should be '%s'",
45                  data,
46                  type
47              ),
48              new HexData(data).type(),
49              Matchers.equalTo(type)
50          );
51      }
52  
53      @MethodSource("values")
54      @ParameterizedTest
55      void convertsRawDataIntoHexString(final Object data, final String hex) {
56          MatcherAssert.assertThat(
57              String.format(
58                  String.format(
59                      "Expected and actual hex values differ, the value for '%s' should be '%s'",
60                      data,
61                      hex
62                  )
63              ),
64              new HexData(data).value(),
65              Matchers.equalTo(hex)
66          );
67      }
68  
69      /**
70       * Arguments for {@link HexDataTest#determinesTypeCorrectly(Object, String)} test.
71       * @return Stream of arguments.
72       */
73      static Stream<Arguments> types() {
74          return Stream.of(
75              Arguments.of(1, "int"),
76              Arguments.of("Hello!", "string"),
77              Arguments.of(new byte[]{1, 2, 3}, "bytes"),
78              Arguments.of(true, "bool"),
79              Arguments.of(0.1d, "float"),
80              Arguments.of(HexDataTest.class, "reference")
81          );
82      }
83  
84      /**
85       * Arguments for {@link HexDataTest#convertsRawDataIntoHexString(Object, String)}.
86       * Example for reference - {@link HexDataTest} is "org.eolang.jeo.representation.HexDataTest"
87       * @return Stream of arguments.
88       */
89      static Stream<Arguments> values() {
90          return Stream.of(
91              Arguments.of(10, "00 00 00 00 00 00 00 0A"),
92              Arguments.of("Hello!", "48 65 6C 6C 6F 21"),
93              Arguments.of(new byte[]{1, 2, 3}, "01 02 03"),
94              Arguments.of(true, "01"),
95              Arguments.of(false, "00"),
96              Arguments.of(0.1d, "3F B9 99 99 99 99 99 9A"),
97              Arguments.of(
98                  HexDataTest.class,
99                  "6F 72 67 2F 65 6F 6C 61 6E 67 2F 6A 65 6F 2F 72 65 70 72 65 73 65 6E 74 61 74 69 6F 6E 2F 48 65 78 44 61 74 61 54 65 73 74"
100             )
101         );
102     }
