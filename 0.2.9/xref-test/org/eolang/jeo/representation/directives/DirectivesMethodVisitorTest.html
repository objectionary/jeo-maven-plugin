40  /**
41   * Test case for {@link DirectivesMethodVisitor}.
42   * We create {@link DirectivesMethodVisitor} only in the context
43   * of using {@link DirectivesClassVisitor} in other words,
44   * {@link DirectivesMethodVisitor} can't be createad
45   * without {@link DirectivesClassVisitor} and it is the main
46   * reason why in all the test we create
47   * {@link DirectivesClassVisitor}.
48   *
49   * @since 0.1.0
50   */
51  @SuppressWarnings("PMD.AvoidDuplicateLiterals")
52  class DirectivesMethodVisitorTest {
53  
54      @Test
55      void parsesMethodInstructions() throws ImpossibleModificationException {
56          final DirectivesClassVisitor visitor = new DirectivesClassVisitor();
57          new ClassReader(
58              new BytecodeClass()
59                  .withMethod(new BytecodeMethodProperties("main", "()I"))
60                  .opcode(Opcodes.BIPUSH, 28)
61                  .opcode(Opcodes.IRETURN)
62                  .up()
63                  .bytecode()
64                  .asBytes()
65          ).accept(visitor, 0);
66          final String xml = new Xembler(visitor).xml();
67          MatcherAssert.assertThat(
68              String.format(
69                  "Can't find a method in the final XML by using XPath. Please, check the resulting XMIR: \n%s\n",
70                  xml
71              ),
72              xml,
73              new HasMethod("main")
74                  .inside("Simple")
75                  .withInstruction(Opcodes.BIPUSH, 28)
76                  .withInstruction(Opcodes.IRETURN)
77          );
78      }

80      /**
81       * In this test we parse the next java code (but represented as bytecode).
82       *
83       * <p>
84       *     {@code
85       *     public class ParamsExample {
86       *       public static void main(String[] args) {
87       *         ParamsExample pe = new ParamsExample();
88       *         pe.printSum(10, 20);
89       *       }
90       *       public void printSum(int a, int b) {
91       *         int sum = a + b;
92       *         System.out.println(sum);
93       *       }
94       *     }
95       *     }
96       * </p>
97       */
98      @Test
99      void parsesMethodParameters() {
100         final String clazz = "ParametersExample";
101         final String method = "printSum";
102         final String xml = new BytecodeClass(clazz)
103             .withMethod(
104                 new BytecodeMethodProperties(
105                     "main",
106                     "([Ljava/lang/String;)V",
107                     Opcodes.ACC_PUBLIC,
108                     Opcodes.ACC_STATIC
109                 )
110             )
111             .opcode(Opcodes.NEW, clazz)
112             .opcode(Opcodes.DUP)
113             .opcode(Opcodes.INVOKESPECIAL, clazz, "<init>", "()V")
114             .opcode(Opcodes.ASTORE, 1)
115             .opcode(Opcodes.ALOAD, 1)
116             .opcode(Opcodes.BIPUSH, 10)
117             .opcode(Opcodes.BIPUSH, 20)
118             .opcode(Opcodes.INVOKEVIRTUAL, clazz, method, "(II)V")
119             .opcode(Opcodes.RETURN)
120             .up()
121             .withMethod(new BytecodeMethodProperties(method, "(II)V", Opcodes.ACC_PUBLIC))
122             .opcode(Opcodes.ILOAD, 1)
123             .opcode(Opcodes.ILOAD, 2)
124             .opcode(Opcodes.IADD)
125             .opcode(Opcodes.ISTORE, 3)
126             .opcode(Opcodes.GETSTATIC, "java/lang/System", "out", "Ljava/io/PrintStream;")
127             .opcode(Opcodes.ILOAD, 3)
128             .opcode(Opcodes.INVOKEVIRTUAL, "java/io/PrintStream", "println", "(I)V")
129             .opcode(Opcodes.RETURN)
130             .up()
131             .xml()
132             .toString();
133         MatcherAssert.assertThat(
134             String.format(
135                 "DirectivesMethod doesn't parse method parameters correctly. Please, check the resulting XMIR: \n%s\n",
136                 xml
137             ),
138             xml,
139             new HasMethod(method)
140                 .inside(clazz)
141                 .withParameter("arg__I__0")
142                 .withParameter("arg__I__1")
143         );
144     }

147      * In this test we parse the next java code.
148      *
149      * <p>
150      *     {@code
151      *     public class ConstructorExample {
152      *       public ConstructorExample() {
153      *           System.out.println("Hello, constructor!");
154      *       }
155      *       public static void main(String[] args) {
156      *         new ConstructorExample();
157      *       }
158      *     }
159      *     }
160      * </p>
161      */
162     @Test
163     void parsesConstructor() {
164         final String xml = new BytecodeClass("ConstructorExample")
165             .withConstructor(Opcodes.ACC_PUBLIC)
166             .opcode(Opcodes.ALOAD, 0)
167             .opcode(Opcodes.INVOKESPECIAL, "java/lang/Object", "<init>", "()V")
168             .opcode(Opcodes.GETSTATIC, "java/lang/System", "out", "Ljava/io/PrintStream;")
169             .opcode(Opcodes.LDC, "Hello, constructor!")
170             .opcode(
171                 Opcodes.INVOKEVIRTUAL,
172                 "java/io/PrintStream",
173                 "println",
174                 "(Ljava/lang/String;)V"
175             )
176             .opcode(Opcodes.RETURN)
177             .up()
178             .withMethod(
179                 new BytecodeMethodProperties(
180                     "main",
181                     "([Ljava/lang/String;)V",
182                     Opcodes.ACC_PUBLIC,
183                     Opcodes.ACC_STATIC
184                 )
185             )
186             .opcode(Opcodes.NEW, "ConstructorExample")
187             .opcode(Opcodes.DUP)
188             .opcode(Opcodes.INVOKESPECIAL, "ConstructorExample", "<init>", "()V")
189             .opcode(Opcodes.POP)
190             .opcode(Opcodes.RETURN)
191             .up()
192             .xml()
193             .toString();
194         MatcherAssert.assertThat(
195             String.format(
196                 "Constructor wasn't parsed correctly, please, check the resulting XMIR: \n%s\n",
197                 new XMLDocument(xml)
198             ),
199             xml,
200             new HasMethod("new")
201                 .inside("ConstructorExample")
202                 .withInstruction(Opcodes.LDC, "Hello, constructor!")
203         );
204     }

222     @Test
223     void parsesConstructorWithParameters() {
224         final String clazz = "ConstructorParams";
225         final String xml = new BytecodeClass(clazz)
226             .withConstructor("(II)V", Opcodes.ACC_PUBLIC)
227             .opcode(Opcodes.ALOAD, 0)
228             .opcode(Opcodes.INVOKESPECIAL, "java/lang/Object", "<init>", "()V")
229             .opcode(Opcodes.GETSTATIC, "java/lang/System", "out", "Ljava/io/PrintStream;")
230             .opcode(Opcodes.ILOAD, 1)
231             .opcode(Opcodes.ILOAD, 2)
232             .opcode(Opcodes.IADD)
233             .opcode(Opcodes.INVOKEVIRTUAL, "java/io/PrintStream", "println", "(I)V")
234             .opcode(Opcodes.RETURN)
235             .up()
236             .withMethod(
237                 new BytecodeMethodProperties(
238                     "main",
239                     "([Ljava/lang/String;)V",
240                     Opcodes.ACC_PUBLIC,
241                     Opcodes.ACC_STATIC
242                 )
243             )
244             .opcode(Opcodes.NEW, clazz)
245             .opcode(Opcodes.DUP)
246             .opcode(Opcodes.BIPUSH, 11)
247             .opcode(Opcodes.BIPUSH, 22)
248             .opcode(Opcodes.INVOKESPECIAL, clazz, "<init>", "(II)V")
249             .opcode(Opcodes.POP)
250             .opcode(Opcodes.RETURN)
251             .up()
252             .xml()
253             .toString();

255             String.format(
256                 "Constructor wasn't parsed correctly, please, check the resulting XMIR: \n%s\n",
257                 new XMLDocument(xml)
258             ),
259             xml,
260             new HasMethod("new")
261                 .inside(clazz)
262                 .withInstruction(Opcodes.ILOAD, 1)
263                 .withInstruction(Opcodes.ILOAD, 2)
264                 .withInstruction(Opcodes.IADD)
265         );
266     }

268     /**
269      * In this class we parse the next java code.
270      * <p>
271      * {@code
272      *     class Foo {
273      *       int bar(double x) {
274      *         if (x > 0.0d) {
275      *           return 5;
276      *         }
277      *         return 8;
278      *       }
279      *     }
280      * }
281      * </p>
282      * Pay Attention! That we just can't verify exact id's of labels since ASM library doesn't
283      * allow it. Hence, we can just check the presence of a label.
284      */
285     @Test
286     void parsesIfStatementCorrectly() {
287         final AllLabels labels = new AllLabels();
288         final Label label = labels.label(UUID.randomUUID().toString());
289         final String xml = new BytecodeClass("Foo")
290             .withMethod("bar", "(D)I", 0)
291             .opcode(Opcodes.DLOAD, 1)
292             .opcode(Opcodes.DCONST_0)
293             .opcode(Opcodes.DCMPL)
294             .opcode(Opcodes.IFLE, label)
295             .opcode(Opcodes.ICONST_5)
296             .opcode(Opcodes.IRETURN)
297             .label(label)
298             .opcode(Opcodes.BIPUSH, 8)
299             .opcode(Opcodes.IRETURN).up()
300             .xml()
301             .toString();
302         MatcherAssert.assertThat(
303             String.format(
304                 "If statement wasn't parsed correctly, please, check the resulting XMIR: \n%s\n",
305                 xml
306             ),
307             xml,
308             new HasMethod("bar")
309                 .inside("Foo")
310                 .withInstruction(Opcodes.IFLE, label)
311                 .withLabel()
312         );
313     }

315     /**
316      * In this class we parse the next java code.
317      * <p>
318      * {@code
319      *   public class Foo {
320      *     public void bar(){
321      *         try {
322      *           throw new Exception();
323      *         } catch (Exception e) {
324      *         }
325      *     }
326      *   }
327      * }
328      */
329     @Test
330     void parsesTryCatchInstructions() {
331         final AllLabels labels = new AllLabels();
332         final Label start = labels.label(UUID.randomUUID().toString());
333         final Label end = labels.label(UUID.randomUUID().toString());
334         final Label handler = labels.label(UUID.randomUUID().toString());
335         final String xml = new BytecodeClass("Foo")
336             .withMethod("bar", "()V", Opcodes.ACC_PUBLIC)
337             .label(start)
338             .opcode(Opcodes.NEW, "java/lang/Exception")
339             .opcode(Opcodes.DUP)
340             .opcode(Opcodes.INVOKESPECIAL, "java/lang/Exception", "<init>", "()V")
341             .opcode(Opcodes.ATHROW)
342             .opcode(Opcodes.ASTORE, 1)
343             .label(end)
344             .label(handler)
345             .opcode(Opcodes.RETURN)
346             .entry(new BytecodeTryCatchBlock(start, end, handler, "java/lang/Exception"))
347             .up()
348             .xml()
349             .toString();
350         MatcherAssert.assertThat(
351             String.format(
352                 "Exception table wasn't parsed correctly, please, check the resulting XMIR: \n%s\n",
353                 xml
354             ),
355             xml,
356             new HasMethod("bar")
357                 .inside("Foo")
358                 .withTryCatch("java/lang/Exception")
359         );
360     }
