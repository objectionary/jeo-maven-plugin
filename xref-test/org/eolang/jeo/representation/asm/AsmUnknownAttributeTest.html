5   package org.eolang.jeo.representation.asm;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Test;
10  
11  /**
12   * Test case for {@link AsmUnknownAttribute}.
13   * @since 0.16.0
14   */
15  final class AsmUnknownAttributeTest {
16  
17      @Test
18      void writesNullDataAsEmpty() {
19          MatcherAssert.assertThat(
20              "Writing attribute with null data should produce empty ByteVector, not throw",
21              new AsmUnknownAttribute("Test", null).write(null, null, 0, 0, 0),
22              Matchers.notNullValue()
23          );
24      }
25  
26      @Test
27      void writesNonEmptyData() {
28          MatcherAssert.assertThat(
29              "Writing attribute with data should produce non-null ByteVector",
30              new AsmUnknownAttribute("Test", new byte[]{0x01, 0x02, 0x03})
31                  .write(null, null, 0, 0, 0),
32              Matchers.notNullValue()
33          );
34      }
35  }
