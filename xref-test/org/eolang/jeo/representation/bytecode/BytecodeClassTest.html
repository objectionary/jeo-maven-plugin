21  @SuppressWarnings({"PMD.ExcessiveMethodLength", "PMD.TooManyMethods"})

23  
24      @Test
25      void cleansMethods() {
26          final String name = "CleanMethods";
27          MatcherAssert.assertThat(
28              String.format(
29                  "Can't clean methods from the class %s",
30                  name
31              ),
32              new BytecodeClass(name)
33                  .withMethod("j$foo", "()V", Opcodes.ACC_PUBLIC)
34                  .up()
35                  .withoutMethods(),
36              Matchers.equalTo(new BytecodeClass(name))
37          );
38      }
39  
40      @Test
41      void parsesAbstractClass() {
42          final int access = Opcodes.ACC_ABSTRACT | Opcodes.ACC_PUBLIC | Opcodes.ACC_SUPER;
43          MatcherAssert.assertThat(
44              String.format(
45                  "Can't parse abstract class with access modifier %s",
46                  access
47              ),
48              new BytecodeObject(new BytecodeClass("AbstractClass", access)).bytecode(),
49              Matchers.notNullValue()
50          );
51      }
52  
53      @Test
54      void createsConstructorWithStoreInstructions() {
55          Assertions.assertDoesNotThrow(
56              () -> new BytecodeObject(
57                  new BytecodeClass("Store")
58                      .withConstructor()
59                      .opcode(Opcodes.ICONST_0)
60                      .opcode(Opcodes.ISTORE, 1)
61                      .opcode(Opcodes.LCONST_0)
62                      .opcode(Opcodes.LSTORE, 2)
63                      .opcode(Opcodes.FCONST_0)
64                      .opcode(Opcodes.FSTORE, 3)
65                      .opcode(Opcodes.DCONST_0)
66                      .opcode(Opcodes.DSTORE, 4)
67                      .opcode(Opcodes.ACONST_NULL)
68                      .opcode(Opcodes.ASTORE, 5)
69                      .opcode(Opcodes.RETURN)
70                      .up()
71              ).bytecode(),
72              "We expect no exception here because all instructions are valid"
73          );
74      }
75  
76      @Test
77      void createsBytecodeWithDefaultConstructor() {
78          MatcherAssert.assertThat(
79              "Can't create bytecode with default public constructor",
80              new BytecodeObject(
81                  new BytecodeClass("DefaultConstructor")
82                      .withConstructor(Opcodes.ACC_PUBLIC)
83                      .opcode(Opcodes.RETURN)
84                      .up()
85              ).bytecode(),
86              Matchers.notNullValue()
87          );
88      }

90      @Test
91      void createsClassWithUnknownInstruction() {
92          MatcherAssert.assertThat(
93              "Exception message is not equal to expected",
94              Assertions.assertThrows(
95                  IllegalStateException.class,
96                  () -> new BytecodeObject(new BytecodeClass("UnknownInstruction")
97                      .withConstructor()
98                      .opcode(305)
99                      .up()
100                 ).bytecode(),
101                 "We expect an exception here because 305 is not a valid opcode"
102             ).getMessage(),
103             Matchers.containsString(
104                 "Bytecode creation for the 'UnknownInstruction' class is not possible due to unmet preconditions."
105             )
106         );
107     }

109     @Test
110     void createsMethodWithArrayParameterUsage() {
111         Assertions.assertDoesNotThrow(
112             () -> {
113                 new BytecodeClass("Some")
114                     .withMethod("j$main", "([Ljava/lang/String;)V", 137)
115                     .label()
116                     .opcode(Opcodes.GETSTATIC, "java/lang/System", "out", "Ljava/io/PrintStream;")
117                     .opcode(Opcodes.LDC, "Hello %d")
118                     .opcode(Opcodes.ICONST_1)
119                     .opcode(Opcodes.ANEWARRAY, "java/lang/Object")
120                     .opcode(Opcodes.DUP)
121                     .opcode(Opcodes.ICONST_0)
122                     .opcode(Opcodes.BIPUSH, 12)
123                     .opcode(
124                         Opcodes.INVOKESTATIC,
125                         "java/lang/Integer",
126                         "valueOf",
127                         "(I)Ljava/lang/Integer;"
128                     )
129                     .opcode(Opcodes.AASTORE)
130                     .opcode(
131                         Opcodes.INVOKEVIRTUAL,
132                         "java/io/PrintStream",
133                         "printf",
134                         "(Ljava/lang/String;[Ljava/lang/Object;)Ljava/io/PrintStream;"
135                     )
136                     .opcode(Opcodes.POP)
137                     .label()
138                     .opcode(Opcodes.RETURN)
139                     .label()
140                     .up();
141             }, "We expect no exception here because all instructions are valid"
142         );
143     }

167                 .opcode(Opcodes.IRETURN)

189     @Test
190     void convertsToDirectivesOverloadedMethods() throws ImpossibleModificationException {
191         final String duplicated = "duplicated";
192         MatcherAssert.assertThat(
193             "We expect to have two methods with the names containing numbers to distinguish them",
194             new Xembler(
195                 new BytecodeClass("OverloadedMethods")
196                     .withMethod(duplicated, "(I)I", Opcodes.ACC_PUBLIC)
197                     .opcode(Opcodes.ILOAD, 1)
198                     .opcode(Opcodes.IRETURN)
199                     .up()
200                     .withMethod(duplicated, "(II)I", Opcodes.ACC_PUBLIC)
201                     .opcode(Opcodes.ILOAD, 1)
202                     .opcode(Opcodes.ILOAD, 2)
203                     .opcode(Opcodes.IADD)
204                     .opcode(Opcodes.IRETURN)
205                     .up()
206                     .directives(new Format())
207             ).xml(),
208             XhtmlMatchers.hasXPaths(
209                 "//o[@name='jm$duplicated']",
210                 "//o[@name='jm$duplicated-2']"
211             )
212         );
213     }

215     @Test
216     void convertsToDirectivesWithMethodAndFieldHavingTheSameName()
217         throws ImpossibleModificationException {
218         final String same = "same";
219         MatcherAssert.assertThat(
220             "We expect to have a method and a field with the same name",
221             new Xembler(
222                 new BytecodeClass("MethodAndFieldWithSameName")
223                     .withField(same)
224                     .withMethod(same, "()I", Opcodes.ACC_PUBLIC)
225                     .opcode(Opcodes.ICONST_0)
226                     .opcode(Opcodes.IRETURN)
227                     .up()
228                     .directives(new Format())
229             ).xml(),
230             XhtmlMatchers.hasXPaths(
231                 "//o[@name='j$same']",
232                 "//o[@name='jm$same']"
233             )
234         );
235     }
