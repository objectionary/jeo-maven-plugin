24  package org.eolang.jeo.representation.bytecode;
25  
26  import com.jcabi.xml.XMLDocument;
27  import org.eolang.jeo.representation.directives.DirectivesProgram;
28  import org.eolang.jeo.representation.directives.HasClass;
29  import org.eolang.jeo.representation.directives.HasMethod;
30  import org.hamcrest.MatcherAssert;
31  import org.junit.jupiter.api.Test;
32  import org.xembly.ImpossibleModificationException;
33  import org.xembly.Xembler;
34  
35  /**
36   * Test case for {@link BytecodeProgram}.
37   * @since 0.6
38   */
39  final class BytecodeProgramTest {
40  
41      @Test
42      void convertsSimpleClassWithoutConstructorToXmir() throws ImpossibleModificationException {
43          MatcherAssert.assertThat(
44              "Can't parse simple class without constructor",
45              new Xembler(new BytecodeProgram(new BytecodeClass()).directives("")).xml(),
46              new HasClass("Simple")
47          );
48      }
49  
50      @Test
51      void convertsSimpleClassWithMethodToXmir() throws ImpossibleModificationException {
52          final String clazz = "WithMethod";
53          final String xml = new Xembler(
54              new BytecodeProgram(
55                  new BytecodeClass(clazz)
56                      .helloWorldMethod()
57              ).directives("")
58          ).xml();
59          MatcherAssert.assertThat(
60              String.format(
61                  "Can't parse simple class with method, result is: '%s'",
62                  new XMLDocument(xml)
63              ),
64              xml,
65              new HasMethod("main").inside(clazz)
66          );
67      }
68  
69      @Test
70      void convertsClassWithPackageToXmir() throws ImpossibleModificationException {
71          final String name = "ClassInPackage";
72          final DirectivesProgram directives = new BytecodeProgram(
73              "some/package",
74              new BytecodeClass(name).helloWorldMethod()
75          ).directives("");
76          final String xml = new Xembler(directives).xml();
77          final String pckg = "some.package";
78          MatcherAssert.assertThat(
79              String.format(
80                  "Can't parse '%s' class that placed under package '%s', result is: %n%s%n",
81                  name,
82                  pckg,
83                  new XMLDocument(xml)
84              ),
85              xml,
86              new HasClass(name).inside(pckg)
87          );
88      }
89  }
