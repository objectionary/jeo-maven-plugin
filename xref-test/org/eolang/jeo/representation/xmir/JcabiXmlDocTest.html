5   package org.eolang.jeo.representation.xmir;
6   
7   import java.io.IOException;
8   import java.nio.charset.StandardCharsets;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.hamcrest.MatcherAssert;
12  import org.junit.jupiter.api.Test;
13  import org.junit.jupiter.api.io.TempDir;
14  
15  /**
16   * Test case for {@link JcabiXmlDoc}.
17   * @since 0.8
18   */
19  final class JcabiXmlDocTest {
20  
21      /**
22       * Example XML.
23       */
24      private static final String XML = "<object><o>1</o></object>";
25  
26      @Test
27      void createsFromFile(@TempDir final Path dir) throws IOException {
28          final Path path = dir.resolve("test.xml");
29          Files.write(path, JcabiXmlDocTest.XML.getBytes(StandardCharsets.UTF_8));
30          MatcherAssert.assertThat(
31              "Can't read XML from file",
32              new JcabiXmlDoc(path).root().xpath("/object/o/text()").get(0),
33              org.hamcrest.Matchers.equalTo("1")
34          );
35      }
36  
37      @Test
38      void findsFirstChildInFileWithComment(@TempDir final Path dir) throws IOException {
39          final Path path = dir.resolve("test.xml");
40          Files.write(
41              path,
42              String.format("<!-- Some comment -->%s", JcabiXmlDocTest.XML)
43                  .getBytes(StandardCharsets.UTF_8)
44          );
45          MatcherAssert.assertThat(
46              "Can't read XML from file",
47              new JcabiXmlDoc(path).root().children().findFirst().orElseThrow(AssertionError::new),
48              org.hamcrest.Matchers.equalTo(new JcabiXmlNode("<o>1</o>"))
49          );
50      }
51  }
