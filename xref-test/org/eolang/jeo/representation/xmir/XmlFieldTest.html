42  /**
43   * Test case for {@link XmlField}.
44   *
45   * @since 0.3
46   */
47  final class XmlFieldTest {
48  
49      @Test
50      void parsesXmirFieldSuccessfully() throws ImpossibleModificationException {
51          final int access = Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
52          final String name = "serialVersionUID";
53          final String descriptor = "J";
54          final long value = 7_099_057_708_183_571_937L;
55          MatcherAssert.assertThat(
56              "Failed to parse XMIR field",
57              new XmlField(
58                  new XmlNode(
59                      new Xembler(
60                          new DirectivesField(
61                              access,
62                              name,
63                              descriptor,
64                              "",
65                              value
66                          )
67                      ).xml()
68                  )
69              ).bytecode(),
70              Matchers.equalTo(
71                  new BytecodeField(
72                      name,
73                      descriptor,
74                      null,
75                      value,
76                      access,
77                      new BytecodeAnnotations()
78                  )
79              )
80          );
81      }
82  
83      @ParameterizedTest
84      @MethodSource("values")
85      void parsesFieldWithValues(final Object value) throws ImpossibleModificationException {
86          final int access = Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
87          final String name = "serialVersionUID";
88          final String descriptor = "Ljava/lang/String;";
89          MatcherAssert.assertThat(
90              "We expect the value to be the same",
91              new XmlField(
92                  new XmlNode(
93                      new Xembler(
94                          new DirectivesField(access, name, descriptor, null, value)
95                      ).xml()
96                  )
97              ).bytecode(),
98              Matchers.equalTo(
99                  new BytecodeField(
100                     name, descriptor, null, value, access
101                 )
102             )
103         );
104     }
105 
106     @Test
107     void retrievesFieldAnnotations() throws ImpossibleModificationException {
108         final String override = "java/lang/Override";
109         final String safe = "java/lang/SafeVarargs";
110         MatcherAssert.assertThat(
111             "Annotations are not found",
112             new XmlField(
113                 new XmlNode(
114                     new Xembler(
115                         new DirectivesField(
116                             new DirectivesAnnotation(override, true),
117                             new DirectivesAnnotation(safe, true)
118                         )
119                     ).xml()
120                 )
121             ).bytecode(),
122             Matchers.equalTo(
123                 new BytecodeField(
124                     "unknown",
125                     "I",
126                     null,
127                     0,
128                     1,
129                     new BytecodeAnnotations(
130                         new BytecodeAnnotation(override, true),
131                         new BytecodeAnnotation(safe, true)
132                     )
133                 )
134             )
135         );
136     }
137 
138     @ParameterizedTest
139     @ValueSource(strings = {
140         "Φ", "Ψ", "Ω", "Δ", "Σ", "Θ", "Λ", "Ξ", "Π", "Υ", "\u03A3", "\u03A6", "\u03A8", "\u03A9"
141     })
142     void parsesXmirFieldWithUnicodeCharacterInTheName(final String original)
143         throws ImpossibleModificationException {
144         final BytecodeField expected = new BytecodeField(
145             original,
146             "I",
147             null,
148             0,
149             Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL
150         );
151         MatcherAssert.assertThat(
152             "We expect the field name with unicode characters to be successfully parsed from directives",
153             new XmlField(new XmlNode(new Xembler(expected.directives()).xml())).bytecode(),
154             Matchers.equalTo(expected)
155         );
156     }

163     @SuppressWarnings("PMD.UnusedPrivateMethod")

166     }
