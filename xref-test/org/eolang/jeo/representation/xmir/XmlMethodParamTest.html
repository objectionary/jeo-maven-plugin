5   package org.eolang.jeo.representation.xmir;
6   
7   import org.eolang.jeo.representation.bytecode.BytecodeMethodParameter;
8   import org.eolang.jeo.representation.directives.Format;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  import org.objectweb.asm.Opcodes;
13  import org.objectweb.asm.Type;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Xembler;
16  
17  /**
18   * Test cases for {@link XmlMethodParam}.
19   * @since 0.6
20   */
21  final class XmlMethodParamTest {
22  
23      @Test
24      void convertsToBytecode() throws ImpossibleModificationException {
25          final BytecodeMethodParameter expected = new BytecodeMethodParameter(
26              1,
27              "foo",
28              Opcodes.ACC_STATIC,
29              Type.INT_TYPE
30          );
31          MatcherAssert.assertThat(
32              "Can't convert XML param to bytecode",
33              new XmlMethodParam(
34                  new NativeXmlNode(new Xembler(expected.directives(new Format())).xml())
35              ).bytecode(),
36              Matchers.equalTo(expected)
37          );
38      }
39  }
