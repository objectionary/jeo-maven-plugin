5   package org.eolang.jeo.representation;
6   
7   import org.cactoos.io.ResourceOf;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  
12  /**
13   * Test case for {@link BytecodeRepresentation}.
14   *
15   * @since 0.1
16   */
17  final class BytecodeRepresentationTest {
18  
19      /**
20       * The name of the resource with the simplest class.
21       */
22      private static final String METHOD_BYTE = "MethodByte.class";
23  
24      @Test
25      void parsesBytecode() {
26          MatcherAssert.assertThat(
27              "The simplest class should contain the object with MethodByte name",
28              new BytecodeRepresentation(new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE))
29                  .toEO().xpath("/program/@name").get(0),
30              Matchers.equalTo(new PrefixedName("MethodByte").encode())
31          );
32      }
33  
34      @Test
35      void retrievesName() {
36          final ResourceOf input = new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE);
37          final String actual = new BytecodeRepresentation(input).name();
38          final String expected = "org/eolang/jeo/MethodByte";
39          MatcherAssert.assertThat(
40              String.format(
41                  "The name should be retrieved without exceptions and equal to the expected '%s'",
42                  expected
43              ),
44              actual,
45              Matchers.equalTo(expected)
46          );
47      }
48  }
