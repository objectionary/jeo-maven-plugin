5   package org.eolang.jeo.representation;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.cactoos.io.ResourceOf;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  
13  /**
14   * Test cases for {@link BytecodeRepresentation}.
15   * This class verifies the parsing and conversion of Java bytecode
16   * to EO representation, including name extraction and XML generation.
17   *
18   * @since 0.1.0
19   */
20  final class BytecodeRepresentationTest {
21  
22      /**
23       * The name of the resource with the simplest class.
24       */
25      private static final String METHOD_BYTE = "MethodByte.class";
26  
27      @Test
28      void parsesBytecode() {
29          MatcherAssert.assertThat(
30              "The simplest class should contain the object with MethodByte name",
31              new BytecodeRepresentation(new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE))
32                  .toXmir().xpath("/object/o/@name").get(0),
33              Matchers.equalTo(new PrefixedName("MethodByte").encode())
34          );
35      }
36  
37      @Test
38      void generatesXmlWithCorrectObjectAttributes() {
39          MatcherAssert.assertThat(
40              "XMIR objects should have either 'name' or 'base' attribute, or both",
41              new BytecodeRepresentation(
42                  new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE)
43              ).toXmir().toString(),
44              Matchers.not(
45                  XhtmlMatchers.hasXPath(
46                      "//o[not(@base) and not(@name) and not(parent::*[@base='Q.org.eolang.bytes'])]"
47                  )
48              )
49          );
50      }
51  
52      /**
53       * This test was added to mitigate the issue:
54       * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1160">Issue #1160</a>.
55       */
56      @Test
57      void generatesXmlWithFormationsThatCIncludeObjectsWithNames() {
58          MatcherAssert.assertThat(
59              "A formation must only include objects with names.",
60              new BytecodeRepresentation(
61                  new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE)
62              ).toXmir().toString(),
63              Matchers.not(XhtmlMatchers.hasXPath("//o[not(@base)]/o[not(@name)]"))
64          );
65      }
66  
67      @Test
68      void retrievesName() {
69          final ResourceOf input = new ResourceOf(BytecodeRepresentationTest.METHOD_BYTE);
70          final String actual = new BytecodeRepresentation(input).name();
71          final String expected = "org/eolang/jeo/MethodByte";
72          MatcherAssert.assertThat(
73              String.format(
74                  "The name should be retrieved without exceptions and equal to the expected '%s'",
75                  expected
76              ),
77              actual,
78              Matchers.equalTo(expected)
79          );
80      }
81  }
