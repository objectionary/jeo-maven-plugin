21  /**
22   * Test case for {@link DirectivesInstruction}.
23   * @since 0.6
24   */
25  final class DirectivesInstructionTest {
26  
27      @Test
28      void covertsInstructionWithTypeToDirectives() {
29          MatcherAssert.assertThat(
30              "We expect that the bytecode instruction argument with type 'Type' will be wrapped in sting, see https://github.com/objectionary/jeo-maven-plugin/issues/1125",
31              new Xembler(
32                  new DirectivesInstruction(
33                      Opcodes.LDC,
34                      Type.getType(Integer.class)
35                  )
36              ).xmlQuietly(),
37              XhtmlMatchers.hasXPaths(
38                  new JeoBaseXpath("/o", "ldc").toXpath(),
39                  new JeoBaseXpath("/o/o", "type").toXpath(),
40                  "/o/o/o[contains(@base,'string')]",
41                  "/o/o/o/o[contains(@base, 'bytes')]"
42              )
43          );
44      }
45  
46      @ParameterizedTest
47      @MethodSource("instructionBases")
48      void transformsIntoEoWithoutCountingOpcodes(
49          final int opcode, final String base
50      ) throws ImpossibleModificationException {
51          final String xml = new Xembler(new DirectivesInstruction(opcode)).xml();
52          MatcherAssert.assertThat(
53              String.format(
54                  "We expect to get the EO representation of the bytecode where each instruction has a simple name without sequence number, please check the final XML:%n%s%n",
55                  xml
56              ),
57              xml,
58              XhtmlMatchers.hasXPath(new JeoBaseXpath("//o", base).toXpath())
59          );
60      }
61  
62      @ParameterizedTest
63      @MethodSource("instructions")
64      void addsBeautifulComment(final BytecodeInstruction instr, final String comment) {
65          MatcherAssert.assertThat(
66              "We expect, that during transformation to XML, we will get a beautiful comment for bytecode instruction",
67              new Xembler(instr.directives()).xmlQuietly(),
68              Matchers.containsString(comment)
69          );
70      }
