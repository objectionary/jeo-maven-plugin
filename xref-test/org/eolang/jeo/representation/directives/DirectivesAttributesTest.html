5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.xembly.ImpossibleModificationException;
11  import org.xembly.Xembler;
12  
13  /**
14   * Test case for {@link DirectivesAttributes}.
15   * @since 0.6
16   */
17  final class DirectivesAttributesTest {
18  
19      @Test
20      void convertsToXmir() throws ImpossibleModificationException {
21          MatcherAssert.assertThat(
22              "We expect that attributes will be converted to XMIR as a sequence of two objects with the name 'attributes'",
23              new Xembler(
24                  new DirectivesAttributes(
25                      new DirectivesAttribute("first"),
26                      new DirectivesAttribute("second")
27                  )
28              ).xml(),
29              XhtmlMatchers.hasXPaths(
30                  "./o[contains(@name, 'attributes')]",
31                  new JeoBaseXpath("./o", "seq.of2").toXpath(),
32                  new JeoBaseXpath("./o/o", "first").toXpath(),
33                  new JeoBaseXpath("./o/o", "second").toXpath()
34              )
35          );
36      }
37  }
