5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.xembly.Xembler;
11  
12  /**
13   * Test cases for {@link DirectivesNumber}.
14   * This class verifies the conversion of numeric values to XML format,
15   * ensuring proper representation of numbers in the directive system.
16   *
17   * @since 0.8.0
18   */
19  final class DirectivesNumberTest {
20  
21      @Test
22      void convertsToXml() {
23          MatcherAssert.assertThat(
24              "We can't convert number to XML",
25              new Xembler(new DirectivesNumber("42")).xmlQuietly(),
26              XhtmlMatchers.hasXPaths(
27                  "/o[contains(@base,'number')]",
28                  "/o[contains(@base,'number')]/o[contains(@base, 'bytes')]/o[text()='42']"
29              )
30          );
31      }
32  }
