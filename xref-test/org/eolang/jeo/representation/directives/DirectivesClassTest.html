8   import com.jcabi.xml.XMLDocument;
9   import java.util.Collections;
10  import org.eolang.jeo.representation.ClassName;
11  import org.eolang.jeo.representation.bytecode.InnerClass;
12  import org.eolang.jeo.representation.bytecode.JavaCodec;

14  import org.hamcrest.Matchers;

17  import org.xembly.ImpossibleModificationException;
18  import org.xembly.Transformers;
19  import org.xembly.Xembler;

21  /**
22   * Test cases for {@link DirectivesClass}.
23   * This class verifies the generation of class directives,
24   * including class properties, inner classes, and bytecode conversion.
25   *
26   * @since 0.1.0
27   */
28  final class DirectivesClassTest {
29  
30      @Test
31      void createsWithSimpleConstructor() throws ImpossibleModificationException {
32          final Format format = new Format();
33          MatcherAssert.assertThat(
34              "Can't create class with simple constructor",
35              new Xembler(
36                  new DirectivesClass(
37                      format,
38                      new ClassName("Neo"),
39                      new DirectivesClassProperties(),
40                      Collections.emptyList(),
41                      Collections.emptyList(),
42                      "",
43                      new DirectivesAnnotations(
44                          "annotations", new DirectivesAnnotation("Override", true)
45                      ),
46                      new DirectivesAttributes(
47                          new InnerClass("Inner", "Outer", "Inner", 0).directives(0, format)
48                      )
49                  ),
50                  new Transformers.Node()
51              ).xml(),
52              XhtmlMatchers.hasXPaths(
53                  "/o[contains(@name,'Neo')]",
54                  "/o[contains(@name,'Neo')]/o[contains(@name,'version')]",
55                  "/o[contains(@name,'Neo')]/o[contains(@name,'access')]",
56                  "/o[contains(@name,'Neo')]/o[contains(@name,'signature')]",
57                  "/o[contains(@name,'Neo')]/o[contains(@name,'supername')]",
58                  "/o[contains(@name,'Neo')]/o[contains(@name,'interfaces')]",
59                  "/o[contains(@name,'Neo')]/o[contains(@name,'annotations')]",
60                  "/o[contains(@name,'Neo')]/o[contains(@name,'attributes')]"
61              )
62          );
63      }
64  
65      @Test
66      void appendsField() throws ImpossibleModificationException {
67          final DirectivesField field = new DirectivesField();
68          final String xml = new Xembler(
69              new DirectivesClass(new ClassName("Neo"), field),
70              new Transformers.Node()
71          ).xml();
72          MatcherAssert.assertThat(
73              "Can't append field to the class",
74              xml,
75              Matchers.containsString(
76                  new Xembler(field, new Transformers.Node()).xml()
77              )
78          );
79      }
80  
81      @Test
82      void appendsMethod() {
83          final String xml = new Xembler(
84              new DirectivesClass(new ClassName("Neo"), new DirectivesMethod("method")),
85              new Transformers.Node()
86          ).xmlQuietly();
87          MatcherAssert.assertThat(
88              String.format(
89                  "Can't append method to the class, or the method is appended wrongly; result is: '%s'",
90                  new XMLDocument(xml)
91              ),
92              xml,
93              XhtmlMatchers.hasXPath("/o[@name='j$Neo']/o[contains(@name,'method')]")
94          );
95      }
96  
97      @Test
98      void appendsFullyQualifiedName() throws ImpossibleModificationException {
99          final String name = "org.eolang.jeo.representation.directives.DirectivesClassTest";
100         MatcherAssert.assertThat(
101             "Can't append fully qualified class name",
102             new Xembler(
103                 new DirectivesClass(
104                     new ClassName(name),
105                     new DirectivesClassProperties(name)
106                 )
107             ).xml(),
108             XhtmlMatchers.hasXPath(
109                 String.format(
110                     "./o[contains(@name,DirectivesClassTest)]/o[contains(@name, 'name')]/o/o[text()='%s']",
111                     new DirectivesValue(0, new Format(), name.replace('.', '/'))
112                         .hex(new JavaCodec())
113                 )
114             )
115         );
116     }
117 
118     @Test
119     void createsDirectivesWithMandatorySignature() throws ImpossibleModificationException {
120         MatcherAssert.assertThat(
121             "Can't create proper xml with mandatory signature",
122             new Xembler(
123                 new Directives()
124                     .add("o")
125                     .append(
126                         new DirectivesClass(
127                             "MyClass",
128                             "org/eolang/SomeClass",
129                             new DirectivesClassProperties()
130                         )
131                     ).up()
132             ).xml(),
133             XhtmlMatchers.hasXPaths(
134                 "//o[contains(@name,'signature')]"
135             )
136         );
137     }
138 
139     /**
140      * This test was added to ensure that when a null signature is passed,
141      * it is converted to an empty string in the XML output.
142      * See more details in
143      * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1246">#1246</a>
144      * @throws ImpossibleModificationException in case of XML modification failure.
145      */
146     @Test
147     void convertsNullSignatureToEmptyString() throws ImpossibleModificationException {
148         MatcherAssert.assertThat(
149             "Signature should be empty string when null is passed",
150             new Xembler(
151                 new Directives().add("o").append(
152                     new DirectivesClass(
153                         "MyClass",
154                         null,
155                         new DirectivesClassProperties()
156                     )
157                 ).up()
158             ).xml(),
159             XhtmlMatchers.hasXPath(
160                 "//o[@name='signature']/o[contains(@base,'bytes')]/o[text()='--']"
161             )
162         );
163     }
