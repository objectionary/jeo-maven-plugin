9   import org.junit.jupiter.api.Test;

13  /**
14   * Test for {@link DirectivesType}.
15   * @since 0.11.0
16   */
17  final class DirectivesTypeTest {
18  
19      @Test
20      void convertsTypeToDirectives() {
21          MatcherAssert.assertThat(
22              "We expect the type to be converted to a string internally",
23              new Xembler(new DirectivesType(0, new Format(), Type.BOOLEAN_TYPE)).xmlQuietly(),
24              XhtmlMatchers.hasXPaths(
25                  new JeoBaseXpath("./o", "type").toXpath(),
26                  "/o/o[@base='Q.org.eolang.string']/o[@base='Q.org.eolang.bytes']/o[text()]"
27              )
28          );
29      }
30  }
