5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  import org.objectweb.asm.Opcodes;
12  import org.xembly.Directives;
13  import org.xembly.ImpossibleModificationException;
14  import org.xembly.Xembler;
15  
16  /**
17   * Test case for {@link org.eolang.jeo.representation.directives.DirectivesClassProperties}.
18   * @since 0.1.0
19   */
20  final class DirectivesClassPropertiesTest {
21  
22      @Test
23      void createsDirectives() throws ImpossibleModificationException {
24          MatcherAssert.assertThat(
25              "Can't create proper xml",
26              new Xembler(
27                  new Directives()
28                      .add("o")
29                      .append(
30                          new DirectivesClassProperties(
31                              1,
32                              "org/eolang/SomeClass",
33                              "java/lang/Object",
34                              "org/eolang/SomeInterface"
35                          )
36                      ).up()
37              ).xml(),
38              XhtmlMatchers.hasXPaths(
39                  "/o/o[contains(@base,'number') and contains(@name,'version')]",
40                  "/o/o[contains(@base,'number') and contains(@name,'access')]",
41                  "/o/o[contains(@base,'string') and contains(@name,'signature')]",
42                  "/o/o[contains(@base,'string') and contains(@name,'supername')]",
43                  new JeoBaseXpath("./o/o", "seq.of1").toXpath(),
44                  "/o/o[contains(@name,'interfaces')]"
45              )
46          );
47      }
48  
49      @Test
50      void createsDirectivesWithMandatorySignature() throws ImpossibleModificationException {
51          MatcherAssert.assertThat(
52              "Can't create proper xml with mandatory signature",
53              new Xembler(
54                  new Directives()
55                      .add("o")
56                      .append(
57                          new DirectivesClassProperties(
58                              1,
59                              null,
60                              "java/lang/Object",
61                              "org/eolang/SomeInterface"
62                          )
63                      ).up()
64              ).xml(),
65              XhtmlMatchers.hasXPaths(
66                  "//o[contains(@name,'signature')]"
67              )
68          );
69      }
70  
71      /**
72       * This test was added to ensure that when a null signature is passed,
73       * it is converted to an empty string in the XML output.
74       * See more details in
75       * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1246">#1246</a>
76       * @throws ImpossibleModificationException in case of XML modification failure.
77       */
78      @Test
79      void convertsNullSignatureToEmptyString() throws ImpossibleModificationException {
80          MatcherAssert.assertThat(
81              "Signature should be empty string when null is passed",
82              new Xembler(
83                  new Directives().add("o").append(
84                      new DirectivesClassProperties(
85                          1,
86                          null,
87                          "java/lang/Object",
88                          "org/eolang/SomeInterface"
89                      )
90                  ).up()
91              ).xml(),
92              XhtmlMatchers.hasXPath(
93                  "//o[@name='signature']/o[contains(@base,'bytes')]/o[text()='--']"
94              )
95          );
96      }
97  
98      @Test
99      void addsClassModifiers() throws ImpossibleModificationException {
100         MatcherAssert.assertThat(
101             "We expect class access modifiers to be added to the directives",
102             new Xembler(
103                 new Directives()
104                     .add("o")
105                     .append(
106                         new DirectivesClassProperties(
107                             new Format(Format.MODIFIERS, true),
108                             Opcodes.ACC_INTERFACE
109                         )
110                     ).up()
111             ).xml(),
112             XhtmlMatchers.hasXPath(
113                 "//o[contains(@name, 'modifiers')]/o[contains(@name, 'interface') and contains(@base, 'true')]"
114             )
115         );
116     }
117 
118     @Test
119     void ignoresClassModifiers() throws ImpossibleModificationException {
120         MatcherAssert.assertThat(
121             "We expect class access modifiers to be avoided by default",
122             new Xembler(
123                 new Directives()
124                     .add("o")
125                     .append(new DirectivesClassProperties(Opcodes.ACC_INTERFACE))
126                     .up()
127             ).xml(),
128             Matchers.not(
129                 XhtmlMatchers.hasXPath(
130                     "//o[contains(@name, 'modifiers')]/o[contains(@name, 'interface') and contains(@base, 'true')]"
131                 )
132             )
133         );
134     }
