5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  import org.objectweb.asm.Opcodes;
12  import org.xembly.Directives;
13  import org.xembly.ImpossibleModificationException;
14  import org.xembly.Xembler;
15  
16  /**
17   * Test case for {@link org.eolang.jeo.representation.directives.DirectivesClassProperties}.
18   * @since 0.1.0
19   */
20  final class DirectivesClassPropertiesTest {
21  
22      @Test
23      void createsDirectives() throws ImpossibleModificationException {
24          MatcherAssert.assertThat(
25              "Can't create proper xml",
26              new Xembler(
27                  new Directives()
28                      .add("o")
29                      .append(
30                          new DirectivesClassProperties(
31                              1,
32                              "org/eolang/SomeClass",
33                              "org/eolang/SomeInterface"
34                          )
35                      ).up()
36              ).xml(),
37              XhtmlMatchers.hasXPaths(
38                  "/o/o[contains(@base,'number') and contains(@name,'version')]",
39                  "/o/o[contains(@base,'number') and contains(@name,'access')]",
40                  "/o/o[contains(@base,'string') and contains(@name,'supername')]",
41                  new JeoBaseXpath("./o/o", "seq.of1").toXpath(),
42                  "/o/o[contains(@name,'interfaces')]"
43              )
44          );
45      }
46  
47      @Test
48      void addsClassModifiers() throws ImpossibleModificationException {
49          MatcherAssert.assertThat(
50              "We expect class access modifiers to be added to the directives",
51              new Xembler(
52                  new Directives()
53                      .add("o")
54                      .append(
55                          new DirectivesClassProperties(
56                              new Format(Format.MODIFIERS, true),
57                              Opcodes.ACC_INTERFACE
58                          )
59                      ).up()
60              ).xml(),
61              XhtmlMatchers.hasXPath(
62                  "//o[contains(@name, 'modifiers')]/o[contains(@name, 'interface') and contains(@base, 'true')]"
63              )
64          );
65      }
66  
67      @Test
68      void ignoresClassModifiers() throws ImpossibleModificationException {
69          MatcherAssert.assertThat(
70              "We expect class access modifiers to be avoided by default",
71              new Xembler(
72                  new Directives()
73                      .add("o")
74                      .append(new DirectivesClassProperties(Opcodes.ACC_INTERFACE))
75                      .up()
76              ).xml(),
77              Matchers.not(
78                  XhtmlMatchers.hasXPath(
79                      "//o[contains(@name, 'modifiers')]/o[contains(@name, 'interface') and contains(@base, 'true')]"
80                  )
81              )
82          );
83      }
84  }
