5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.objectweb.asm.Handle;
11  import org.objectweb.asm.Opcodes;
12  import org.xembly.ImpossibleModificationException;
13  import org.xembly.Xembler;
14  
15  /**
16   * Test cases for {@link DirectivesHandle}.
17   * This class verifies the conversion of method handles to directive format,
18   * ensuring proper representation of bytecode handles and their properties.
19   *
20   * @since 0.3.0
21   */
22  final class DirectivesHandleTest {
23  
24      @Test
25      void convertsHandleToDirectives() throws ImpossibleModificationException {
26          final String xml = new Xembler(
27              new DirectivesHandle(
28                  0,
29                  new Format(),
30                  new Handle(Opcodes.H_INVOKESTATIC, "java/lang/Math", "max", "(II)I", false)
31              )
32          ).xml();
33          MatcherAssert.assertThat(
34              String.format("Can't convert Handle to DirectivesHandle. Expected XML:%n%s%n", xml),
35              xml,
36              XhtmlMatchers.hasXPaths(
37                  new JeoBaseXpath("./o", "handle").toXpath(),
38                  "/o/o[contains(@base,'number')]",
39                  "/o/o[contains(@base,'string')]"
40              )
41          );
42      }
43  }
