5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import com.jcabi.xml.XMLDocument;
9   import org.eolang.jeo.representation.ClassName;
10  import org.eolang.jeo.representation.bytecode.BytecodeClass;
11  import org.eolang.jeo.representation.bytecode.BytecodeProgram;
12  import org.hamcrest.MatcherAssert;
13  import org.junit.jupiter.api.Test;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Transformers;
16  import org.xembly.Xembler;
17  
18  /**
19   * Test cases for {@link DirectivesProgram}.
20   * @since 0.1
21   */
22  final class DirectivesProgramTest {
23  
24      @Test
25      void createsCorrectDirectives() {
26          final ClassName name = new ClassName("Foo");
27          final String actual = new Xembler(
28              new DirectivesProgram(new DirectivesClass(name), new DirectivesMetas(name)),
29              new Transformers.Node()
30          ).xmlQuietly();
31          MatcherAssert.assertThat(
32              String.format(
33                  "Can't correctly and without errors create program directives, received XML: %n%s",
34                  new XMLDocument(actual)
35              ),
36              actual,
37              XhtmlMatchers.hasXPath("/program/objects/o[@name='Foo']")
38          );
39      }
40  
41      @Test
42      void setsVersion() throws ImpossibleModificationException {
43          MatcherAssert.assertThat(
44              "We expect that version and revision number will be set from MANIFEST.MF file",
45              new Xembler(
46                  new BytecodeProgram(
47                      new BytecodeClass("Some")
48                  ).directives("")
49              ).xml(),
50              XhtmlMatchers.hasXPaths(
51                  "/program[@version='1.2.3']",
52                  "/program[@revision='1234567']",
53                  "/program[@dob='2023-03-19T00:00:00']"
54              )
55          );
56      }
57  
58      @Test
59      void setsMilliseconds() throws ImpossibleModificationException {
60          final ClassName clazz = new ClassName("Some");
61          final DirectivesProgram program = new DirectivesProgram(
62              "some code",
63              10,
64              new DirectivesClass(clazz),
65              new DirectivesMetas(clazz)
66          );
67          MatcherAssert.assertThat(
68              "We expect that milliseconds will be set",
69              new Xembler(program).xml(),
70              XhtmlMatchers.hasXPath("/program[@ms='10']")
71          );
72      }
73  }
