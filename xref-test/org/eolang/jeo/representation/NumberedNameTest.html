5   package org.eolang.jeo.representation;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Assertions;
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.params.ParameterizedTest;
12  import org.junit.jupiter.params.provider.CsvSource;
13  
14  /**
15   * Test cases for {@link NumberedName}.
16   * This class verifies the functionality of numbered name generation,
17   * ensuring proper handling of name uniqueness and numbering schemes.
18   *
19   * @since 0.9.0
20   */
21  final class NumberedNameTest {
22  
23      @ParameterizedTest
24      @CsvSource({
25          "foo, 1, foo",
26          "foo, 2, foo-2",
27          "foo, 3, foo-3"
28      })
29      void formatsName(final String name, final int number, final String expected) {
30          MatcherAssert.assertThat(
31              String.format(
32                  "Incorrectly formatted name, expected %s, got %s",
33                  expected,
34                  new NumberedName(number, name)
35              ),
36              new NumberedName(number, name).toString(),
37              Matchers.equalTo(expected)
38          );
39      }
40  
41      @Test
42      void throwsExceptionWhenNumberIsLessThanOne() {
43          MatcherAssert.assertThat(
44              "We expect that exception message will be human-readable",
45              Assertions.assertThrows(
46                  IllegalArgumentException.class,
47                  () -> new NumberedName(0, "foo").toString(),
48                  "Exception was not thrown"
49              ).getMessage(),
50              Matchers.equalTo("Number must be greater than 0, but was: 0")
51          );
52      }
53  
54      @ParameterizedTest
55      @CsvSource({
56          "foo, foo",
57          "foo-2, foo",
58          "bar-3, bar",
59          "foobar-4, foobar"
60      })
61      void decodesName(final String encoded, final String expected) {
62          MatcherAssert.assertThat(
63              "Incorrectly decoded name",
64              new NumberedName(encoded).plain(),
65              Matchers.equalTo(expected)
66          );
67      }
68  }
