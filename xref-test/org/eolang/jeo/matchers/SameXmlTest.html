5   package org.eolang.jeo.matchers;
6   
7   import com.jcabi.xml.XML;
8   import com.jcabi.xml.XMLDocument;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.core.IsEqual;
11  import org.junit.jupiter.api.Test;
12  
13  /**
14   * Test cases for {@link SameXml}.
15   * This class verifies the XML comparison functionality,
16   * ensuring that XML documents can be properly matched regardless of formatting differences.
17   *
18   * @since 0.6.0
19   */
20  final class SameXmlTest {
21  
22      @Test
23      void matchesXmls() {
24          final XML input = new XMLDocument(
25              String.join(
26                  "",
27                  "<root>",
28                  "<o line=\"val\">data</o>",
29                  "<anotherElement>",
30                  "<o line=\"\">data</o>",
31                  "</anotherElement>",
32                  "</root>"
33              )
34          );
35          final String lineless = new XMLDocument(
36              String.join(
37                  "",
38                  "<root>",
39                  "<o>data</o>",
40                  "<anotherElement>",
41                  "<o>data</o>",
42                  "</anotherElement>",
43                  "</root>"
44              )
45          ).toString();
46          final boolean matches = new SameXml(input).matchesSafely(lineless);
47          MatcherAssert.assertThat(
48              String.format(
49                  "Input XML ('%s') should match with same lineless XML ('%s'), but was not",
50                  input,
51                  lineless
52              ),
53              matches,
54              new IsEqual<>(true)
55          );
56      }
57  }
