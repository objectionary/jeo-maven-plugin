7   import java.nio.file.Path;

13  import org.hamcrest.MatcherAssert;
14  import org.hamcrest.Matchers;
15  import org.junit.jupiter.api.Test;

17  /**
18   * Test for {@link FilteredClasses}.
19   * @since 0.14.0
20   */
21  final class FilteredClassesTest {
22  
23      @Test
24      void returnsFilteredPathsWhenFilterMatchesSomeWithDefaultLogger() {
25          final Path first = Paths.get("A.class");
26          final Path second = Paths.get("B.class");
27          final List<Path> result = new FilteredClasses(
28              new Project(Stream.of(first, second, Paths.get("C.txt"))),
29              new GlobFilter(Collections.singleton("*.class"), Collections.emptySet())
30          ).all().collect(java.util.stream.Collectors.toList());
31          MatcherAssert.assertThat(
32              "Should return only paths that match the filter",
33              result,
34              Matchers.allOf(
35                  Matchers.iterableWithSize(2),
36                  Matchers.hasItem(first),
37                  Matchers.hasItem(second)
38              )
39          );
40      }
41  
42      @Test
43      void returnsEmptyWhenNoPathsMatchFilter() {
44          MatcherAssert.assertThat(
45              "No paths should match the filter",
46              new FilteredClasses(
47                  new Project(Stream.of(Paths.get("A.txt"), Paths.get("B.txt"))),
48                  new GlobFilter(Collections.singleton("*.class"), Collections.emptySet())
49              ).all().collect(java.util.stream.Collectors.toList()),
50              Matchers.empty()
51          );
52      }
53  
54      @Test
55      void logsMessageWithCorrectCount() {
56          final List<String> logs = new ArrayList<>(1);
57          final Path root = Paths.get("/dev/null");
58          new FilteredClasses(
59              new Project(
60                  root,
61                  Stream.of(
62                      Paths.get("FirstAdded.class"),
63                      Paths.get("SecondAdded.class"),
64                      Paths.get("Skipped.txt")
65                  )
66              ),
67              new GlobFilter(Collections.emptySet(), Collections.singleton("*.txt")), logs::add
68          ).all();
69          MatcherAssert.assertThat(
70              "Should log the correct number of files found",
71              logs,
72              Matchers.allOf(
73                  Matchers.iterableWithSize(1),
74                  Matchers.hasItem(
75                      Matchers.containsString(
76                          String.format(
77                              "Found 2 files in %s using no inclusions and 1 exclusions (*.txt)",
78                              root
79                          )
80                      )
81                  )
82              )
83          );
84      }
85  
86      /**
87       * Test project implementation of {@link Classes}.
88       * <p>
89       *     This class is used to simulate a project structure with
90       *     a root path and a stream of paths.
91       * </p>
92       * @since 0.14.0
93       */
94      private static final class Project implements Classes {
95          /**
96           * Project root path.
97           */
98          private final Path dir;
99  
100         /**
101          * Stream of paths representing project files.
102          */
103         private final Stream<Path> paths;
104 
105         /**
106          * Constructor.
107          * @param paths Stream of paths representing project files
108          */
109         private Project(final Stream<Path> paths) {
110             this(Paths.get("/dev/null"), paths);
111         }
112 
113         /**
114          * Constructor.
115          * @param root Project root path
116          * @param paths Stream of paths representing project files
117          */
118         private Project(final Path root, final Stream<Path> paths) {
119             this.dir = root;
120             this.paths = paths;
121         }
122 
123         @Override
124         public Path root() {
125             return this.dir;
126         }
127 
128         @Override
129         public Stream<Path> all() {
130             return this.paths;
131         }
132 
133         @Override
134         public String toString() {
135             return this.dir.toString();
136         }
137     }
