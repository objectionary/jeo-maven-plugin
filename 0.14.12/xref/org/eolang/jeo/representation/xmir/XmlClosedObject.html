5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.Optional;
8   
9   /**
10   * Closed object in XMIR format.
11   * @since 0.11.0
12   */
13  final class XmlClosedObject {
14  
15      /**
16       * XML node of the closed object.
17       */
18      private final XmlNode node;
19  
20      /**
21       * Constructor.
22       * @param node XML node of the closed object.
23       */
24      XmlClosedObject(final XmlNode node) {
25          this.node = node;
26      }
27  
28      /**
29       * Get the type of the object.
30       * @return Type.
31       */
32      String base() {
33          return this.node.attribute("base").orElseThrow(() -> this.notFound("base"));
34      }
35  
36      /**
37       * Get the option type of the object.
38       * @return Optional type.
39       */
40      Optional<String> optbase() {
41          return this.node.attribute("base");
42      }
43  
44      /**
45       * Create a new exception if the attribute is not found.
46       * @param name Name of the attribute that is not found.
47       * @return Exception indicating that the attribute is not found.
48       */
49      private IllegalStateException notFound(final String name) {
50          return new IllegalStateException(
51              String.format(
52                  "The '%s' node doesn't have '%s' attribute, but it should have one",
53                  this.node, name
54              )
55          );
56      }
57  
58  }
