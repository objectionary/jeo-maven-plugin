5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.stream.Stream;
8   import lombok.ToString;
9   
10  /**
11   * Xml representation of a sequence of XML nodes.
12   * <p>
13   *     Mirrors {@link org.eolang.jeo.representation.directives.DirectivesSeq}
14   * </p>
15   * @since 0.11.0
16   */
17  @ToString
18  final class XmlSeq {
19  
20      /**
21       * Jeo object representing the sequence of XML nodes.
22       */
23      private final XmlJeoObject origin;
24  
25      /**
26       * Constructor.
27       * @param node XML node representing the sequence.
28       */
29      XmlSeq(final XmlNode node) {
30          this(new XmlJeoObject(node));
31      }
32  
33      /**
34       * Constructor.
35       * @param origin XML Jeo object representing the sequence.
36       */
37      XmlSeq(final XmlJeoObject origin) {
38          this.origin = origin;
39      }
40  
41      /**
42       * Name of the sequence.
43       * @return Name of the sequence.
44       */
45      public String name() {
46          return this.origin.name();
47      }
48  
49      /**
50       * Retrieve the child nodes of the sequence.
51       * @return Stream of XML nodes representing the children of the sequence.
52       */
53      public Stream<XmlNode> children() {
54          return this.origin.children();
55      }
56  
57      /**
58       * Whether the sequence is named.
59       * @return True if the sequence has a name attribute, false otherwise.
60       */
61      boolean named() {
62          return this.origin.named();
63      }
64  }
