15  import org.eolang.jeo.representation.xmir.XmlObject;
16  import org.hamcrest.MatcherAssert;
17  import org.hamcrest.Matchers;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.params.ParameterizedTest;
20  import org.junit.jupiter.params.provider.ValueSource;
21  import org.xembly.Xembler;
22  
23  /**
24   * Test case for {@link AsmProgram}.
25   * @since 0.6
26   */
27  final class AsmProgramTest {
28  
29      @Test
30      void convertsToBytecode() {
31          final Bytecode same = new BytecodeObject()
32              .replaceTopClass(new BytecodeClass().helloWorldMethod())
33              .bytecode();
34          MatcherAssert.assertThat(
35              "We expect to receive the same bytecode",
36              new AsmProgram(same.bytes()).bytecode(0).bytecode().toString(),
37              Matchers.equalTo(same.toString())
38          );
39      }
40  
41      @Test
42      void skipsDebugLabels() throws Exception {
43          MatcherAssert.assertThat(
44              "We expect to skip debug labels and lines",
45              new AsmProgram(new BytesOf(new ResourceOf("MethodByte.class")).asBytes())
46                  .bytecode()
47                  .top()
48                  .methods()
49                  .get(1)
50                  .instructions()
51                  .stream()
52                  .anyMatch(entry -> entry instanceof BytecodeLabel || entry instanceof BytecodeLine),
53              Matchers.is(false)
54          );
55      }
56  
57      @Test
58      void parsesAnnotations() throws Exception {
59          final byte[] original = new BytesOf(new ResourceOf("FixedWidth.class")).asBytes();
60          final Bytecode expected = new Bytecode(original);
61          final Bytecode actual = new AsmProgram(original).bytecode().bytecode();
62          MatcherAssert.assertThat(
63              "We expect to receive the same bytecode",
64              actual,
65              Matchers.equalTo(expected)
66          );
67      }
68  
69      @Test
70      void parsesNestMembersBytecodeAttributes() throws Exception {
71          final byte[] original = new BytesOf(new ResourceOf("Check.class")).asBytes();
72          final Bytecode actual = new AsmProgram(original).bytecode(0).bytecode();
73          MatcherAssert.assertThat(
74              "We expect to receive the same bytecode with the 'NestMembers' attribute",
75              actual.toString(),
76              Matchers.equalTo(new Bytecode(original).toString())
77          );
78      }
79  
80      /**
81       * Checks that the AsmProgram correctly parses the 'NestHost' attribute from the bytecode.
82       * Comes from the issue:
83       * <a href="https://github.com/objectionary/jeo-maven-plugin/issues/1274">#1274</a>
84       * @throws Exception In case of error
85       */
86      @Test
87      void parsesNestHostBytecodeAttribute() throws Exception {
88          final byte[] original = new BytesOf(new ResourceOf("Check$1Local.class")).asBytes();
89          MatcherAssert.assertThat(
90              "We expect to receive the same bytecode with the 'NestHost' attribute",
91              new AsmProgram(original).bytecode(0).bytecode().toString(),
92              Matchers.equalTo(new Bytecode(original).toString())
93          );
94      }
95  
96      @Test
97      void convertsToBytecodeThenToXmirAndThenBackToBytecodeScalaBytecode() throws Exception {
98          final BytecodeObject original = new AsmProgram(
99              new BytesOf(new ResourceOf("LogManager.class")).asBytes()
100         ).bytecode(0);
101         final Bytecode actual = new XmlObject(
102             new Xembler(original.directives(new Format())).xml()
103         ).bytecode().bytecode();
104         final Bytecode expected = original.bytecode();
105         MatcherAssert.assertThat(
106             "We expect to receive the same scala bytecode",
107             actual,
108             Matchers.equalTo(expected)
109         );
110     }
111 
112     @ParameterizedTest
113     @ValueSource(strings = {
114         "FixedWidth.class",
115         "DeprecatedMethod.class",
116         "ParamAnnotation.class"
117     })
118     void convertsToBytecodeThenToXmirAndThenBackToBytecode(final String resource) throws Exception {
119         final BytecodeObject bytecode = new AsmProgram(
120             new BytesOf(new ResourceOf(resource)).asBytes()
121         ).bytecode();
122         MatcherAssert.assertThat(
123             "We expect to receive the same bytecode",
124             new XmlObject(
125                 new Xembler(bytecode.directives(new Format())).xml()
126             ).bytecode().bytecode(),
127             Matchers.equalTo(bytecode.bytecode())
128         );
129     }
130 }
