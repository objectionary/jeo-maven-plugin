84  /**
85   * ASM bytecode parser.
86   * We used to use the Visitor pattern, but it's too verbose and not very readable.
87   * So, we decided to switch to a Tree-API-based approach.
88   * You can read more about different approaches right here:
89   * <a href="https://asm.ow2.io/asm4-guide.pdf">https://asm.ow2.io/asm4-guide.pdf</a>
90   * The recent version with the Visitor pattern is still available in the history:
91   * <a href="https://github.com/objectionary/jeo-maven-plugin/tree/29daa0a167b5c2ba4caaceafb6e6bafc381ac05c">github</a>
92   * @since 0.6
93   * @todo #537:60min Refactor {@link AsmProgram} class.
94   *  It's too big and contains a lot of methods.
95   *  We need to refactor it to make it more readable and maintainable.
96   *  Maybe it's worth splitting it into several classes.
97   *  Don't forget to add/update the tests.
98   *  Don't forget to remove PMD and Checkstyle suppressions.
99   * @checkstyle CyclomaticComplexityCheck (500 lines)
100  * @checkstyle JavaNCSSCheck (500 lines)
101  * @checkstyle ClassFanOutComplexityCheck (500 lines)
102  * @checkstyle MethodLengthCheck (500 lines)
103  * @checkstyle AnonInnerLengthCheck (500 lines)
104  */
105 @SuppressWarnings({
106     "PMD.CouplingBetweenObjects",
107     "PMD.TooManyMethods",
108     "PMD.NcssCount",
109     "PMD.UncommentedEmptyMethodBody",
110     "PMD.ExcessiveMethodLength",
111     "PMD.GodClass"
112 })

128     /**
129      * Convert to bytecode.
130      * @return Bytecode.
131      */
132     public BytecodeProgram bytecode() {
133         final ClassNode node = new ClassNode();
134         new ClassReader(this.bytes).accept(node, 0);
135         final ClassName full = new ClassName(node.name);
136         return new BytecodeProgram(
137             full.pckg(),
138             AsmProgram.clazz(node)
139         );
140     }

148         final ClassName full = new ClassName(node.name);
149         return new BytecodeClass(
150             full.name(),
151             AsmProgram.methods(node),
152             AsmProgram.fields(node),

156                 node.version,
157                 node.access,
158                 node.signature,
159                 node.superName,
160                 node.interfaces.toArray(new String[0])
161             )
162         );
163     }

222     /**
223      * Convert asm method to domain method parameters.
224      * @param node Asm method node.
225      * @return Domain method parameters.
226      */
227     @SuppressWarnings({"unchecked", "rawtypes"})
228     private static BytecodeMethodParameters parameters(final MethodNode node) {
229         final List<AnnotationNode>[] invisible;
230         if (node.invisibleParameterAnnotations == null) {
231             invisible = new List[Optional.ofNullable(node.visibleParameterAnnotations)
232                 .orElse(new List[0]).length];
233         } else {
234             invisible = node.invisibleParameterAnnotations;
235         }
236         final List<AnnotationNode>[] visible;
237         if (node.visibleParameterAnnotations == null) {
238             visible = new List[Optional.ofNullable(node.invisibleParameterAnnotations)
239                 .orElse(new List[0]).length];
240         } else {
241             visible = node.visibleParameterAnnotations;
242         }
243         final Type[] types = Type.getArgumentTypes(node.desc);
244         final List<BytecodeMethodParameter> params = new ArrayList<>(types.length);
245         for (int index = 0; index < types.length; ++index) {
246             final List<BytecodeAnnotation> annotations = new ArrayList<>(0);
247             if (visible.length > index) {
248                 annotations.addAll(
249                     AsmProgram.safe(visible[index], true).collect(Collectors.toList())
250                 );
251             }
252             if (invisible.length > index) {
253                 annotations.addAll(
254                     AsmProgram.safe(invisible[index], false).collect(Collectors.toList())
255                 );
256             }
257             params.add(
258                 new BytecodeMethodParameter(
259                     index,
260                     types[index],
261                     new BytecodeAnnotations(annotations)
262                 )
263             );
264         }
265         return new BytecodeMethodParameters(params);
266     }

336     /**
337      * Convert asm instruction to domain instruction.
338      * @param node Asm instruction node.
339      * @return Domain instruction.
340      */
341     private static BytecodeEntry instruction(final AbstractInsnNode node) {
342         final BytecodeEntry result;
343         switch (node.getType()) {
344             case AbstractInsnNode.INSN:
345                 result = new BytecodeInstruction(node.getOpcode());
346                 break;
347             case AbstractInsnNode.INT_INSN:
348                 final IntInsnNode instr = IntInsnNode.class.cast(node);
349                 result = new BytecodeInstruction(
350                     instr.getOpcode(),
351                     instr.operand
352                 );
353                 break;
354             case AbstractInsnNode.VAR_INSN:
355                 final VarInsnNode varinstr = VarInsnNode.class.cast(node);
356                 result = new BytecodeInstruction(
357                     varinstr.getOpcode(), varinstr.var
358                 );
359                 break;
360             case AbstractInsnNode.TYPE_INSN:
361                 final TypeInsnNode typeinstr = TypeInsnNode.class.cast(node);
362                 result = new BytecodeInstruction(
363                     typeinstr.getOpcode(),
364                     typeinstr.desc
365                 );
366                 break;
367             case AbstractInsnNode.FIELD_INSN:
368                 final FieldInsnNode field = FieldInsnNode.class.cast(node);
369                 result = new BytecodeInstruction(
370                     field.getOpcode(),
371                     field.owner,
372                     field.name,
373                     field.desc
374                 );
375                 break;
376             case AbstractInsnNode.METHOD_INSN:
377                 final MethodInsnNode method = MethodInsnNode.class.cast(node);
378                 result = new BytecodeInstruction(
379                     method.getOpcode(),
380                     method.owner,
381                     method.name,
382                     method.desc,
383                     method.itf
384                 );
385                 break;
386             case AbstractInsnNode.INVOKE_DYNAMIC_INSN:
387                 final InvokeDynamicInsnNode dynamic = InvokeDynamicInsnNode.class.cast(node);
388                 result = new BytecodeInstruction(
389                     dynamic.getOpcode(),
390                     Stream.concat(
391                         Stream.of(
392                             dynamic.name,
393                             dynamic.desc,
394                             dynamic.bsm
395                         ),
396                         Arrays.stream(dynamic.bsmArgs)
397                     ).toArray(Object[]::new)
398                 );
399                 break;
400             case AbstractInsnNode.JUMP_INSN:
401                 final JumpInsnNode jump = JumpInsnNode.class.cast(node);
402                 result = new BytecodeInstruction(
403                     jump.getOpcode(),
404                     jump.label.getLabel()
405                 );
406                 break;
407             case AbstractInsnNode.LABEL:
408                 final LabelNode label = LabelNode.class.cast(node);
409                 result = new BytecodeLabel(
410                     label.getLabel()
411                 );
412                 break;
413             case AbstractInsnNode.LDC_INSN:
414                 final LdcInsnNode ldc = LdcInsnNode.class.cast(node);
415                 result = new BytecodeInstruction(
416                     ldc.getOpcode(),
417                     ldc.cst
418                 );
419                 break;
420             case AbstractInsnNode.IINC_INSN:
421                 final IincInsnNode iinc = IincInsnNode.class.cast(node);
422                 result = new BytecodeInstruction(
423                     iinc.getOpcode(),
424                     iinc.var,
425                     iinc.incr
426                 );
427                 break;
428             case AbstractInsnNode.TABLESWITCH_INSN:
429                 final TableSwitchInsnNode table = TableSwitchInsnNode.class.cast(node);
430                 result = new BytecodeInstruction(
431                     table.getOpcode(),
432                     Stream.concat(
433                         Stream.of(table.min, table.max, table.dflt.getLabel()),
434                         table.labels.stream().map(LabelNode::getLabel)
435                     ).toArray(Object[]::new)
436                 );
437                 break;
438             case AbstractInsnNode.LOOKUPSWITCH_INSN:
439                 final LookupSwitchInsnNode lookup = LookupSwitchInsnNode.class.cast(node);
440                 result = new BytecodeInstruction(
441                     lookup.getOpcode(),
442                     Stream.concat(
443                         Stream.of(lookup.dflt.getLabel()),
444                         Stream.concat(
445                             lookup.keys.stream(),
446                             lookup.labels.stream().map(LabelNode::getLabel)
447                         )
448                     ).toArray(Object[]::new)
449                 );
450                 break;
451             case AbstractInsnNode.MULTIANEWARRAY_INSN:
452                 final MultiANewArrayInsnNode multiarr = MultiANewArrayInsnNode.class.cast(node);
453                 result = new BytecodeInstruction(
454                     multiarr.getOpcode(),
455                     multiarr.desc,
456                     multiarr.dims
457                 );
458                 break;
459             case AbstractInsnNode.FRAME:
460                 final FrameNode frame = FrameNode.class.cast(node);
461                 result = new BytecodeFrame(frame.type, frame.local, frame.stack);
462                 break;
463             case AbstractInsnNode.LINE:
464                 result = new BytecodeLine();
465                 break;
466             default:
467                 throw new IllegalStateException(
468                     String.format("Unknown instruction type: %s", node)
469                 );
470         }
471         return result;
472     }

531     /**
532      * Convert asm annotation to domain annotation.
533      * @param node Asm annotation node.
534      * @param visible Is it visible?
535      * @return Domain annotation.
536      */
537     private static BytecodeAnnotation annotation(final AnnotationNode node, final boolean visible) {
538         final List<BytecodeAnnotationValue> properties = new ArrayList<>(0);
539         final List<Object> values = Optional.ofNullable(node.values)
540             .orElse(new ArrayList<>(0));
541         for (int index = 0; index < values.size(); index += 2) {
542             properties.add(
543                 AsmProgram.annotationProperty(
544                     (String) values.get(index),
545                     values.get(index + 1)
546                 )
547             );
548         }
549         return new BytecodeAnnotation(node.desc, visible, properties);
550     }

552     /**
553      * Convert asm annotation property to domain annotation property.
554      * @param name Property name.
555      * @param value Property value.
556      * @return Domain annotation.
557      */
558     private static BytecodeAnnotationValue annotationProperty(
559         final String name, final Object value
560     ) {
561         final BytecodeAnnotationValue result;
562         if (value instanceof String[]) {
563             final String[] params = (String[]) value;
564             result = new BytecodeEnumAnnotationValue(name, params[0], params[1]);
565         } else if (value instanceof AnnotationNode) {
566             final AnnotationNode cast = AnnotationNode.class.cast(value);
567             result = new BytecodeAnnotationAnnotationValue(
568                 name,
569                 cast.desc,
570                 Optional.ofNullable(cast.values)
571                     .map(Collection::stream)
572                     .orElseGet(Stream::empty)
573                     .map(val -> AsmProgram.annotationProperty("", val))
574                     .collect(Collectors.toList())
575             );
576         } else if (value instanceof List) {
577             result = new BytecodeArrayAnnotationValue(
578                 name,
579                 ((Collection<?>) value).stream()
580                     .map(val -> AsmProgram.annotationProperty("", val))
581                     .collect(Collectors.toList())
582             );
583         } else {
584             result = new BytecodePlainAnnotationValue(name, value);
585         }
586         return result;
587     }

589     /**
590      * Convert asm default value to domain default value.
591      * @param node Asm method node.
592      * @return Domain default value.
593      */
594     private static List<BytecodeDefaultValue> defvalues(final MethodNode node) {
595         final List<BytecodeDefaultValue> result;
596         if (node.annotationDefault == null) {
597             result = Collections.emptyList();
598         } else {
599             result = Collections.singletonList(
600                 new BytecodeDefaultValue(
601                     AsmProgram.annotationProperty(null, node.annotationDefault)
602                 )
603             );
604         }
605         return result;
606     }
