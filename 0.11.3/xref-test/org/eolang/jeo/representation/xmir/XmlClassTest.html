5   package org.eolang.jeo.representation.xmir;
6   
7   import org.eolang.jeo.representation.bytecode.BytecodeClass;
8   import org.eolang.jeo.representation.bytecode.BytecodeClassProperties;
9   import org.eolang.jeo.representation.directives.DirectivesClass;
10  import org.eolang.jeo.representation.directives.DirectivesClassProperties;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.junit.jupiter.api.Test;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Xembler;
16  
17  /**
18   * Test cases for {@link XmlClass}.
19   * This class verifies the functionality of XML class representation,
20   * ensuring proper creation and manipulation of class objects from XML.
21   *
22   * @since 0.1.0
23   */
24  final class XmlClassTest {
25  
26      @Test
27      void createsByName() {
28          final String expected = "FooClass";
29          final XmlClass klass = new XmlClass(expected);
30          MatcherAssert.assertThat(
31              String.format(
32                  "%s should create a class with name %s%n",
33                  klass,
34                  expected
35              ),
36              klass.bytecode(),
37              Matchers.equalTo(new BytecodeClass(expected))
38          );
39      }
40  
41      @Test
42      void createsBytecode() throws ImpossibleModificationException {
43          final String name = "Foo";
44          final int access = 100;
45          final String signature = "java/lang/Object";
46          final String supername = "java/lang/Runnable";
47          final String interfce = "java/lang/Cloneable";
48          final String pckg = "org.eolang.foo";
49          MatcherAssert.assertThat(
50              "We expect that class created from directives is equal to expected",
51              new XmlClass(
52                  pckg,
53                  new NativeXmlNode(
54                      new Xembler(
55                          new DirectivesClass(
56                              name,
57                              new DirectivesClassProperties(
58                                  access,
59                                  signature,
60                                  supername,
61                                  interfce
62                              )
63                          )
64                      ).xml()
65                  )
66              ).bytecode(),
67              Matchers.equalTo(
68                  new BytecodeClass(
69                      String.format("%s.%s", pckg, name),
70                      new BytecodeClassProperties(access, signature, supername, interfce)
71                  )
72              )
73          );
74      }
75  }
