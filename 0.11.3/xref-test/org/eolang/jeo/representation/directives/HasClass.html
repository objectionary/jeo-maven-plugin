5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.xml.XMLDocument;
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.stream.Collectors;
11  import java.util.stream.Stream;
12  import org.eolang.jeo.representation.PrefixedName;
13  import org.hamcrest.Description;
14  import org.hamcrest.TypeSafeMatcher;
15  
16  /**
17   * Hamcrest matcher to verify that an XMIR document contains a class with a specific name.
18   * This matcher supports additional checks for package verification and provides
19   * detailed error descriptions when matches fail.
20   *
21   * @since 0.1.0
22   */
23  @SuppressWarnings({
24      "JTCOP.RuleAllTestsHaveProductionClass",
25      "JTCOP.RuleCorrectTestName",
26      "JTCOP.RuleInheritanceInTests"
27  })
28  public final class HasClass extends TypeSafeMatcher<String> {
29  
30      /**
31       * Class name.
32       */
33      private final String name;
34  
35      /**
36       * Additional checks.
37       * List of xpaths that should be checked.
38       */
39      private final List<String> additional;
40  
41      /**
42       * Constructor.
43       *
44       * @param name Class name to match in the XMIR document
45       */
46      public HasClass(final String name) {
47          this.name = new PrefixedName(name).encode();
48          this.additional = new ArrayList<>(0);
49      }
50  
51      /**
52       * Add additional check for package.
53       *
54       * @param pckg Package name to verify
55       * @return This matcher for method chaining
56       */
57      public HasClass inside(final String pckg) {
58          this.additional.add("/object/metas/meta/head[text()='package']/text()");
59          this.additional.add(
60              String.format(
61                  "/object/metas/meta/tail[text()='%s']/text()",
62                  new PrefixedName(pckg).encode()
63              )
64          );
65          this.additional.add(
66              String.format(
67                  "/object/metas/meta/part[text()='%s']/text()",
68                  new PrefixedName(pckg).encode()
69              )
70          );
71          return this;
72      }
73  
74      @Override
75      public boolean matchesSafely(final String item) {
76          final XMLDocument document = new XMLDocument(item);
77          return this.checks().stream().map(document::xpath).noneMatch(List::isEmpty);
78      }
79  
80      @Override
81      public void describeTo(final Description description) {
82          description.appendText("XMIR doesn't satisfy the following XPath expressions:\n")
83              .appendValueList("", "\n", "", this.checks());
84      }
85  
86      /**
87       * Get all XPath expressions for validation.
88       *
89       * @return List of XPath expressions to check
90       */
91      private List<String> checks() {
92          return Stream.concat(
93              Stream.of(String.format("/object/o[@name='%s']/text()", this.name)),
94              this.additional.stream()
95          ).collect(Collectors.toList());
96      }
97  }
