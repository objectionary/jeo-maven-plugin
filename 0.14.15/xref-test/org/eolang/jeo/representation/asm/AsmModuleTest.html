28  final class AsmModuleTest {
29  
30      @Test
31      void convertsSimpleModuleToDomainBytecode() {
32          final String name = "name";
33          final int access = 0;
34          final String version = "0.1.0";
35          final ModuleNode node = new ModuleNode(name, access, version);
36          final String main = "main";
37          node.mainClass = main;
38          final BytecodeModule bytecode = new AsmModule(node).bytecode();
39          MatcherAssert.assertThat(
40              "We expect the module to be converted to domain bytecode correctly",
41              bytecode,
42              org.hamcrest.Matchers.equalTo(
43                  new BytecodeModule(
44                      name,
45                      access,
46                      version,
47                      main,
48                      Collections.emptyList(),
49                      Collections.emptyList(),
50                      Collections.emptyList(),
51                      Collections.emptyList(),
52                      Collections.emptyList(),
53                      Collections.emptyList()
54                  )
55              )
56          );
57      }
58  
59      @Test
60      void convertsModuleWithPackagesToDomainBytecode() {
61          final String name = "name";
62          final int access = 0;
63          final String version = "0.1.0";
64          final ModuleNode node = new ModuleNode(
65              name,
66              access,
67              version
68          );
69          final String main = "main";
70          node.mainClass = main;
71          node.packages = new ArrayList<>(2);
72          final String example = "com.example";
73          final String sample = "org.sample";
74          node.packages.add(example);
75          node.packages.add(sample);
76          final BytecodeModule bytecode = new AsmModule(node).bytecode();
77          MatcherAssert.assertThat(
78              "We expect the module with packages to be converted to domain bytecode correctly",
79              bytecode,
80              org.hamcrest.Matchers.equalTo(
81                  new BytecodeModule(
82                      name,
83                      access,
84                      version,
85                      main,
86                      Arrays.asList(example, sample),
87                      Collections.emptyList(),
88                      Collections.emptyList(),
89                      Collections.emptyList(),
90                      Collections.emptyList(),
91                      Collections.emptyList()
92                  )
93              )
94          );
95      }
96  
97      @Test
98      void convertsModuleWithEverythingToDomainBytecode() {
99          final String name = "name";
100         final int access = 0;
101         final String version = "0.1.0";
102         final ModuleNode node = new ModuleNode(
103             name,
104             access,
105             version
106         );
107         final String main = "main";
108         node.mainClass = main;
109         node.packages = new ArrayList<>(2);
110         final String example = "com.example";
111         final String sample = "org.sample";
112         node.packages.add(example);
113         node.packages.add(sample);
114         node.requires = new ArrayList<>(1);
115         node.requires.add(new ModuleRequireNode("required", 0, "0.1.1"));
116         node.exports = new ArrayList<>(1);
117         final List<String> emodules = Collections.singletonList("m1");
118         node.exports.add(new ModuleExportNode("exported", 0, emodules));
119         node.opens = new ArrayList<>(1);
120         final List<String> omodules = Collections.singletonList("m2");
121         node.opens.add(new ModuleOpenNode("opened", 0, omodules));
122         node.provides = new ArrayList<>(1);
123         final List<String> impls = new ArrayList<>(1);
124         impls.add("impl");
125         node.provides.add(new ModuleProvideNode("service", impls));
126         node.uses = new ArrayList<>(1);
127         node.uses.add("used");
128         final BytecodeModule bytecode = new AsmModule(node).bytecode();
129         MatcherAssert.assertThat(
130             "We expect the module with everything to be converted to domain bytecode correctly",
131             bytecode,
132             org.hamcrest.Matchers.equalTo(
133                 new BytecodeModule(
134                     name,
135                     access,
136                     version,
137                     main,
138                     Arrays.asList(example, sample),
139                     Collections.singletonList(
140                         new BytecodeModuleRequired("required", 0, "0.1.1")
141                     ),
142                     Collections.singletonList(
143                         new BytecodeModuleExported("exported", 0, emodules)
144                     ),
145                     Collections.singletonList(
146                         new BytecodeModuleOpened("opened", 0, omodules)
147                     ),
148                     Collections.singletonList(
149                         new BytecodeModuleProvided("service", impls)
150                     ),
151                     Collections.singletonList("used")
152                 )
153             )
154         );
155     }
