5   package org.eolang.jeo.representation.bytecode;
6   
7   import lombok.EqualsAndHashCode;
8   import lombok.ToString;
9   import org.eolang.jeo.representation.directives.DirectivesModuleRequired;
10  import org.eolang.jeo.representation.directives.Format;
11  import org.objectweb.asm.ModuleVisitor;
12  
13  /**
14   * A node that represents a required module with its name and access of a module descriptor.
15   * @since 0.15.0
16   */
17  @ToString
18  @EqualsAndHashCode
19  public final class BytecodeModuleRequired {
20  
21      /**
22       * The fully qualified name (using dots) of the dependence.
23       * */
24      private final String module;
25  
26      /**
27       * The access flag of the dependence.
28       * Among {@code ACC_TRANSITIVE}, {@code ACC_STATIC_PHASE},
29       * {@code ACC_SYNTHETIC} and {@code ACC_MANDATED}.
30       */
31      private final int access;
32  
33      /**
34       * The module version at compile time.
35       */
36      private final String version;
37  
38      /**
39       * Constructor.
40       * @param module The fully qualified name (using dots) of the dependence
41       * @param access The access flag of the dependence
42       * @param version The module version at compile time
43       */
44      public BytecodeModuleRequired(final String module, final int access, final String version) {
45          this.module = module;
46          this.access = access;
47          this.version = version;
48      }
49  
50      /**
51       * Writes this required module to the given module visitor.
52       * @param visitor The module visitor
53       */
54      public void write(final ModuleVisitor visitor) {
55          visitor.visitRequire(this.module, this.access, this.version);
56      }
57  
58      /**
59       * Converts this required module to directives.
60       * @param format Directive format
61       * @return Directives representation of this required module
62       */
63      public DirectivesModuleRequired directives(final Format format) {
64          return new DirectivesModuleRequired(format, this.module, this.access, this.version);
65      }
66  }
