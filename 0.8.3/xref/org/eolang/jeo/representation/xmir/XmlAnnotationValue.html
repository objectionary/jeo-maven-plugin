5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.List;
8   import java.util.stream.Collectors;
9   import org.eolang.jeo.representation.bytecode.BytecodeAnnotationAnnotationValue;
10  import org.eolang.jeo.representation.bytecode.BytecodeAnnotationValue;
11  import org.eolang.jeo.representation.bytecode.BytecodeArrayAnnotationValue;
12  import org.eolang.jeo.representation.bytecode.BytecodeEnumAnnotationValue;
13  import org.eolang.jeo.representation.bytecode.BytecodePlainAnnotationValue;
14  
15  /**
16   * Xmir annotation property.
17   * @since 0.3
18   */
19  public final class XmlAnnotationValue {
20  
21      /**
22       * Annotation property XML node.
23       */
24      private final XmlNode node;
25  
26      /**
27       * Constructor.
28       * @param xmlnode XML node.
29       */
30      public XmlAnnotationValue(final XmlNode xmlnode) {
31          this.node = xmlnode;
32      }
33  
34      /**
35       * Transform to bytecode.
36       * @return Bytecode annotation property.
37       */
38      public BytecodeAnnotationValue bytecode() {
39          final List<Object> params = this.params();
40          final BytecodeAnnotationValue result;
41          switch (this.type()) {
42              case "PLAIN":
43                  result = new BytecodePlainAnnotationValue(
44                      String.valueOf(params.get(0)),
45                      params.get(1)
46                  );
47                  break;
48              case "ARRAY":
49                  result = new BytecodeArrayAnnotationValue(
50                      String.valueOf(params.get(0)),
51                      params.stream()
52                          .skip(1)
53                          .map(BytecodeAnnotationValue.class::cast)
54                          .collect(Collectors.toList())
55                  );
56                  break;
57              case "ANNOTATION":
58                  result = new BytecodeAnnotationAnnotationValue(
59                      String.valueOf(params.get(0)),
60                      String.valueOf(params.get(1)),
61                      params.stream()
62                          .skip(2)
63                          .map(BytecodeAnnotationValue.class::cast)
64                          .collect(Collectors.toList())
65                  );
66                  break;
67              case "ENUM":
68                  result = new BytecodeEnumAnnotationValue(
69                      String.valueOf(params.get(0)),
70                      String.valueOf(params.get(1)),
71                      String.valueOf(params.get(2))
72                  );
73                  break;
74              default:
75                  throw new IllegalArgumentException(
76                      String.format("Unknown annotation property type: %s", this.type())
77                  );
78          }
79          return result;
80      }
81  
82      /**
83       * Type of the property.
84       * @return Type.
85       */
86      private String type() {
87          return (String) new XmlOperand(
88              this.node.children().collect(Collectors.toList()).get(0)
89          ).asObject();
90      }
91  
92      /**
93       * Property parameters.
94       * @return Parameters.
95       */
96      private List<Object> params() {
97          return this.node.children()
98              .skip(1)
99              .map(XmlOperand::new)
100             .map(XmlOperand::asObject)
101             .collect(Collectors.toList());
102     }
103 }
