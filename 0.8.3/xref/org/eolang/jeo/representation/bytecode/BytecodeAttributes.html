5   package org.eolang.jeo.representation.bytecode;
6   
7   import java.util.Arrays;
8   import java.util.List;
9   import java.util.stream.Collectors;
10  import lombok.EqualsAndHashCode;
11  import lombok.ToString;
12  import org.eolang.jeo.representation.asm.AsmLabels;
13  import org.eolang.jeo.representation.directives.DirectivesAttributes;
14  import org.objectweb.asm.ClassVisitor;
15  import org.objectweb.asm.MethodVisitor;
16  
17  /**
18   * Bytecode attributes.
19   * @since 0.6
20   */
21  @ToString
22  @EqualsAndHashCode
23  public final class BytecodeAttributes {
24  
25      /**
26       * All attributes.
27       */
28      private final List<BytecodeAttribute> all;
29  
30      /**
31       * Constructor.
32       * @param all All attributes.
33       */
34      public BytecodeAttributes(final BytecodeAttribute... all) {
35          this(Arrays.asList(all));
36      }
37  
38      /**
39       * Constructor.
40       * @param all All attributes.
41       */
42      public BytecodeAttributes(final List<BytecodeAttribute> all) {
43          this.all = all;
44      }
45  
46      /**
47       * Convert to directives.
48       * @param name Name of the attributes in EO representation.
49       * @return Directives.
50       */
51      public DirectivesAttributes directives(final String name) {
52          return new DirectivesAttributes(
53              name,
54              this.all.stream().map(BytecodeAttribute::directives).collect(Collectors.toList())
55          );
56      }
57  
58      /**
59       * Write to class.
60       * @param clazz Bytecode where to write.
61       */
62      void write(final ClassVisitor clazz) {
63          this.all.forEach(attr -> attr.write(clazz));
64      }
65  
66      /**
67       * Write to method.
68       * @param method Bytecode where to write.
69       * @param labels Method labels.
70       */
71      void write(final MethodVisitor method, final AsmLabels labels) {
72          this.all.forEach(attr -> attr.write(method, labels));
73      }
74  }
