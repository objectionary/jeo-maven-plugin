5   package org.eolang.jeo.representation;
6   
7   import com.yegor256.Together;
8   import org.cactoos.list.ListOf;
9   import org.cactoos.text.FormattedText;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.RepeatedTest;
13  import org.junit.jupiter.api.Test;
14  
15  /**
16   * Test suite for {@link License}.
17   *
18   * @since 0.6.27
19   */
20  final class LicenseTest {
21  
22      @Test
23      void readsLicenseFileCorrectly() throws Exception {
24          final String name = "LICENSE.txt";
25          MatcherAssert.assertThat(
26              new FormattedText("Unexpected file:'%s' content", name).asString(),
27              new License(name).value(),
28              Matchers.containsString("MIT")
29          );
30      }
31  
32      @RepeatedTest(10)
33      void readsLicenseConcurrently() {
34          final License license = new License();
35          final int total = 10;
36          MatcherAssert.assertThat(
37              "Unexpected license content",
38              new ListOf<>(new Together<>(total, i -> license.value())),
39              Matchers.everyItem(Matchers.containsString("MIT"))
40          );
41      }
42  }
