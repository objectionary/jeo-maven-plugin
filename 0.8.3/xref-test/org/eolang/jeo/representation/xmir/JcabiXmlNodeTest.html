5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.List;
8   import java.util.Optional;
9   import java.util.stream.Collectors;
10  import org.cactoos.list.ListOf;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.hamcrest.core.IsEqual;
14  import org.junit.jupiter.api.Test;
15  
16  /**
17   * Test for {@link JcabiXmlNode}.
18   * @since 0.7
19   */
20  final class JcabiXmlNodeTest {
21  
22      @Test
23      void retrievesTheFirstChild() {
24          MatcherAssert.assertThat(
25              "Can't retrieve the first child, or the first child is not the expected one",
26              new JcabiXmlNode("<o><o name='inner'/></o>").firstChild(),
27              Matchers.equalTo(new JcabiXmlNode("<o name='inner'/>"))
28          );
29      }
30  
31      @Test
32      void retrievesChildren() {
33          final List<XmlNode> children = new JcabiXmlNode(
34              "<o><o name='inner1'/><o name='inner2'/></o>"
35          ).children().collect(Collectors.toList());
36          MatcherAssert.assertThat(
37              "Size of children is not as expected",
38              children,
39              Matchers.hasSize(2)
40          );
41          MatcherAssert.assertThat(
42              "Can't retrieve the children, or the children are not the expected ones",
43              children,
44              Matchers.contains(
45                  new JcabiXmlNode("<o name='inner1'/>"),
46                  new JcabiXmlNode("<o name='inner2'/>")
47              )
48          );
49      }
50  
51      @Test
52      void retrievesAttribute() {
53          final Optional<String> attribute = new JcabiXmlNode("<o name='some'/>").attribute("name");
54          MatcherAssert.assertThat(
55              "Can't retrieve the attribute",
56              attribute.isPresent(),
57              Matchers.is(true)
58          );
59          MatcherAssert.assertThat(
60              "he attribute is not the expected one",
61              attribute.get(),
62              Matchers.equalTo("some")
63          );
64      }
65  
66      @Test
67      void retrievesText() {
68          MatcherAssert.assertThat(
69              "Can't retrieve the text, or the text is not the expected one",
70              new JcabiXmlNode("<o>text</o>").text(),
71              Matchers.equalTo("text")
72          );
73      }
74  
75      @Test
76      void retrievesChild() {
77          final XmlNode child = new JcabiXmlNode(
78              "<program><o>text</o></program>"
79          ).child("o");
80          final String expected = "<o>text</o>";
81          MatcherAssert.assertThat(
82              String.format(
83                  "Retrieved XML: %s does not match with expected %s",
84                  child,
85                  expected
86              ),
87              child,
88              Matchers.equalTo(new JcabiXmlNode(expected))
89          );
90      }
91  
92      @Test
93      void retrievesChildObjects() {
94          final List<XmlNode> objects = new JcabiXmlNode(
95              "<program><o>o1</o><o>o2</o></program>"
96          ).children().collect(Collectors.toList());
97          final List<JcabiXmlNode> expected = new ListOf<>(
98              new JcabiXmlNode("<o>o1</o>"),
99              new JcabiXmlNode("<o>o2</o>")
100         );
101         MatcherAssert.assertThat(
102             String.format(
103                 "Retrieved child objects: %s don't match with expected: %s",
104                 objects,
105                 expected
106             ),
107             objects,
108             new IsEqual<>(expected)
109         );
110     }
111 
112     @Test
113     void retrievesByXpath() {
114         final List<String> atual = new JcabiXmlNode(
115             "<program><o>o1</o><o>o2</o></program>"
116         ).xpath("//o/text()");
117         final List<String> expected = new ListOf<>("o1", "o2");
118         MatcherAssert.assertThat(
119             String.format(
120                 "Retrieved xpath: %s don't match with expected: %s",
121                 atual,
122                 expected
123             ),
124             atual,
125             Matchers.equalTo(expected)
126         );
127     }
