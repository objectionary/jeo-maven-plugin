24  package org.eolang.jeo.representation.directives;
25  
26  import java.util.Iterator;
27  import java.util.regex.Matcher;
28  import java.util.regex.Pattern;
29  import org.xembly.Directive;
30  import org.xembly.Directives;
31  
32  /**
33   * Directives for a comment.
34   * @since 0.6
35   */
36  public final class DirectivesComment implements Iterable<Directive> {
37  
38      /**
39       * Unsafe characters.
40       */
41      private static final Pattern UNSAFE_CHARS = Pattern.compile("[&<>\"'-]");
42  
43      /**
44       * Chars that are discouraged in XML.
45       * By this pattern, we remove all discouraged characters from the comment.
46       */
47      private static final Pattern DISCOURAGED = Pattern.compile(
48          String.join(
49              "|",
50              "[\\x00-\\x08]",
51              "[\\x{1}-\\x{8}]",
52              "[\\x{B}-\\x{C}]",
53              "[\\x{E}-\\x{1F}]",
54              "[\\x{7F}-\\x{84}]",
55              "[\\x{86}-\\x{9F}]",
56              "[\\x{FDD0}-\\x{FDDF}]",
57              "[\\x{1FFFE}-\\x{1FFFF}]",
58              "[\\x{2FFFE}-\\x{2FFFF}]",
59              "[\\x{3FFFE}-\\x{3FFFF}]",
60              "[\\x{4FFFE}-\\x{4FFFF}]",
61              "[\\x{5FFFE}-\\x{5FFFF}]",
62              "[\\x{6FFFE}-\\x{6FFFF}]",
63              "[\\x{7FFFE}-\\x{7FFFF}]",
64              "[\\x{8FFFE}-\\x{8FFFF}]",
65              "[\\x{9FFFE}-\\x{9FFFF}]",
66              "[\\x{AFFFE}-\\x{AFFFF}]",
67              "[\\x{BFFFE}-\\x{BFFFF}]",
68              "[\\x{CFFFE}-\\x{CFFFF}]",
69              "[\\x{DFFFE}-\\x{DFFFF}]",
70              "[\\x{EFFFE}-\\x{EFFFF}]",
71              "[\\x{FFFFE}-\\x{FFFFF}]",
72              "[\\x{10FFFE}-\\x{10FFFF}]"
73          )
74      );
75  
76      /**
77       * Comment.
78       */
79      private final String comment;
80  
81      /**
82       * Constructor.
83       * @param comment Comment.
84       */
85      DirectivesComment(final String comment) {
86          this.comment = comment;
87      }
88  
89      @Override
90      public Iterator<Directive> iterator() {
91          final Iterator<Directive> result;
92          if (this.comment.isEmpty()) {
93              result = new Directives().iterator();
94          } else {
95              result = new Directives().comment(this.escaped()).iterator();
96          }
97          return result;
98      }
99  
100     /**
101      * Escapes unsafe characters.
102      * @return Escaped comment.
103      */
104     private String escaped() {
105         final Matcher matcher = DirectivesComment.UNSAFE_CHARS.matcher(this.comment);
106         final StringBuffer result = new StringBuffer(0);
107         while (matcher.find()) {
108             final String replacement;
109             switch (matcher.group()) {
110                 case "&":
111                     replacement = "&amp;";
112                     break;
113                 case "<":
114                     replacement = "&lt;";
115                     break;
116                 case ">":
117                     replacement = "&gt;";
118                     break;
119                 case "\"":
120                     replacement = "&quot;";
121                     break;
122                 case "'":
123                     replacement = "&apos;";
124                     break;
125                 case "-":
126                     replacement = "&#45;";
127                     break;
128                 default:
129                     throw new IllegalStateException(
130                         String.format("Unexpected value: %s", matcher.group())
131                     );
132             }
133             matcher.appendReplacement(result, replacement);
134         }
135         matcher.appendTail(result);
136         return DirectivesComment.DISCOURAGED.matcher(result).replaceAll("");
137     }
138 }
