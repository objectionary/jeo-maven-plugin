45  /**
46   * Custom class writer.
47   *
48   * @since 0.3
49   */
50  public final class CustomClassWriter extends ClassVisitor {
51  
52      /**
53       * Bytecode writer.
54       */
55      private final ClassesAwareWriter writer;
56  
57      /**
58       * Constructor.
59       * @param verify Verify bytecode.
60       */
61      CustomClassWriter(final boolean verify) {
62          this(CustomClassWriter.prestructor(verify));
63      }
64  
65      /**
66       * Constructor.
67       * @param writer Writer.
68       */
69      private CustomClassWriter(final ClassesAwareWriter writer) {
70          this(new DefaultVersion().api(), writer);
71      }
72  
73      /**
74       * Constructor.
75       * @param api Java ASM API version.
76       * @param writer Writer.
77       */
78      private CustomClassWriter(final int api, final ClassesAwareWriter writer) {
79          super(api, writer);
80          this.writer = writer;
81      }
82  
83      /**
84       * Generate class bytecode.
85       * @return Bytecode.
86       */
87      public Bytecode bytecode() {
88          return new Bytecode(this.writer.toByteArray());
89      }
90  
91      /**
92       * Visits a method of the class.
93       * @param access Access flags.
94       * @param name Method name.
95       * @param descriptor Method descriptor.
96       * @param signature Method signature.
97       * @param exceptions Method exceptions.
98       * @param compute If frames should be computed.
99       * @return Method visitor.
100      * @checkstyle ParameterNumberCheck (5 lines)
101      */
102     MethodVisitor visitMethod(
103         final int access,
104         final String name,
105         final String descriptor,
106         final String signature,
107         final String[] exceptions,
108         final boolean compute
109     ) {
110         final MethodVisitor result;
111         if (compute) {
112             result = this.visitMethodWithoutFrames(access, name, descriptor, signature, exceptions);
113         } else {
114             result = this.visitMethod(access, name, descriptor, signature, exceptions);
115         }
116         return result;
117     }
118 
119     /**
120      * Visits a method of the class and compute all required stack values, locals, and frames.
121      * @param access Access flags.
122      * @param name Method name.
123      * @param descriptor Method descriptor.
124      * @param signature Method signature.
125      * @param exceptions Method exceptions.
126      * @return Method visitor.
127      * @todo #528:90min Remove ad-hoc solution for method frames, stack and locals computation.
128      *  This method is a workaround for the issue with the ASM library that doesn't allow
129      *  to compute frames and stack values per method basis. Here I use reflection to change
130      *  the computation mode to COMPUTE_ALL_FRAMES and then restore it back to the original value.
131      *  This is a hacky solution and should be removed once the issue is fixed in the ASM library.
132      *  Another option is to compute all this values ourselves.
133      *  You can read more about this issue here:
134      *  https://stackoverflow.com/questions/78262674/how-to-mix-manual-and-automatic-calculation-of-max-locals-max-stack-and-frames
135      * @checkstyle ParameterNumberCheck (25 lines)
136      */
137     @SuppressWarnings("PMD.AvoidAccessibilityAlteration")
138     private MethodVisitor visitMethodWithoutFrames(
139         final int access,
140         final String name,
141         final String descriptor,
142         final String signature,
143         final String... exceptions
144     ) {
145         final ClassVisitor delegate = this.getDelegate();
146         try {
147             final Field field = ClassWriter.class.getDeclaredField("compute");
148             field.setAccessible(true);
149             final int previous = field.getInt(delegate);
150             field.setInt(delegate, 4);
151             final MethodVisitor original = this.visitMethod(
152                 access, name, descriptor, signature, exceptions
153             );
154             field.setInt(delegate, previous);
155             return original;
156         } catch (final NoSuchFieldException | IllegalAccessException exception) {
157             throw new IllegalStateException(
158                 String.format(
159                     "Can't set compute field for ASM ClassWriter '%s' and change the computation mode to COMPUTE_ALL_FRAMES",
160                     delegate
161                 ),
162                 exception
163             );
164         }
165     }
166 
167     /**
168      * Which class writer to use.
169      * @param verify Verify bytecode.
170      * @return A verified class writer if verify is true, otherwise custom class writer.
171      */
172     private static ClassesAwareWriter prestructor(final boolean verify) {
173         final ClassesAwareWriter result;
174         if (verify) {
175             result = new VerifiedClassWriter();
176         } else {
177             result = new ClassesAwareWriter();
178         }
179         return result;
180     }
181 
182     /**
183      * Class writer that knows about additional classes loaded.
184      * This class works in couple with {@link PluginStartup#init()} ()} method that sets
185      * the maven classloader as the current thread classloader.
186      * Originally we faced with the problem that {@link ClassWriter} uses classes from ClassLoader
187      * to perform {@link MethodVisitor#visitMaxs(int, int)} method and if it can't
188      * find the class it throws {@link ClassNotFoundException}. To prevent this we override
189      * {@link ClassWriter#getClassLoader()} method and return the current thread classloader that
190      * knows about all classes that were compiled on the previous maven phases.
191      * You can read more about this problem here:
192      * - https://gitlab.ow2.org/asm/asm/-/issues/317918
193      * - https://stackoverflow.com/questions/11292701/error-while-instrumenting-class-files-asm-classwriter-getcommonsuperclass
194      *
195      * @since 0.1
196      */
197     private static class ClassesAwareWriter extends ClassWriter {
198 
199         /**
200          * Constructor.
201          * Do not compute frames automatically.
202          */
203         private ClassesAwareWriter() {
204             this(0);
205         }
206 
207         /**
208          * Constructor.
209          * @param flags Flags. See {@link ClassWriter#COMPUTE_FRAMES} for more information.
210          */
211         private ClassesAwareWriter(final int flags) {
212             super(flags);
213         }
214 
215         @Override
216         public final ClassLoader getClassLoader() {
217             return Thread.currentThread().getContextClassLoader();
218         }
219     }
220 
221     /**
222      * Class writer that verifies the bytecode.
223      * @since 0.2
224      */
225     private static final class VerifiedClassWriter extends ClassesAwareWriter {
226 
227         @Override
228         public byte[] toByteArray() {
229             final byte[] bytes = super.toByteArray();
230             VerifiedClassWriter.verify(bytes);
231             return bytes;
232         }
233 
234         /**
235          * Verify the bytecode.
236          * @param bytes The bytecode to verify.
237          */
238         private static void verify(final byte[] bytes) {
239             final ClassNode clazz = new ClassNode();
240             new ClassReader(bytes)
241                 .accept(new CheckClassAdapter(clazz, false), ClassReader.SKIP_DEBUG);
242             final Optional<Type> syper = Optional.ofNullable(clazz.superName)
243                 .map(Type::getObjectType);
244             final List<Type> interfaces = clazz.interfaces.stream().map(Type::getObjectType)
245                 .collect(Collectors.toList());
246             for (final MethodNode method : clazz.methods) {
247                 try {
248                     final SimpleVerifier verifier =
249                         new SimpleVerifier(
250                             Type.getObjectType(clazz.name),
251                             syper.orElse(null),
252                             interfaces,
253                             (clazz.access & Opcodes.ACC_INTERFACE) != 0
254                         );
255                     verifier.setClassLoader(Thread.currentThread().getContextClassLoader());
256                     new Analyzer<>(verifier).analyze(clazz.name, method);
257                 } catch (final ClassFormatError | AnalyzerException exception) {
258                     throw new IllegalStateException(
259                         String.format(
260                             "Bytecode verification failed for the class '%s' and method '%s'",
261                             clazz.name,
262                             method
263                         ),
264                         exception
265                     );
266                 }
267             }
268         }
269     }
