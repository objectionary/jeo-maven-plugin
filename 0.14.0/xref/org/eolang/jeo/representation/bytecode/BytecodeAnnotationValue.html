5   package org.eolang.jeo.representation.bytecode;
6   
7   import org.eolang.jeo.representation.directives.Format;
8   import org.objectweb.asm.AnnotationVisitor;
9   import org.xembly.Directive;
10  
11  /**
12   * Bytecode annotation value.
13   * All the instances of this class know how to write themselves to the given
14   * {@link AnnotationVisitor}. Since the annotation values are not always plain
15   * values, this interface is used to abstract the writing process.
16   *
17   * @since 0.3
18   */
19  public interface BytecodeAnnotationValue {
20  
21      /**
22       * Write the value to the given visitor.
23       * @param visitor Visitor.
24       */
25      void writeTo(AnnotationVisitor visitor);
26  
27      Iterable<Directive> directives(int index, Format format);
28  }
