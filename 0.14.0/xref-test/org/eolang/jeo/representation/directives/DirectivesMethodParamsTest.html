5   package org.eolang.jeo.representation.directives;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.xembly.Directives;
11  import org.xembly.Xembler;
12  
13  /**
14   * Test cases for {@link DirectivesMethodParams}.
15   * @since 0.1
16   */
17  final class DirectivesMethodParamsTest {
18  
19      @Test
20      void convertsToNonEmptyXmirStructureEvenIfParamsAreEmpty() {
21          MatcherAssert.assertThat(
22              "We expect the XML to contain the 'params' element even if no parameters are defined",
23              new Xembler(
24                  new Directives().add("o").append(new DirectivesMethodParams()).up()
25              ).xmlQuietly(),
26              XhtmlMatchers.hasXPath("/o/o[contains(@name, 'params')]")
27          );
28      }
29  }
