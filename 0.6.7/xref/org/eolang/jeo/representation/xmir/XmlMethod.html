48  /**
49   * XML method.
50   *
51   * @since 0.1
52   */
53  @SuppressWarnings({"PMD.AvoidDuplicateLiterals", "PMD.TooManyMethods"})
54  @ToString
55  @EqualsAndHashCode
56  public final class XmlMethod {
57  
58      /**
59       * Method node.
60       */
61      @ToString.Include
62      private final XmlNode node;
63  
64      /**
65       * Method labels.
66       */
67      @EqualsAndHashCode.Exclude
68      private final AllLabels labels;
69  
70      /**
71       * Constructor.
72       * @param xmlnode Method node.
73       */
74      public XmlMethod(final XmlNode xmlnode) {
75          this.node = xmlnode;
76          this.labels = new AllLabels();
77      }
78  
79      /**
80       * Constructor.
81       *
82       * @param name Method name.
83       * @param access Access modifiers.
84       * @param descriptor Method descriptor.
85       * @param exceptions Method exceptions.
86       * @checkstyle ParameterNumberCheck (5 lines)
87       */
88      XmlMethod(
89          final String name,
90          final int access,
91          final String descriptor,
92          final String... exceptions
93      ) {
94          this(XmlMethod.prestructor(name, access, descriptor, 0, 0, exceptions));
95      }
96  
97      /**
98       * Constructor.
99       *
100      * @param stack Max stack.
101      * @param locals Max locals.
102      */
103     XmlMethod(final int stack, final int locals) {
104         this(XmlMethod.prestructor("foo", Opcodes.ACC_PUBLIC, "()V", stack, locals));
105     }
106 
107     /**
108      * Convert to bytecode.
109      * @return Bytecode method.
110      */
111     public BytecodeMethod bytecode() {
112         try {
113             return new BytecodeMethod(
114                 this.trycatchEntries()
115                     .stream()
116                     .map(XmlTryCatchEntry::bytecode)
117                     .collect(Collectors.toList()),
118                 this.instructions()
119                     .stream()
120                     .map(XmlBytecodeEntry::bytecode)
121                     .collect(Collectors.toList()),
122                 this.annotations(),
123                 this.properties(),
124                 this.defvalue()
125                     .map(XmlDefaultValue::bytecode)
126                     .filter(Optional::isPresent)
127                     .map(Optional::get)
128                     .map(Collections::singletonList)
129                     .orElse(Collections.emptyList()),
130                 this.maxs().map(XmlMaxs::bytecode)
131                     .orElse(new BytecodeMaxs(0, 0))
132             );
133         } catch (final IllegalStateException exception) {
134             throw new ParsingException(
135                 String.format(
136                     "Unexpected exception during parsing the method '%s'",
137                     this.name()
138                 ),
139                 exception
140             );
141         } catch (final IllegalArgumentException exception) {
142             throw new ParsingException(
143                 String.format(
144                     "Can't transform method '%s' to bytecode",
145                     this.name()
146                 ),
147                 exception
148             );
149         } catch (final IndexOutOfBoundsException exception) {
150             throw new ParsingException(
151                 String.format(
152                     "Can't transform method '%s' to bytecode, because of index out of bounds, most probably due to broken XMIR",
153                     this.name()
154                 ),
155                 exception
156             );
157         }
158     }
159 
160     /**
161      * Method name.
162      * @return Name.
163      */
164     private String name() {
165         return new MethodName(
166             new Signature(
167                 this.node.attribute("name")
168                     .orElseThrow(
169                         () -> new IllegalStateException("Method 'name' attribute is not present")
170                     )
171             ).name()
172         ).bytecode();
173     }
174 
175     /**
176      * All instructions.
177      * @return Instructions.
178      */
179     private List<XmlBytecodeEntry> instructions() {
180         return this.node.child("name", "@")
181             .children()
182             .filter(element -> element.attribute("base").isPresent())
183             .map(XmlNode::toEntry)
184             .collect(Collectors.toList());
185     }
186 
187     /**
188      * Method max stack and locals.
189      *
190      * @return Maxs.
191      */
192     private Optional<XmlMaxs> maxs() {
193         return this.node.optchild("name", "maxs").map(XmlMaxs::new);
194     }
195 
196     /**
197      * Method properties as bytecode.
198      *
199      * @return Properties.
200      */
201     private BytecodeMethodProperties properties() {
202         try {
203             return new BytecodeMethodProperties(
204                 this.access(),
205                 this.name(),
206                 this.descriptor(),
207                 this.signature(),
208                 this.params(),
209                 this.exceptions()
210             );
211         } catch (final IllegalStateException exception) {
212             throw new ParsingException(
213                 String.format(
214                     "Unexpected exception during parsing the method '%s' properties",
215                     this.name()
216                 ),
217                 exception
218             );
219         }
220     }
221 
222     /**
223      * Method access modifiers.
224      *
225      * @return Access modifiers.
226      */
227     private int access() {
228         return new XmlValue(this.child(0)).integer();
229     }
230 
231     /**
232      * Method descriptor.
233      *
234      * @return Descriptor.
235      */
236     private String descriptor() {
237         return new XmlValue(this.child(1)).string();
238     }
239 
240     /**
241      * Method signature.
242      *
243      * @return Signature.
244      */
245     private String signature() {
246         return new XmlValue(this.child(2)).string();
247     }
248 
249     private XmlNode child(final int index) {
250         return this.node.children().collect(Collectors.toList()).get(index);
251     }
252 
253     /**
254      * Method trycatch entries.
255      *
256      * @return Trycatch entries.
257      */
258     private List<XmlTryCatchEntry> trycatchEntries() {
259         return this.node.children()
260             .filter(
261                 element -> element.attribute("name")
262                     .map(s -> s.contains("trycatchblocks"))
263                     .orElse(false))
264             .flatMap(XmlNode::children)
265             .map(entry -> new XmlTryCatchEntry(entry, this.labels))
266             .collect(Collectors.toList());
267     }
268 
269     /**
270      * Method annotations.
271      *
272      * @return Annotations.
273      */
274     private BytecodeAnnotations annotations() {
275         return this.node.children()
276             .filter(element -> element.hasAttribute("name", "annotations"))
277             .findFirst()
278             .map(XmlAnnotations::new)
279             .map(XmlAnnotations::bytecode)
280             .orElse(new BytecodeAnnotations());
281     }
282 
283     /**
284      * Annotation default value.
285      *
286      * @return Optional XMIR of the default value.
287      */
288     private Optional<XmlDefaultValue> defvalue() {
289         return this.node.optchild("base", new JeoFqn("annotation-default-value").fqn())
290             .map(XmlDefaultValue::new);
291     }
292 
293     /**
294      * Method parameters.
295      * @return Parameters.
296      */
297     private BytecodeMethodParameters params() {
298         return this.node.optchild("base", new JeoFqn("params").fqn())
299             .map(XmlParams::new).map(XmlParams::params)
300             .orElse(new BytecodeMethodParameters());
301     }
302 
303     /**
304      * Method exceptions.
305      *
306      * @return Exceptions.
307      */
308     private String[] exceptions() {
309         return this.child(3)
310             .children()
311             .map(XmlValue::new)
312             .map(XmlValue::string)
313             .toArray(String[]::new);
314     }
315 
316     /**
317      * Create Method XmlNode by directives.
318      *
319      * @param name Method name.
320      * @param access Method access modifiers.
321      * @param descriptor Method descriptor.
322      * @param stack Max stack.
323      * @param locals Max locals.
324      * @param exceptions Method exceptions.
325      * @return Method XmlNode.
326      * @checkstyle ParameterNumberCheck (5 lines)
327      */
328     private static XmlNode prestructor(
329         final String name,
330         final int access,
331         final String descriptor,
332         final int stack,
333         final int locals,
334         final String... exceptions
335     ) {
336         return new XmlNode(
337             new Xembler(
338                 new DirectivesMethod(
339                     name,
340                     new DirectivesMethodProperties(
341                         access,
342                         descriptor,
343                         "",
344                         exceptions,
345                         new DirectivesMaxs(stack, locals),
346                         new DirectivesMethodParams()
347                     )
348                 ),
349                 new Transformers.Node()
350             ).xmlQuietly()
351         );
352     }
