5   package org.eolang.jeo.representation.directives;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Test;
10  import org.xembly.Directive;
11  import org.xembly.Xembler;
12  
13  /**
14   * Test cases for {@link DirectivesWithoutComments}.
15   * @since 0.11.0
16   */
17  final class DirectivesWithoutCommentsTest {
18  
19      @Test
20      void removesComments() {
21          final DirectivesValue with = new DirectivesValue("data");
22          final Iterable<Directive> without = new DirectivesWithoutComments(with);
23          MatcherAssert.assertThat(
24              String.format("Plain directives should contain comments, but they do not %s", with),
25              new Xembler(with).xmlQuietly(),
26              Matchers.containsString("<!--")
27          );
28          MatcherAssert.assertThat(
29              "Directives without comments do not contain comments",
30              new Xembler(without).xmlQuietly(),
31              Matchers.not(Matchers.containsString("<!--"))
32          );
33      }
34  }
