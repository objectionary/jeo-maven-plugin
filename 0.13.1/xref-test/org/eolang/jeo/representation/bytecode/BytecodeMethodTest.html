335     /**
336      * Disassembles and assembles the given compiled class.
337      * @param compiled Compiled class as a path to the resource.
338      * @return Stream of methods.
339      * @checkstyle IllegalCatchCheck (25 lines)
340      */
341     @SuppressWarnings("PMD.AvoidCatchingGenericException")
342     static Stream<Arguments> disassembleAssemble(final String compiled) {
343         try {
344             return new XmlObject(
345                 new BytecodeRepresentation(
346                     new Bytecode(new BytesOf(new ResourceOf(compiled)).asBytes())
347                 ).toXmir()
348             ).bytecode()
349                 .top()
350                 .methods().stream()
351                 .map(method -> Arguments.of(method, method.name(), method.currentMaxs()));
352         } catch (final Exception ex) {
353             throw new IllegalStateException(
354                 String.format(
355                     "Can't disassemble and assemble the class %s",
356                     compiled
357                 ),
358                 ex
359             );
360         }
361     }

363     /**
364      * Provides methods for testing.
365      * @param clazz Resource class name.
366      * @return Stream of arguments.
367      */
368     static Stream<Arguments> methods(final String clazz) {
369         return new AsmProgram(
370             new JavaSourceClass(clazz).compile().bytes()
371         ).bytecode().top().methods().stream().map(
372             method -> Arguments.of(method, method.name(), method.currentMaxs())
373         );
