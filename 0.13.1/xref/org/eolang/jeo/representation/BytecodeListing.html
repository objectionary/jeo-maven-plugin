5   package org.eolang.jeo.representation;
6   
7   import java.io.PrintWriter;
8   import java.io.StringWriter;
9   import org.objectweb.asm.ClassReader;
10  import org.objectweb.asm.util.TraceClassVisitor;
11  
12  /**
13   * Pretty-printing utility for Java bytecode.
14   *
15   * <p>This class provides human-readable representation of Java bytecode using ASM's
16   * TraceClassVisitor. It converts raw bytecode into a textual format that shows
17   * the structure and instructions of a class file.</p>
18   * @since 0.6.0
19   * @todo #845:90min Decompile Bytecode Listing.
20   *  Currently we print bytecode listing to XMIR without decompiling it.
21   *  It's just raw, but human-readable bytecode.
22   *  We can make one step forward and try to decompile this code to Java.
23   *  This would be a great improvement for readability of this listing.
24   */
25  public final class BytecodeListing {
26  
27      /**
28       * Raw bytecode.
29       */
30      private final byte[] bytecode;
31  
32      /**
33       * Constructor.
34       * @param bytecode The raw bytecode array to format
35       */
36      BytecodeListing(final byte... bytecode) {
37          this.bytecode = bytecode.clone();
38      }
39  
40      @Override
41      public String toString() {
42          final ClassReader reader = new ClassReader(this.bytecode);
43          final StringWriter writer = new StringWriter();
44          reader.accept(new TraceClassVisitor(new PrintWriter(writer)), 0);
45          return writer.toString();
46      }
47  }
