7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.Arrays;
12  import java.util.Collection;
13  import java.util.List;
14  import java.util.Objects;
15  import java.util.Optional;
16  import java.util.function.Predicate;
17  import java.util.stream.Collectors;
18  import java.util.stream.Stream;
19  import org.objectweb.asm.ClassReader;
20  import org.objectweb.asm.Opcodes;
21  import org.objectweb.asm.Type;
22  import org.objectweb.asm.tree.ClassNode;
23  import org.objectweb.asm.tree.MethodNode;
24  import org.objectweb.asm.tree.analysis.Analyzer;
25  import org.objectweb.asm.tree.analysis.AnalyzerException;
26  import org.objectweb.asm.tree.analysis.SimpleVerifier;
27  import org.objectweb.asm.util.CheckClassAdapter;
28  
29  /**
30   * Bytecode classes verifier.
31   *
32   * <p>This class knows how to verify generated bytecode. It performs comprehensive
33   * verification of bytecode files using ASM's analysis framework. It requires all
34   * the classes to be loaded into the current classloader. See {@link PluginStartup#init()}
35   * for details on how all generated classes are loaded.</p>
36   * @since 0.6.0
37   */
38  final class BytecodeClasses {
39  
40      /**
41       * Input directory where all the generated class files are placed.
42       */
43      private final Path input;
44  
45      /**
46       * Several filters to apply on class files.
47       */
48      private final List<? extends Predicate<Path>> filters;
49  
50      /**
51       * Constructor.
52       * @param input Input directory where all the generated class files are placed.
53       * @param filters List of filters to apply on class files
54       */
55      @SafeVarargs
56      BytecodeClasses(final Path input, final Predicate<Path>... filters) {
57          this(input, Arrays.asList(filters));
58      }
59  
60      /**
61       * Constructor with filters.
62       * @param input Input directory where all the generated class files are placed.
63       * @param filters List of filters to apply on class files
64       */
65      private BytecodeClasses(final Path input, final List<Predicate<Path>> filters) {
66          this.input = input;
67          this.filters = filters;
68      }
69  
70      /**
71       * All the class files.
72       * @return Stream of paths to class files
73       */
74      Stream<Path> all() {
75          try {
76              return this.classes().stream();
77          } catch (final IOException exception) {
78              throw new IllegalStateException(
79                  String.format(
80                      "Can't read '%s' directory with classes",
81                      this.input
82                  ),
83                  exception
84              );
85          }
86      }
87  
88      /**
89       * Verify bytecode in the folder.
90       */
91      void verify() {
92          this.all().forEach(BytecodeClasses::verify);
93      }
94  
95      /**
96       * Find all bytecode files.
97       * @return Collection of bytecode files
98       * @throws IOException If some I/O problem arises
99       */
100     private Collection<Path> classes() throws IOException {
101         if (Objects.isNull(this.input)) {
102             throw new IllegalStateException(
103                 "The classes directory is not set, jeo-maven-plugin does not know where to look for classes."
104             );
105         }
106         if (!Files.exists(this.input)) {
107             throw new IllegalStateException(
108                 String.format(
109                     "The classes directory '%s' does not exist, jeo-maven-plugin does not know where to look for classes.",
110                     this.input
111                 )
112             );
113         }
114         try (Stream<Path> walk = Files.walk(this.input)) {
115             return walk
116                 .filter(Files::isRegularFile)
117                 .filter(path -> path.toString().endsWith(".class"))
118                 .filter(path -> this.filters.stream().allMatch(filter -> filter.test(path)))
119                 .collect(Collectors.toList());
120         }
121     }
122 
123     /**
124      * Read bytes of the class file.
125      * @param clazz Path to the class file to read
126      * @return Bytes of the class file
127      */
128     private static byte[] read(final Path clazz) {
129         try {
130             return Files.readAllBytes(clazz);
131         } catch (final IOException exception) {
132             throw new IllegalStateException(
133                 String.format("Can't read bytecode from the file '%s'", clazz),
134                 exception
135             );
136         }
137     }
138 
139     /**
140      * Verify the bytecode of the class file.
141      * @param clazz Path to the class file to verify
142      */
143     private static void verify(final Path clazz) {
144         final byte[] bytes = BytecodeClasses.read(clazz);
145         try {
146             BytecodeClasses.verify(bytes);
147         } catch (final IllegalArgumentException exception) {
148             throw new IllegalArgumentException(
149                 String.format("Invalid bytecode of class '%s'", clazz), exception
150             );
151         }
152     }
153 
154     /**
155      * Verify the generated bytecode.
156      * @param bytes Bytecode bytes to verify
157      */
158     private static void verify(final byte[] bytes) {
159         final ClassNode clazz = new ClassNode();
160         new ClassReader(bytes)
161             .accept(new CheckClassAdapter(clazz, false), ClassReader.SKIP_DEBUG);
162         final Optional<Type> syper = Optional.ofNullable(clazz.superName)
163             .map(Type::getObjectType);
164         final List<Type> interfaces = clazz.interfaces.stream().map(Type::getObjectType)
165             .collect(Collectors.toList());
166         for (final MethodNode method : clazz.methods) {
167             try {
168                 final SimpleVerifier verifier =
169                     new SimpleVerifier(
170                         Type.getObjectType(clazz.name),
171                         syper.orElse(null),
172                         interfaces,
173                         (clazz.access & Opcodes.ACC_INTERFACE) != 0
174                     );
175                 verifier.setClassLoader(Thread.currentThread().getContextClassLoader());
176                 new Analyzer<>(verifier).analyze(clazz.name, method);
177             } catch (final ClassFormatError | AnalyzerException exception) {
178                 throw new IllegalStateException(
179                     String.format(
180                         "Bytecode verification failed for the class '%s' and method '%s'",
181                         clazz.name,
182                         method.name
183                     ),
184                     exception
185                 );
186             }
187         }
188         Logger.info(
189             BytecodeClasses.class,
190             String.format("Bytecode verification passed for the class '%s'", clazz.name)
191         );
192     }
193 }
