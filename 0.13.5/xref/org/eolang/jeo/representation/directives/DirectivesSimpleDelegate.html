5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.xembly.Directive;
9   import org.xembly.Directives;
10  
11  /**
12   * Simple EO delegate object.
13   * <p>
14   *     Trivial object that represents a simple EO delegate object
15   *     with an attribute 'base' and a name '@'.
16   * </p>
17   * @since 0.12.0
18   */
19  public final class DirectivesSimpleDelegate implements Iterable<Directive> {
20  
21      /**
22       * Base of the object.
23       */
24      private final String base;
25  
26      /**
27       * Constructor.
28       * @param base The 'base' attribute of the abstract object.
29       */
30      public DirectivesSimpleDelegate(final String base) {
31          this.base = base;
32      }
33  
34      @Override
35      public Iterator<Directive> iterator() {
36          return new Directives().add("o")
37              .attr("base", this.base)
38              .attr("name", "@")
39              .up()
40              .iterator();
41      }
42  }
