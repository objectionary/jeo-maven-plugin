5   package benchmark;
6   
7   import java.io.File;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.Comparator;
12  import java.util.stream.Stream;
13  import org.eolang.jeo.Assembler;
14  import org.openjdk.jmh.annotations.Benchmark;
15  import org.openjdk.jmh.annotations.Fork;
16  import org.openjdk.jmh.annotations.Level;
17  import org.openjdk.jmh.annotations.Measurement;
18  import org.openjdk.jmh.annotations.Scope;
19  import org.openjdk.jmh.annotations.Setup;
20  import org.openjdk.jmh.annotations.State;
21  import org.openjdk.jmh.annotations.TearDown;
22  import org.openjdk.jmh.annotations.Warmup;
23  import org.openjdk.jmh.runner.Runner;
24  import org.openjdk.jmh.runner.RunnerException;
25  import org.openjdk.jmh.runner.options.OptionsBuilder;
26  
27  /**
28   * Assemble benchmark.
29   * @since 0.8
30   * @checkstyle DesignForExtensionCheck (500 lines)
31   */
32  @Fork(1)
33  @Warmup(iterations = 1, time = 2)
34  @Measurement(iterations = 1, time = 3)
35  @State(Scope.Benchmark)
36  @SuppressWarnings("PMD.JUnit4TestShouldUseAfterAnnotation")
37  public class AssembleBenchmark {
38  
39      /**
40       * Temporary directory.
41       */
42      private Path dir;
43  
44      /**
45       * Assembler.
46       */
47      private Assembler assembler;
48  
49      /**
50       * This method is used to run the benchmark from IDE.
51       * Don't remove it.
52       * @param args Arguments.
53       * @throws RunnerException If something goes wrong.
54       */
55      @SuppressWarnings("PMD.ProhibitPublicStaticMethods")
56      public static void main(final String[] args) throws RunnerException {
57          new Runner(
58              new OptionsBuilder()
59                  .include(AssembleBenchmark.class.getSimpleName())
60                  .build()
61          ).run();
62      }
63  
64      @Setup(Level.Trial)
65      public void init() throws IOException {
66          this.dir = Files.createTempDirectory("assemble");
67          final Path input = this.dir.resolve("input");
68          this.dir.toFile().deleteOnExit();
69          Files.createDirectories(input);
70          Files.write(input.resolve("test.xmir"), new Representation().disassemble());
71          this.assembler = new Assembler(input, this.dir);
72      }
73  
74      @Benchmark
75      public void assemble() {
76          this.assembler.assemble();
77      }
78  
79      @TearDown(Level.Trial)
80      public void tearDown() {
81          try (Stream<Path> files = Files.walk(this.dir).sorted(Comparator.reverseOrder())) {
82              files.map(Path::toFile).forEach(File::delete);
83          } catch (final IOException ioex) {
84              throw new IllegalStateException("Failed to delete assemble temporary directory", ioex);
85          }
86      }
87  }
