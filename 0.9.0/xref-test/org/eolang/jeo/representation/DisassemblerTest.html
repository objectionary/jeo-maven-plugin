5   package org.eolang.jeo.representation;
6   
7   import java.io.IOException;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import org.cactoos.bytes.BytesOf;
11  import org.cactoos.bytes.UncheckedBytes;
12  import org.cactoos.io.ResourceOf;
13  import org.eolang.jeo.Disassembler;
14  import org.hamcrest.MatcherAssert;
15  import org.hamcrest.Matchers;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.io.TempDir;
18  
19  /**
20   * Tests for {@link Disassembler}.
21   *
22   * @since 0.1.0
23   */
24  final class DisassemblerTest {
25  
26      @Test
27      void transpilesSuccessfully(@TempDir final Path temp) throws IOException {
28          final String name = "MethodByte.class";
29          Files.write(
30              temp.resolve(name),
31              new UncheckedBytes(new BytesOf(new ResourceOf(name))).asBytes()
32          );
33          new Disassembler(temp, temp).disassemble();
34          MatcherAssert.assertThat(
35              String.format("Can't find the transpiled file for the class '%s'.", name),
36              Files.exists(
37                  temp.resolve("org")
38                      .resolve("eolang")
39                      .resolve("jeo")
40                      .resolve("MethodByte.xmir")
41              ),
42              Matchers.is(true)
43          );
44      }
45  }
