5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Arrays;
8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.stream.Collectors;
11  import org.xembly.Directive;
12  import org.xembly.Directives;
13  
14  /**
15   * Directives that represent a pure EO object.
16   * Similar to {@link DirectivesJeoObject}, but for objects that are parts of the EO language.
17   * @since 0.6
18   */
19  public final class DirectivesEoObject implements Iterable<Directive> {
20  
21      /**
22       * The base of the object.
23       */
24      private final String base;
25  
26      /**
27       * The name of the object.
28       */
29      private final String name;
30  
31      /**
32       * Inner components.
33       */
34      private final List<Directives> inner;
35  
36      /**
37       * Constructor.
38       * @param base The base of the object.
39       * @param name The name of the object.
40       * @param inner Inner components.
41       */
42      @SafeVarargs
43      public DirectivesEoObject(
44          final String base, final String name, final Iterable<Directive>... inner
45      ) {
46          this(base, name, Arrays.stream(inner).map(Directives::new).collect(Collectors.toList()));
47      }
48  
49      /**
50       * Constructor.
51       * @param base The base of the object.
52       * @param name The name of the object.
53       * @param inner Inner components.
54       */
55      DirectivesEoObject(final String base, final String name, final List<Directives> inner) {
56          this.base = base;
57          this.name = name;
58          this.inner = inner;
59      }
60  
61      @Override
62      public Iterator<Directive> iterator() {
63          return new DirectivesClosedObject(
64              new EoFqn(this.base).fqn(),
65              "",
66              this.name,
67              this.inner.stream().reduce(new Directives(), Directives::append)
68          ).iterator();
69      }
70  }
