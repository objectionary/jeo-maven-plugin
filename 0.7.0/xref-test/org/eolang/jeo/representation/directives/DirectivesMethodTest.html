24  package org.eolang.jeo.representation.directives;
25  
26  import com.jcabi.matchers.XhtmlMatchers;
27  import org.eolang.jeo.representation.bytecode.BytecodeAnnotation;
28  import org.eolang.jeo.representation.bytecode.BytecodeAnnotations;
29  import org.eolang.jeo.representation.bytecode.BytecodeMaxs;
30  import org.eolang.jeo.representation.bytecode.BytecodeMethod;
31  import org.eolang.jeo.representation.bytecode.BytecodeMethodProperties;
32  import org.eolang.jeo.representation.xmir.NativeXmlNode;
33  import org.eolang.jeo.representation.xmir.XmlMethod;
34  import org.hamcrest.MatcherAssert;
35  import org.hamcrest.Matchers;
36  import org.junit.jupiter.api.Test;
37  import org.xembly.ImpossibleModificationException;
38  import org.xembly.Xembler;
39  
40  /**
41   * Test case for {@link DirectivesMethod}.
42   * @since 0.5
43   */
44  final class DirectivesMethodTest {
45  
46      @Test
47      void transformsToXmir() throws ImpossibleModificationException {
48          final String name = "Hello";
49          final int access = 100;
50          final String descriptor = "()I";
51          final String signature = "";
52          MatcherAssert.assertThat(
53              "We expect that directives will generate correct method",
54              new XmlMethod(
55                  new NativeXmlNode(
56                      new Xembler(
57                          new DirectivesMethod(
58                              name,
59                              new DirectivesMethodProperties(access, descriptor, signature)
60                          )
61                      ).xml()
62                  )
63              ).bytecode(),
64              Matchers.equalTo(
65                  new BytecodeMethod(
66                      new BytecodeMethodProperties(name, descriptor, signature, access),
67                      new BytecodeMaxs()
68                  )
69              )
70          );
71      }
72  
73      @Test
74      void transformsAnnotationsToXmir() throws ImpossibleModificationException {
75          final String descriptor = "Consumer";
76          final boolean visible = true;
77          final String name = "foo";
78          MatcherAssert.assertThat(
79              "We expect that annotation dirictes will be added to the method",
80              new XmlMethod(
81                  new NativeXmlNode(
82                      new Xembler(
83                          new DirectivesMethod(name)
84                              .withAnnotation(new DirectivesAnnotation(descriptor, visible))
85                      ).xml()
86                  )
87              ).bytecode(),
88              Matchers.equalTo(
89                  new BytecodeMethod(
90                      name,
91                      new BytecodeAnnotations(
92                          new BytecodeAnnotation(descriptor, visible)
93                      )
94                  )
95              )
96          );
97      }
98  
99      @Test
100     void addsPrefixToTheMethodName() throws ImpossibleModificationException {
101         MatcherAssert.assertThat(
102             "We expect that 'j$' prefix will be added to the method name",
103             new Xembler(new BytecodeMethod("φTerm").directives()).xml(),
104             XhtmlMatchers.hasXPaths("./o[contains(@name, 'j$φTerm')]")
105         );
106     }
107 }
