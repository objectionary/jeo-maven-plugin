5   package org.eolang.jeo;
6   
7   import java.nio.file.Path;
8   import java.nio.file.Paths;
9   import java.util.Set;
10  import java.util.stream.Stream;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.junit.jupiter.api.DisplayName;
14  import org.junit.jupiter.params.ParameterizedTest;
15  import org.junit.jupiter.params.provider.Arguments;
16  import org.junit.jupiter.params.provider.MethodSource;
17  
18  /**
19   * Test for {@link GlobFilter}.
20   * @since 0.13.0
21   */
22  final class GlobFilterTest {
23  
24      /**
25       * Applies the glob filter to a path and checks if it matches the expected result.
26       * @param description Description of the test case
27       * @param includes Set of glob patterns to include
28       * @param excludes Set of glob patterns to exclude
29       * @param path Path to test against the glob patterns
30       * @param expected Expected result of the filter test
31       * @checkstyle ParameterNumberCheck (20 lines)
32       */
33      @ParameterizedTest(name = "{0}")
34      @MethodSource("cases")
35      @DisplayName("Test GlobFilter behavior")
36      void appliesGlobFilter(
37          final String description,
38          final Set<String> includes,
39          final Set<String> excludes,
40          final Path path,
41          final boolean expected
42      ) {
43          MatcherAssert.assertThat(
44              "We expect the filter to match the path correctly",
45              new GlobFilter(includes, excludes).test(path),
46              Matchers.is(expected)
47          );
48      }
49  
50      /**
51       * Test cases for GlobFilter.
52       * @return Stream of test cases
53       */
54      static Stream<Arguments> cases() {
55          return Stream.of(
56              Arguments.of(
57                  "Included by pattern",
58                  GlobFilterTest.setOf("**/*.java"),
59                  GlobFilterTest.setOf(),
60                  Paths.get("src/Included.java"),
61                  true
62              ),
63              Arguments.of(
64                  "Excluded takes precedence",
65                  GlobFilterTest.setOf("**/*.java"),
66                  GlobFilterTest.setOf("src/**"),
67                  Paths.get("src/Excluded.java"),
68                  false
69              ),
70              Arguments.of(
71                  "Included, not excluded",
72                  GlobFilterTest.setOf("src/**/*.java"),
73                  GlobFilterTest.setOf("**/test/**"),
74                  Paths.get("src/main/Main.java"),
75                  true
76              ),
77              Arguments.of(
78                  "Only excluded",
79                  GlobFilterTest.setOf(),
80                  GlobFilterTest.setOf("**/*.class"),
81                  Paths.get("target/classes/Main.class"),
82                  false
83              ),
84              Arguments.of(
85                  "Not matched by includes",
86                  GlobFilterTest.setOf("**/*.java"),
87                  GlobFilterTest.setOf(),
88                  Paths.get("README.md"),
89                  false
90              ),
91              Arguments.of(
92                  "No includes: allow all except excluded",
93                  GlobFilterTest.setOf(),
94                  GlobFilterTest.setOf("**/*.tmp"),
95                  Paths.get("build/output.log"),
96                  true
97              ),
98              Arguments.of(
99                  "No includes and excluded match",
100                 GlobFilterTest.setOf(),
101                 GlobFilterTest.setOf("**/*.tmp"),
102                 Paths.get("build/output.tmp"),
103                 false
104             ),
105             Arguments.of(
106                 "Empty includes and excludes",
107                 GlobFilterTest.setOf(),
108                 GlobFilterTest.setOf(),
109                 Paths.get("any/path/file.txt"),
110                 true
111             )
112         );
113     }
114 
115     /**
116      * Set of strings for test cases.
117      * @param values Values to include in the set
118      * @return Set of strings
119      */
120     private static Set<String> setOf(final String... values) {
121         return Stream.of(values).collect(java.util.stream.Collectors.toSet());
122     }
123 }
