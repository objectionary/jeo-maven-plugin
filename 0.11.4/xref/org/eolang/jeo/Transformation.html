5   package org.eolang.jeo;
6   
7   import java.nio.file.Path;
8   
9   /**
10   * Transformation interface.
11   *
12   * <p>This interface defines the contract for transformations that convert
13   * files from one format to another. Implementations handle specific transformation
14   * types like assembling XMIR to bytecode or disassembling bytecode to XMIR.</p>
15   * @since 0.6.0
16   */
17  public interface Transformation {
18  
19      /**
20       * The path to the file to be transformed.
21       * @return Path to the source file
22       */
23      Path source();
24  
25      /**
26       * The path to the transformed file.
27       * @return Path to the target file after transformation
28       */
29      Path target();
30  
31      /**
32       * Transform the file.
33       * @return Transformed file content as byte array
34       */
35      byte[] transform();
36  }
