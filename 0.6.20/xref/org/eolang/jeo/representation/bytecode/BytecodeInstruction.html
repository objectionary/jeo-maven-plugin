86      @Override
87      public void writeTo(final MethodVisitor visitor, final AsmLabels labels) {
88          Instruction.find(this.opcode)
89              .generate(
90                  visitor,
91                  this.args.stream()
92                      .map(
93                          arg -> {
94                              final Object result;
95                              if (arg instanceof BytecodeLabel) {
96                                  result = labels.label((BytecodeLabel) arg);
97                              } else {
98                                  result = arg;
99                              }
100                             return result;
101                         }
102                     ).collect(Collectors.toList())
103             );
104     }

108         return new DirectivesInstruction(this.opcode, counting, this.args.toArray());
109     }

122      * Impact of each instruction on the stack.
123      * @return Stack impact.
124      * @checkstyle CyclomaticComplexityCheck (350 lines)
125      * @checkstyle MethodLengthCheck (350 lines)
126      * @checkstyle JavaNCSSCheck (350 lines)
127      * @checkstyle AvoidNestedBlocksCheck (350 lines)
128      */
129     @SuppressWarnings({"PMD.NcssCount", "PMD.ExcessiveMethodLength"})
130     public int impact() {
131         final int result;
132         final Instruction instruction = Instruction.find(this.opcode);
133         switch (instruction) {
134             case LASTORE:
135             case DASTORE:
136                 result = -4;
137                 break;
138             case IASTORE:
139             case FASTORE:
140             case AASTORE:
141             case BASTORE:
142             case CASTORE:
143             case SASTORE:
144             case LCMP:
145             case DCMPL:
146             case DCMPG:
147                 result = -3;
148                 break;
149             case LSTORE:
150             case DSTORE:
151             case POP2:
152             case LADD:
153             case LSUB:
154             case LMUL:
155             case LDIV:
156             case LREM:
157             case DADD:
158             case DSUB:
159             case DMUL:
160             case DDIV:
161             case DREM:
162             case IF_ICMPEQ:
163             case IF_ICMPNE:
164             case IF_ICMPLT:
165             case IF_ICMPGE:
166             case IF_ICMPGT:
167             case IF_ICMPLE:
168             case IF_ACMPEQ:
169             case IF_ACMPNE:
170             case LOR:
171             case LAND:
172             case LXOR:
173             case LRETURN:
174             case DRETURN:
175                 result = -2;
176                 break;
177             case IALOAD:
178             case FALOAD:
179             case AALOAD:
180             case BALOAD:
181             case CALOAD:
182             case SALOAD:
183             case IADD:
184             case ISUB:
185             case IMUL:
186             case IDIV:
187             case IREM:
188             case FADD:
189             case FSUB:
190             case FMUL:
191             case FDIV:
192             case FREM:
193             case LSHL:
194             case LSHR:
195             case LUSHR:
196             case POP:
197             case FCMPL:
198             case FCMPG:
199             case L2I:
200             case L2F:
201             case D2I:
202             case D2F:
203             case ISTORE:
204             case FSTORE:
205             case ASTORE:
206             case IFEQ:
207             case IFNE:
208             case IFLT:
209             case IFGE:
210             case IFGT:
211             case IFLE:
212             case IFNULL:
213             case IFNONNULL:
214             case IOR:
215             case IAND:
216             case IXOR:
217             case ISHL:
218             case ISHR:
219             case IUSHR:
220             case IRETURN:
221             case FRETURN:
222             case ARETURN:
223             case MONITORENTER:
224             case MONITOREXIT:
225             case TABLESWITCH:
226             case LOOKUPSWITCH:
227             case ATHROW:
228                 result = -1;
229                 break;
230             case NOP:
231             case SWAP:
232             case I2F:
233             case F2I:
234             case I2B:
235             case I2C:
236             case I2S:
237             case L2D:
238             case D2L:
239             case LALOAD:
240             case DALOAD:
241             case DNEG:
242             case FNEG:
243             case LNEG:
244             case INEG:
245             case GOTO:
246             case JSR:
247             case RET:
248             case RETURN:
249             case IINC:
250             case NEWARRAY:
251             case ANEWARRAY:
252             case ARRAYLENGTH:
253             case CHECKCAST:
254             case INSTANCEOF:
255                 result = 0;
256                 break;
257             case ACONST_NULL:
258             case ICONST_M1:
259             case ICONST_0:
260             case ICONST_1:
261             case ICONST_2:
262             case ICONST_3:
263             case ICONST_4:
264             case ICONST_5:
265             case FCONST_0:
266             case FCONST_1:
267             case FCONST_2:
268             case BIPUSH:
269             case SIPUSH:
270             case ILOAD:
271             case FLOAD:
272             case ALOAD:
273             case DUP:
274             case DUP_X1:
275             case DUP_X2:
276             case I2L:
277             case I2D:
278             case F2L:
279             case F2D:
280             case NEW:
281                 result = 1;
282                 break;
283             case LCONST_0:
284             case LCONST_1:
285             case DCONST_0:
286             case DCONST_1:
287             case LLOAD:
288             case DLOAD:
289             case DUP2:
290             case DUP2_X1:
291             case DUP2_X2:
292                 result = 2;
293                 break;
294             case LDC: {
295                 final Class<?> clazz = this.args.get(0).getClass();
296                 if (clazz == Long.class || clazz == Double.class) {
297                     result = 2;
298                     break;
299                 } else {
300                     result = BytecodeInstruction.size(Type.getType(clazz));
301                     break;
302                 }
303             }
304             case GETSTATIC:
305                 result = BytecodeInstruction.size(Type.getType(String.valueOf(this.args.get(2))));
306                 break;
307             case PUTSTATIC:
308                 result = BytecodeInstruction.size(
309                     Type.getType(String.valueOf(this.args.get(2)))
310                 ) * -1;
311                 break;
312             case GETFIELD:
313                 result = BytecodeInstruction.size(
314                     Type.getType(String.valueOf(this.args.get(2)))
315                 ) - 1;
316                 break;
317             case PUTFIELD:
318                 result = BytecodeInstruction.size(
319                     Type.getType(String.valueOf(this.args.get(2)))
320                 ) * -1 - 1;
321                 break;
322             case INVOKEVIRTUAL:
323             case INVOKESPECIAL:
324             case INVOKEINTERFACE:
325                 result = BytecodeInstruction.methodImpact(String.valueOf(this.args.get(2))) - 1;
326                 break;
327             case INVOKESTATIC:
328                 result = BytecodeInstruction.methodImpact(String.valueOf(this.args.get(2)));
329                 break;
330             case INVOKEDYNAMIC:
331                 result = BytecodeInstruction.methodImpact(String.valueOf(this.args.get(1)));
332                 break;
333             case MULTIANEWARRAY:
334                 result = -(int) (this.args.get(1)) + 1;
335                 break;
336             default:
337                 throw new UnsupportedOperationException(
338                     String.format(
339                         "Unsupported opcode: %s", new OpcodeName(this.opcode).simplified()
340                     )
341                 );
342         }
343         return result;
344     }

390         final boolean result;
391         switch (Instruction.find(this.opcode)) {
392             case IFEQ:
393             case IFNE:
394             case IFLT:
395             case IFGE:
396             case IFGT:
397             case IFLE:
398             case IF_ICMPEQ:
399             case IF_ICMPNE:
400             case IF_ICMPLT:
401             case IF_ICMPGE:
402             case IF_ICMPGT:
403             case IF_ICMPLE:
404             case IF_ACMPEQ:
405             case IF_ACMPNE:
406             case IFNULL:
407             case IFNONNULL:
408                 result = true;
409                 break;
410             default:
411                 result = false;
412                 break;
413         }
414         return result;
415     }

459     /**
460      * Jump to a label.
461      * Where to jump.
462      * @return Jump label.
463      * @checkstyle CyclomaticComplexityCheck (100 lines)
464      */
465     public List<BytecodeLabel> jumps() {
466         final List<BytecodeLabel> result;
467         switch (Instruction.find(this.opcode)) {
468             case GOTO:
469             case JSR:
470             case IFEQ:
471             case IFNE:
472             case IFLT:
473             case IFGE:
474             case IFGT:
475             case IFLE:
476             case IF_ICMPEQ:
477             case IF_ICMPNE:
478             case IF_ICMPLT:
479             case IF_ICMPGE:
480             case IF_ICMPGT:
481             case IF_ICMPLE:
482             case IF_ACMPEQ:
483             case IF_ACMPNE:
484             case IFNULL:
485             case IFNONNULL:
486                 result = Collections.singletonList((BytecodeLabel) this.args.get(0));
487                 break;
488             case TABLESWITCH:
489             case LOOKUPSWITCH:
490                 result = this.args.stream()
491                     .filter(BytecodeLabel.class::isInstance)
492                     .map(BytecodeLabel.class::cast)
493                     .collect(Collectors.toList());
494                 break;
495             default:
496                 throw new IllegalStateException(
497                     String.format(
498                         "Instruction %s is not a branch instruction",
499                         new OpcodeName(this.opcode).simplified()
500                     )
501                 );
502         }
503         return result;
504     }

515     /**
516      * Is this instruction a variable instruction?
517      * @throws IllegalStateException If it is not.
518      */
519     private void assertVarInstruction() {
520         if (!this.isVarInstruction()) {
521             throw new IllegalStateException(
522                 String.format(
523                     "Instruction %s is not a variable instruction",
524                     new OpcodeName(this.opcode).simplified()
525                 )
526             );
527         }
528     }

530     /**
531      * Size of the type.
532      * @param type Type.
533      * @return Size.
534      */
535     private static int size(final Type type) {
536         final int result;
537         if (Objects.equals(type, Type.DOUBLE_TYPE) || Objects.equals(type, Type.LONG_TYPE)) {
538             result = 2;
539         } else if (Objects.equals(type, Type.VOID_TYPE)) {
540             result = 0;
541         } else {
542             result = 1;
543         }
544         return result;
545     }

547     /**
548      * Impact of the method invocation on stack.
549      * @param descriptor Method descriptor.
550      * @return Impact.
551      */
552     private static int methodImpact(final String descriptor) {
553         return BytecodeInstruction.size(Type.getReturnType(descriptor))
554             - Arrays.stream(Type.getArgumentTypes(descriptor))
555             .mapToInt(BytecodeInstruction::size)
556             .sum();
557     }

560         return Instruction.find(this.opcode) == Instruction.ATHROW;

1884         /**
1885          * Check if the instruction is a variable instruction.
1886          * @return True if the instruction is a variable instruction.
1887          * @checkstyle CyclomaticComplexityCheck (50 lines)
1888          */
1889         boolean isVarInstruction() {
1890             final boolean result;
1891             switch (this) {
1892                 case ILOAD:
1893                 case LLOAD:
1894                 case FLOAD:
1895                 case DLOAD:
1896                 case ALOAD:
1897                 case ISTORE:
1898                 case LSTORE:
1899                 case FSTORE:
1900                 case DSTORE:
1901                 case ASTORE:
1902                 case RET:
1903                     result = true;
1904                     break;
1905                 default:
1906                     result = false;
1907                     break;
1908             }
1909             return result;
1910         }
1911 
1912         /**
1913          * Local variable size.
1914          * @return Local variable size.
1915          * @checkstyle CyclomaticComplexityCheck (50 lines)
1916          */
1917         int size() {
1918             final int res;
1919             switch (this) {
1920                 case ILOAD:
1921                 case FLOAD:
1922                 case ALOAD:
1923                 case ISTORE:
1924                 case FSTORE:
1925                 case ASTORE:
1926                     res = 1;
1927                     break;
1928                 case LLOAD:
1929                 case LSTORE:
1930                 case DLOAD:
1931                 case DSTORE:
1932                     res = 2;
1933                     break;
1934                 case RET:
1935                     res = 0;
1936                     break;
1937                 default:
1938                     throw new IllegalStateException(
1939                         String.format(
1940                             "Instruction %s is not a variable instruction",
1941                             this.name()
1942                         )
1943                     );
1944             }
1945             return res;
1946         }

1948         /**
1949          * Get instruction by opcode.
1950          *
1951          * @param opcode Opcode.
1952          * @return Instruction.
1953          */
1954         static Instruction find(final int opcode) {
1955             for (final Instruction instruction : Instruction.values()) {
1956                 if (instruction.opcode == opcode) {
1957                     return instruction;
1958                 }
1959             }
1960             throw new UnrecognizedOpcode(opcode);
1961         }
