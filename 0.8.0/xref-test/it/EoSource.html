5   package it;
6   
7   import com.jcabi.xml.XML;
8   import java.io.IOException;
9   import org.cactoos.io.ResourceOf;
10  import org.eolang.parser.EoSyntax;
11  
12  /**
13   * EO source.
14   * @since 0.1
15   */
16  @SuppressWarnings("JTCOP.RuleCorrectTestName")
17  final class EoSource {
18  
19      /**
20       * Resource of EO program.
21       */
22      private final String resource;
23  
24      /**
25       * Constructor.
26       * @param resource Resource of EO program.
27       */
28      EoSource(final String resource) {
29          this.resource = resource;
30      }
31  
32      /**
33       * Parse the EO program into XMIR.
34       * @return XMIR.
35       */
36      XML parse() {
37          try {
38              return new EoSyntax("scenario", new ResourceOf(this.resource)).parsed();
39          } catch (final IOException exception) {
40              throw new IllegalStateException(
41                  String.format("Can't parse '%s'", this.resource),
42                  exception
43              );
44          }
45      }
46  }
