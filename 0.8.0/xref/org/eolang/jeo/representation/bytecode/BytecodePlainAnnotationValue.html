5   package org.eolang.jeo.representation.bytecode;
6   
7   import lombok.EqualsAndHashCode;
8   import lombok.ToString;
9   import org.eolang.jeo.representation.directives.DirectivesPlainAnnotationValue;
10  import org.objectweb.asm.AnnotationVisitor;
11  import org.xembly.Directive;
12  
13  /**
14   * An annotation value that is a plain value.
15   * @since 0.6
16   */
17  @ToString
18  @EqualsAndHashCode
19  public final class BytecodePlainAnnotationValue implements BytecodeAnnotationValue {
20  
21      /**
22       * The name of the annotation property.
23       */
24      private final String name;
25  
26      /**
27       * The actual value.
28       */
29      private final Object value;
30  
31      /**
32       * Constructor.
33       * @param name The name of the annotation property.
34       * @param value The actual value.
35       */
36      public BytecodePlainAnnotationValue(final String name, final Object value) {
37          this.name = name;
38          this.value = value;
39      }
40  
41      @Override
42      public void writeTo(final AnnotationVisitor visitor) {
43          visitor.visit(this.name, this.value);
44      }
45  
46      @Override
47      public Iterable<Directive> directives() {
48          return new DirectivesPlainAnnotationValue(this.name, this.value);
49      }
50  }
