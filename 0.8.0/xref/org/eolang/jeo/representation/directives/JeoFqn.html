5   package org.eolang.jeo.representation.directives;
6   
7   /**
8    * Jeo fully qualified name.
9    * Adds a jeo-maven-plugin package name to an object base.
10   * For example,
11   * - `opcode` -> `org.eolang.jeo.opcode`
12   * - `representation` -> `org.eolang.jeo.representation`
13   * - `int` -> `org.eolang.jeo.int`
14   * @since 0.6
15   * @todo #710:90min Refactor the usage of {@link JeoFqn} in the project.
16   *  The solution with FQN transformation is not the best one.
17   *  We create this object many times and use it in many places.
18   *  All the usages are spread across the project.
19   *  We need to refactor the project to use the FQN in a more centralized way.
20   *  The same problem is related to {@link EoFqn}.
21   */
22  public final class JeoFqn {
23  
24      /**
25       * Base name.
26       */
27      private final String base;
28  
29      /**
30       * Constructor.
31       * @param base Base name.
32       */
33      public JeoFqn(final String base) {
34          this.base = base;
35      }
36  
37      /**
38       * Get a fully qualified name.
39       * @return Fully qualified name.
40       */
41      public String fqn() {
42          return String.format("Q.jeo.%s", this.base);
43      }
44  
45      @Override
46      public String toString() {
47          return this.fqn();
48      }
49  
50  }
