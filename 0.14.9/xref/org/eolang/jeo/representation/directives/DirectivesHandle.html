5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.objectweb.asm.Handle;
9   import org.xembly.Directive;
10  
11  /**
12   * Directives Handle.
13   * This is the XMIR representation of the Java ASM Handle object.
14   * @since 0.1
15   */
16  public final class DirectivesHandle implements Iterable<Directive> {
17  
18      /**
19       * The bytecode instruction index.
20       */
21      private final int index;
22  
23      /**
24       * Format of the directives.
25       */
26      private final Format format;
27  
28      /**
29       * ASM Handle object.
30       */
31      private final Handle handle;
32  
33      /**
34       * Constructor.
35       * @param index The bytecode instruction index.
36       * @param format Format of the directives.
37       * @param handle ASM Handle object.
38       */
39      public DirectivesHandle(final int index, final Format format, final Handle handle) {
40          this.index = index;
41          this.format = format;
42          this.handle = handle;
43      }
44  
45      @Override
46      public Iterator<Directive> iterator() {
47          return new DirectivesJeoObject(
48              "handle",
49              new NumName("h", this.index).toString(),
50              new DirectivesValue(0, this.format, this.handle.getTag()),
51              new DirectivesValue(1, this.format, this.handle.getOwner()),
52              new DirectivesValue(2, this.format, this.handle.getName()),
53              new DirectivesValue(3, this.format, this.handle.getDesc()),
54              new DirectivesValue(4, this.format, this.handle.isInterface())
55          ).iterator();
56      }
57  }
