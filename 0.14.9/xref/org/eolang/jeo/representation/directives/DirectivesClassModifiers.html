5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.Iterator;
8   import org.objectweb.asm.Opcodes;
9   import org.xembly.Directive;
10  
11  /**
12   * Directives for class modifiers.
13   * Similar to {@link DirectivesMethodModifiers} but for class-level modifiers.
14   * @since 0.15.0
15   */
16  public final class DirectivesClassModifiers implements Iterable<Directive> {
17  
18      /**
19       * Format of the directives.
20       */
21      private final Format format;
22  
23      /**
24       * Access modifiers.
25       */
26      private final int modifiers;
27  
28      /**
29       * Constructor.
30       * @param format Format of the directives.
31       * @param modifiers Access modifiers.
32       */
33      DirectivesClassModifiers(final Format format, final int modifiers) {
34          this.format = format;
35          this.modifiers = modifiers;
36      }
37  
38      @Override
39      public Iterator<Directive> iterator() {
40          final Modifiers all = new Modifiers(this.modifiers);
41          return new DirectivesJeoObject(
42              "modifiers",
43              "modifiers",
44              new DirectivesValue(this.format, "public", all.is(Opcodes.ACC_PUBLIC)),
45              new DirectivesValue(this.format, "private", all.is(Opcodes.ACC_PRIVATE)),
46              new DirectivesValue(this.format, "protected", all.is(Opcodes.ACC_PROTECTED)),
47              new DirectivesValue(this.format, "final", all.is(Opcodes.ACC_FINAL)),
48              new DirectivesValue(this.format, "super", all.is(Opcodes.ACC_SUPER)),
49              new DirectivesValue(this.format, "interface", all.is(Opcodes.ACC_INTERFACE)),
50              new DirectivesValue(this.format, "abstract", all.is(Opcodes.ACC_ABSTRACT)),
51              new DirectivesValue(this.format, "synthetic", all.is(Opcodes.ACC_SYNTHETIC)),
52              new DirectivesValue(this.format, "annotation", all.is(Opcodes.ACC_ANNOTATION)),
53              new DirectivesValue(this.format, "enum", all.is(Opcodes.ACC_ENUM)),
54              new DirectivesValue(this.format, "mandated", all.is(Opcodes.ACC_MANDATED)),
55              new DirectivesValue(this.format, "module", all.is(Opcodes.ACC_MODULE)),
56              new DirectivesValue(this.format, "record", all.is(Opcodes.ACC_RECORD)),
57              new DirectivesValue(this.format, "deprecated", all.is(Opcodes.ACC_DEPRECATED))
58          ).iterator();
59      }
60  }
