23  /**
24   * Set of integration tests for Java source parsing and transformation.
25   * The default pipeline for all the tests is:
26   * - Generate random Java source code
27   * - Compile it into bytecode
28   * - Transform bytecode into XMIR
29   * - Transform XMIR back into bytecode
30   * - Compare the original bytecode with the transformed one.
31   * In this class we compare bytecode as strings because the order of values in
32   * the bytecode constant pool might differ, however, the bytecode is still the same.
33   * Moreover, we don't care about line numbers and other supplementary information.
34   * We ignore it by using `-g:vars` compiler option.
35   * @since 0.1
36   * @todo #1130:90min Enable {@link #transformsRandomJavaSourceCodeIntoEoAndBack(Path)} test.
37   *  This test is disabled because it's flaky and sometimes fails on some machines, like
38   *  MacOS with JDK 17. We should investigate the reason of the failure and enable it.
39   */
40  final class JavaSourceCompilationIT {
41  
42      @Test
43      @Disabled
44      @EnabledIf(value = "hasJavaCompiler", disabledReason = "Java compiler is not available")
45      void transformsRandomJavaSourceCodeIntoEoAndBack(@TempDir final Path temp) throws IOException {
46          final Bytecode expected = JavaSourceCompilationIT.compile(temp);
47          MatcherAssert.assertThat(
48              "Bytecode is not equal to the original one, check that transformation is correct and does not change the bytecode",
49              new XmirRepresentation(new BytecodeRepresentation(expected).toXmir())
50                  .toBytecode()
51                  .toString(),
52              Matchers.equalTo(expected.toString())
53          );
54      }
55  
56      /**
57       * Compile random java class into bytecode.
58       * @param where Where to compile.
59       * @return Bytecode.
60       */
61      private static Bytecode compile(final Path where) throws IOException {
62          final Path java = where.resolve("HelloWorld.java");
63          final RandomJavaClass rand = new RandomJavaClass();
64          final String src = rand.src();
65          Files.write(
66              java,
67              src.getBytes(StandardCharsets.UTF_8)
68          );
69          ToolProvider.getSystemJavaCompiler().run(
70              System.in,
71              System.out,
72              System.err,
73              "-g:vars",
74              "-source", "11",
75              "-parameters",
76              java.toString()
77          );
78          return new Bytecode(
79              Files.readAllBytes(
80                  Files.find(where, 1, (path, attr) -> path.toString().endsWith(".class"))
81                      .findFirst()
82                      .orElseThrow(() -> new IllegalStateException("No class file found"))
83              )
84          );
85      }
86  
87      /**
88       * Check if java compiler is available.
89       * Don't care about PMD warning: this method is used in @EnabledIf annotation.
90       * @return True if java compiler is available.
91       */
92      @SuppressWarnings("PMD.UnusedPrivateMethod")
93      private static boolean hasJavaCompiler() {
94          return ToolProvider.getSystemJavaCompiler() != null;
95      }
96  }
