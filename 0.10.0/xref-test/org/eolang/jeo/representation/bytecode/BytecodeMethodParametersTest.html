5   package org.eolang.jeo.representation.bytecode;
6   
7   import com.jcabi.matchers.XhtmlMatchers;
8   import org.hamcrest.MatcherAssert;
9   import org.junit.jupiter.api.Test;
10  import org.objectweb.asm.Type;
11  import org.xembly.ImpossibleModificationException;
12  import org.xembly.Xembler;
13  
14  /**
15   * Bytecode method parameters.
16   * @since 0.6
17   */
18  final class BytecodeMethodParametersTest {
19  
20      @Test
21      void convertsToDirectivesWithTwoParams() throws ImpossibleModificationException {
22          MatcherAssert.assertThat(
23              "Can't convert bytecode method parameters to correct directives",
24              new Xembler(
25                  new BytecodeMethodParameters(
26                      new BytecodeMethodParameter(0, Type.INT_TYPE),
27                      new BytecodeMethodParameter(1, Type.INT_TYPE)
28                  ).directives()
29              ).xml(),
30              XhtmlMatchers.hasXPaths(
31                  "/o[contains(@base,'params')]",
32                  "/o[contains(@base,'params')]/o[contains(@base,'param') and contains(@as,'arg0')]",
33                  "/o[contains(@base,'params')]/o[contains(@base,'param') and contains(@as,'arg1')]"
34              )
35          );
36      }
37  }
