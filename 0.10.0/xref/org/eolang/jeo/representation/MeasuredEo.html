5   package org.eolang.jeo.representation;
6   
7   import com.jcabi.xml.XML;
8   import com.jcabi.xml.XMLDocument;
9   import org.eolang.jeo.representation.directives.DirectivesObject;
10  import org.xembly.Directives;
11  import org.xembly.ImpossibleModificationException;
12  import org.xembly.Xembler;
13  
14  /**
15   * This class tracks the time it takes to convert a bytecode to a XMIR program.
16   * @since 0.6
17   */
18  final class MeasuredEo {
19  
20      /**
21       * Original transformation.
22       */
23      private final XML xmir;
24  
25      /**
26       * Ctor.
27       * @param directives Directives to build the EO from.
28       */
29      MeasuredEo(final DirectivesObject directives) {
30          this(new XMLDocument(new Xembler(directives).xmlQuietly()));
31      }
32  
33      /**
34       * Ctor.
35       * @param xmir Original xmir representation.
36       */
37      private MeasuredEo(final XML xmir) {
38          this.xmir = xmir;
39      }
40  
41      /**
42       * XML representation of the EO.
43       * @return XML representation
44       * @throws ImpossibleModificationException If something goes wrong
45       */
46      XML asXml() throws ImpossibleModificationException {
47          final long start = System.currentTimeMillis();
48          final long end = System.currentTimeMillis();
49          return new XMLDocument(
50              new Xembler(
51                  new Directives()
52                      .xpath("/program[@ms]/@ms")
53                      .set(String.format("%d", end - start))
54              ).apply(this.xmir.deepCopy())
55          );
56      }
57  }
