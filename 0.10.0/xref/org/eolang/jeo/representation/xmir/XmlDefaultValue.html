5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.Optional;
8   import org.eolang.jeo.representation.bytecode.BytecodeDefaultValue;
9   import org.eolang.jeo.representation.bytecode.BytecodeMethod;
10  
11  /**
12   * XMIR of annotation default value.
13   * @since 0.3
14   */
15  public final class XmlDefaultValue {
16  
17      /**
18       * Default value XMIR node.
19       */
20      private final XmlNode node;
21  
22      /**
23       * Constructor.
24       * @param node Default value XMIR node.
25       */
26      XmlDefaultValue(final XmlNode node) {
27          this.node = node;
28      }
29  
30      /**
31       * Convert to bytecode.
32       * @return Bytecode default value.
33       */
34      public Optional<BytecodeDefaultValue> bytecode() {
35          return this.node.children().findFirst().map(
36              property -> new BytecodeDefaultValue(
37                  new XmlAnnotationValue(property).bytecode()
38              )
39          );
40      }
41  
42      /**
43       * Write to method.
44       * @param method Method.
45       */
46      public void writeTo(final BytecodeMethod method) {
47          this.bytecode().ifPresent(method::defvalue);
48      }
49  }
