5   package org.eolang.jeo.representation.directives;
6   
7   import java.util.ArrayList;
8   import java.util.Arrays;
9   import java.util.Iterator;
10  import java.util.List;
11  import java.util.stream.Collectors;
12  import org.xembly.Directive;
13  import org.xembly.Directives;
14  
15  /**
16   * Directives for method parameters.
17   * @since 0.1
18   */
19  public final class DirectivesMethodParams implements Iterable<Directive> {
20  
21      /**
22       * Parameters.
23       */
24      private final List<Iterable<Directive>> params;
25  
26      /**
27       * Constructor.
28       */
29      public DirectivesMethodParams() {
30          this(new ArrayList<>(0));
31      }
32  
33      /**
34       * Constructor.
35       * @param params Parameters.
36       */
37      @SafeVarargs
38      public DirectivesMethodParams(final Iterable<Directive>... params) {
39          this(Arrays.asList(params));
40      }
41  
42      /**
43       * Constructor.
44       * @param params Parameters.
45       */
46      public DirectivesMethodParams(final List<Iterable<Directive>> params) {
47          this.params = params;
48      }
49  
50      @Override
51      public Iterator<Directive> iterator() {
52          return new DirectivesJeoObject(
53              "params",
54              "params",
55              this.params.stream().map(Directives::new).collect(Collectors.toList())
56          ).iterator();
57      }
58  }
