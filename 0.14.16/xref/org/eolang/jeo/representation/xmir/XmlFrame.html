5   package org.eolang.jeo.representation.xmir;
6   
7   import java.util.Optional;
8   import org.eolang.jeo.representation.bytecode.BytecodeFrame;
9   
10  /**
11   * Xmir representation of bytecode frame.
12   *
13   * @since 0.3
14   */
15  public final class XmlFrame implements XmlBytecodeEntry {
16  
17      /**
18       * Xmir node.
19       */
20      private final XmlJeoObject node;
21  
22      /**
23       * Constructor.
24       * @param xmlnode Xmir node
25       */
26      public XmlFrame(final XmlNode xmlnode) {
27          this(new XmlJeoObject(xmlnode));
28      }
29  
30      /**
31       * Constructor.
32       * @param node XML Jeo object node
33       */
34      private XmlFrame(final XmlJeoObject node) {
35          this.node = node;
36      }
37  
38      /**
39       * Constructor.
40       * @param lines Lines of XML
41       */
42      XmlFrame(final String... lines) {
43          this(String.join("\n", lines));
44      }
45  
46      /**
47       * Constructor.
48       * @param xml XML
49       */
50      private XmlFrame(final String xml) {
51          this(new NativeXmlNode(xml));
52      }
53  
54      /**
55       * Convert to bytecode.
56       * @return Bytecode frame.
57       */
58      public BytecodeFrame bytecode() {
59          final Object[] locals = this.locals();
60          final Object[] stack = this.stack();
61          return new BytecodeFrame(
62              this.type(),
63              this.nlocals().orElse(locals.length),
64              locals,
65              this.nstack().orElse(stack.length),
66              stack
67          );
68      }
69  
70      /**
71       * Type of frame.
72       * @return Type.
73       */
74      private int type() {
75          return (int) new XmlOperand(
76              this.byName("type").orElseThrow(
77                  () -> new IllegalStateException(
78                      String.format("Can't find 'type' child in '%s'", this.node)
79                  )
80              )
81          ).asObject();
82      }
83  
84      /**
85       * Local variables.
86       * @return Local variables.
87       */
88      private Object[] locals() {
89          return new XmlFrameValues(
90              this.byName("locals").orElseThrow(
91                  () -> new IllegalStateException(
92                      String.format("Can't find 'locals' child in '%s'", this.node)
93                  )
94              )
95          ).values();
96      }
97  
98      /**
99       * Stack elements.
100      * @return Stack elements.
101      */
102     private Object[] stack() {
103         return new XmlFrameValues(
104             this.byName("stack").orElseThrow(
105                 () -> new IllegalStateException(
106                     String.format("Can't find 'stack' child in '%s'", this.node)
107                 )
108             )
109         ).values();
110     }
111 
112     private Optional<Integer> nlocals() {
113         return this.byName("nlocals")
114             .map(XmlOperand::new)
115             .map(XmlOperand::asObject)
116             .map(Integer.class::cast);
117     }
118 
119     private Optional<Integer> nstack() {
120         return this.byName("nstack")
121             .map(XmlOperand::new)
122             .map(XmlOperand::asObject)
123             .map(Integer.class::cast);
124     }
125 
126     private Optional<XmlNode> byName(final String name) {
127         return this.node.children()
128             .filter(
129                 child -> child.attribute("name")
130                     .map(s -> s.startsWith(name))
131                     .orElse(false))
132             .findFirst();
133     }
134 }
