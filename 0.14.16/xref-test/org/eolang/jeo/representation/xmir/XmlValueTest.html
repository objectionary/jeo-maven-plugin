19  /**
20   * Test case for {@link XmlValue}.
21   * @since 0.6
22   */
23  final class XmlValueTest {
24  
25      @Test
26      void parsesHexStringAsInteger() {
27          final int expected = 1057;
28          final int actual = (int) new XmlValue(
29              new NativeXmlNode(
30                  "<o base='Q.org.eolang.number'><o base='Q.org.eolang.bytes'><o>40-90-84-00-00-00-00-00</o></o></o>"
31              )
32          ).object();
33          MatcherAssert.assertThat(
34              String.format(
35                  "Can't parse hex string as integer, or the result is wrong; expected %d, got %d",
36                  expected,
37                  actual
38              ),
39              actual,
40              Matchers.is(expected)
41          );
42      }
43  
44      @Test
45      void initializesFromNamedObject() {
46          MatcherAssert.assertThat(
47              "XmlValue should be initialized from XmlNamedObject",
48              new XmlValue(
49                  new XmlNamedObject(
50                      new JcabiXmlNode(
51                          "<o base='Q.org.eolang.string'><o base='Q.org.eolang.bytes'><o>--</o></o></o>"
52                      )
53                  )
54              ).string(),
55              Matchers.emptyString()
56          );
57      }
58  
59      @Test
60      void parsesNumber() {
61          MatcherAssert.assertThat(
62              "Xml value should parse number from hex string",
63              new XmlValue(
64                  new XmlNamedObject(
65                      new JcabiXmlNode(
66                          new StringBuilder()
67                              .append("<o base=\"Q.org.eolang.number\" name=\"access\">\n")
68                              .append("  <o base=\"Q.org.eolang.bytes\">\n")
69                              .append("    <o>40-40-00-00-00-00-00-00</o>\n")
70                              .append("  </o>\n")
71                              .append("</o>").toString()
72                      )
73                  )
74              ).object(),
75              Matchers.equalTo(0x20)
76          );
77      }
78  
79      @ParameterizedTest
80      @MethodSource("values")
81      void decodesEncodesCorrectly(final Object origin) {
82          MatcherAssert.assertThat(
83              "Decoding and encoding are not consistent",
84              new XmlValue(
85                  new NativeXmlNode(new Xembler(new DirectivesValue(origin)).xmlQuietly())
86              ).object(),
87              Matchers.equalTo(origin)
88          );
89      }
90  
91      @ParameterizedTest
92      @ValueSource(strings = {
93          "Φ", "Ψ", "Ω", "Ϊ", "Ϋ", "ά", "έ", "ή", "ί", "ΰ", "α", "β", "γ", "δ", "ε", "ζ", "η", "θ", "ι", "κ", "λ", "μ"
94      })
95      void decodesUnicodeCharacters(final String unicode) throws ImpossibleModificationException {
96          MatcherAssert.assertThat(
97              "Can't decode unicode characters",
98              new XmlValue(
99                  new NativeXmlNode(
100                     new Xembler(
101                         new DirectivesValue(unicode)
102                     ).xml()
103                 )
104             ).string(),
105             Matchers.equalTo(unicode)
106         );
107     }
