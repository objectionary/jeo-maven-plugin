24  package org.eolang.jeo.representation.directives;
25  
26  import java.util.Iterator;
27  import lombok.ToString;
28  import org.eolang.jeo.representation.bytecode.BytecodeObject;
29  import org.eolang.jeo.representation.bytecode.Codec;
30  import org.eolang.jeo.representation.bytecode.EoCodec;
31  import org.eolang.jeo.representation.bytecode.PlainLongCodec;
32  import org.xembly.Directive;
33  import org.xembly.Directives;
34  
35  /**
36   * Data Object Directive in EO language.
37   *
38   * @since 0.1.0
39   */
40  @ToString
41  public final class DirectivesValue implements Iterable<Directive> {
42  
43      /**
44       * Array of hexadecimal characters.
45       * Used for converting bytes to hexadecimal.
46       * See {@link #bytesToHex(byte[])}.
47       */
48      private static final char[] HEX_ARRAY = "0123456789ABCDEF".toCharArray();
49  
50      /**
51       * Maximum long value that can be represented as double.
52       * Any value greater than this will be represented incorrectly.
53       */
54      private static final long MAX_LONG_DOUBLE = 9_007_199_254_740_992L;
55  
56      /**
57       * Minimum long value that can be represented as double.
58       * Any value less than this will be represented incorrectly.
59       */
60      private static final long MIN_LONG_DOUBLE = -9_007_199_254_740_992L;
61  
62      /**
63       * Default codec.
64       */
65      private static final Codec CODEC = new EoCodec();
66  
67      /**
68       * Name.
69       */
70      private final String name;
71  
72      /**
73       * Value.
74       */
75      private final BytecodeObject value;
76  
77      /**
78       * Constructor.
79       *
80       * @param data Data.
81       * @param <T> Data type.
82       */
83      public <T> DirectivesValue(final T data) {
84          this("", data);
85      }
86  
87      /**
88       * Constructor.
89       *
90       * @param name Name.
91       * @param data Data.
92       * @param <T> Data type.
93       */
94      public <T> DirectivesValue(final String name, final T data) {
95          this(name, new BytecodeObject(data));
96      }
97  
98      /**
99       * Constructor.
100      *
101      * @param name Name.
102      * @param value Value.
103      */
104     public DirectivesValue(final String name, final BytecodeObject value) {
105         this.name = name;
106         this.value = value;
107     }
108 
109     @Override
110     public Iterator<Directive> iterator() {
111         final String type = this.type();
112         final Iterable<Directive> res;
113         final Codec codec = DirectivesValue.CODEC;
114         switch (type) {
115             case "byte":
116             case "short":
117             case "int":
118             case "float":
119             case "double":
120                 res = this.jeoNumber(type, codec);
121                 break;
122             case "long":
123                 if (this.fits()) {
124                     res = this.jeoNumber(type, codec);
125                 } else {
126                     res = this.jeoObject(type, new PlainLongCodec(codec));
127                 }
128                 break;
129             case "string":
130                 res = this.eoObject(type, codec);
131                 break;
132             case "bool":
133                 res = this.booleanObject(type);
134                 break;
135             default:
136                 res = this.jeoObject(type, codec);
137                 break;
138         }
139         return res.iterator();
140     }
141 
142     /**
143      * Value of the data.
144      *
145      * @param codec Codec
146      * @return Value
147      */
148     String hex(final Codec codec) {
149         return DirectivesValue.bytesToHex(this.value.encode(codec));
150     }
151 
152     /**
153      * Type of the data.
154      *
155      * @return Type
156      */
157     String type() {
158         return this.value.type();
159     }
160 
161     /**
162      * Check if the value fits into the double.
163      *
164      * @return True if fits.
165      */
166     private boolean fits() {
167         final long val = ((Number) this.value.value()).longValue();
168         return val >= DirectivesValue.MIN_LONG_DOUBLE && val <= DirectivesValue.MAX_LONG_DOUBLE;
169     }
170 
171     /**
172      * EO object.
173      *
174      * @param base Base.
175      * @param codec Codec to use for bytes encoding.
176      * @return EO object directives.
177      */
178     private DirectivesEoObject eoObject(final String base, final Codec codec) {
179         return new DirectivesEoObject(
180             base,
181             this.name,
182             new DirectivesComment(this.comment()),
183             new DirectivesBytes(this.hex(codec))
184         );
185     }
186 
187     /**
188      * JEO object.
189      *
190      * @param base Base.
191      * @param codec Codec to use for bytes encoding.
192      * @return JEO object directives.
193      */
194     private DirectivesJeoObject jeoObject(final String base, final Codec codec) {
195         return new DirectivesJeoObject(
196             base,
197             this.name,
198             new DirectivesComment(this.comment()),
199             new DirectivesBytes(this.hex(codec))
200         );
201     }
202 
203     /**
204      * JEO number.
205      *
206      * @param base Object base.
207      * @param codec Codec to use for bytes encoding.
208      * @return JEO number directives.
209      */
210     private DirectivesJeoObject jeoNumber(final String base, final Codec codec) {
211         return new DirectivesJeoObject(
212             base,
213             this.name,
214             new DirectivesComment(this.comment()),
215             new DirectivesNumber(this.hex(codec))
216         );
217     }
218 
219     /**
220      * Boolean object.
221      *
222      * @param type Type of the object, usually "bool".
223      * @return Boolean object directives.
224      */
225     private Iterable<Directive> booleanObject(final String type) {
226         final String base;
227         if ((boolean) this.value.value()) {
228             base = "org.eolang.true";
229         } else {
230             base = "org.eolang.false";
231         }
232         return new DirectivesJeoObject(
233             type,
234             this.name,
235             new Directives().add("o").attr("base", base).up()
236         );
237     }
238 
239     /**
240      * Bytes of the representative comment.
241      *
242      * @return Sting comment.
243      */
244     private String comment() {
245         final String result;
246         final Object object = this.value.value();
247         if (object instanceof String) {
248             result = String.format("\"%s\"", object);
249         } else {
250             result = String.valueOf(object);
251         }
252         return result;
253     }
254 
255     /**
256      * Bytes to HEX.
257      * The efficient way to convert bytes to hexadecimal.
258      * ATTENTION!
259      * Do not modify this method.
260      * It is an optimized version that saves memory and CPU.
261      * Actually, the solution is based on the following StackOverflow answer:
262      * <a href="https://stackoverflow.com/a/9855338/10423604">here</a>
263      * You can find the full explanation or any other examples there.
264      *
265      * @param bytes Bytes.
266      * @return Hexadecimal value as string.
267      */
268     private static String bytesToHex(final byte[] bytes) {
269         final String res;
270         if (bytes == null || bytes.length == 0) {
271             res = "--";
272         } else {
273             final int length = bytes.length;
274             final char[] hex = new char[length * 3];
275             for (int index = 0; index < length; ++index) {
276                 final int value = bytes[index] & 0xFF;
277                 hex[index * 3] = DirectivesValue.HEX_ARRAY[value >>> 4];
278                 hex[index * 3 + 1] = DirectivesValue.HEX_ARRAY[value & 0x0F];
279                 hex[index * 3 + 2] = '-';
280             }
281             if (hex.length == 3) {
282                 res = new String(hex);
283             } else {
284                 res = new String(hex, 0, hex.length - 1);
285             }
286         }
287         return res;
288     }
