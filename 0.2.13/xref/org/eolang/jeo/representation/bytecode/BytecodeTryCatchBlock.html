24  package org.eolang.jeo.representation.bytecode;
25  
26  import com.jcabi.log.Logger;
27  import lombok.ToString;
28  import org.objectweb.asm.Label;
29  import org.objectweb.asm.MethodVisitor;
30  
31  /**
32   * Bytecode try-catch block.
33   * @since 0.1
34   */
35  @ToString
36  public final class BytecodeTryCatchBlock implements BytecodeEntry {
37  
38      /**
39       * Start label.
40       */
41      private final Label start;
42  
43      /**
44       * End label.
45       */
46      private final Label end;
47  
48      /**
49       * Handler label.
50       */
51      private final Label handler;
52  
53      /**
54       * Exception type.
55       */
56      private final String type;
57  
58      /**
59       * Constructor.
60       * @param startlabel Start label.
61       * @param endlabel End label.
62       * @param handlerlabel Handler label.
63       * @param exception Exception type.
64       * @checkstyle ParameterNumberCheck (5 lines)
65       */
66      public BytecodeTryCatchBlock(
67          final Label startlabel,
68          final Label endlabel,
69          final Label handlerlabel,
70          final String exception
71      ) {
72          this.start = startlabel;
73          this.end = endlabel;
74          this.handler = handlerlabel;
75          this.type = exception;
76      }
77  
78      @Override
79      public void writeTo(final MethodVisitor visitor) {
80          Logger.debug(
81              this,
82              String.format(
83                  "Writing try-catch entry into the method with the following values: start=%s, end=%s, handler=%s, type=%s",
84                  this.start,
85                  this.end,
86                  this.handler,
87                  this.type
88              )
89          );
90          visitor.visitTryCatchBlock(this.start, this.end, this.handler, this.type);
91      }
92  }
