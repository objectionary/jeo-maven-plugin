41  /**
42   * Set of integration tests for Java source parsing and transformation.
43   * The default pipeline for all the tests is:
44   * - Generate random Java source code
45   * - Compile it into bytecode
46   * - Transform bytecode into XMIR
47   * - Transform XMIR back into bytecode
48   * - Compare the original bytecode with the transformed one.
49   * In this class we compare bytecode as strings because the order of values in
50   * the bytecode constant pool might differ, however the bytecode is still the same.
51   * @since 0.1
52   * @todo #218:90min Add debug information to bytecode.
53   *  Currently we do not add any debug information to the bytecode.
54   *  We added -g:none flag to the compiler in order to avoid adding
55   *  any debug information.
56   *  See {@link #transformsRandomJavaSourceCodeIntoEoAndBack(java.nio.file.Path)} for more info.
57   *  We have to add debug information to the bytecode.
58   *  When it is done, we have to remove -g:none flag from the test.
59   */
60  class JavaSourceCompilationIT {
61  
62      @Test
63      @EnabledIf(value = "hasJavaCompiler", disabledReason = "Java compiler is not available")
64      void transformsRandomJavaSourceCodeIntoEoAndBack(@TempDir final Path temp) throws IOException {
65          final Bytecode expected = JavaSourceCompilationIT.compile(temp, new RandomJavaClass());
66          MatcherAssert.assertThat(
67              "Bytecode is not equal to the original one, check that transformation is correct and does not change the bytecode",
68              new EoRepresentation(
69                  new BytecodeRepresentation(expected).toEO()
70              ).toBytecode().toString(),
71              Matchers.equalTo(expected.toString())
72          );
73      }
74  
75      /**
76       * Compile random java class into bytecode.
77       * @param where Where to compile.
78       * @param clazz Random java class.
79       * @return Bytecode.
80       */
81      private static Bytecode compile(
82          final Path where,
83          final RandomJavaClass clazz
84      ) throws IOException {
85          final Path src = where.resolve(String.format("%s.java", clazz.name()));
86          Files.write(src, clazz.src().getBytes(StandardCharsets.UTF_8));
87          ToolProvider.getSystemJavaCompiler().run(
88              System.in,
89              System.out,
90              System.err,
91              "-g:none",
92              "-source", "11",
93              "-target", "11",
94              src.toString()
95          );
96          return new Bytecode(
97              Files.readAllBytes(where.resolve(String.format("%s.class", clazz.name())))
98          );
99      }
100 
101     /**
102      * Check if java compiler is available.
103      * Don't care about PMD warning: this method is used in @EnabledIf annotation.
104      * @return True if java compiler is available.
105      */
106     @SuppressWarnings("PMD.UnusedPrivateMethod")
107     private static boolean hasJavaCompiler() {
108         return ToolProvider.getSystemJavaCompiler() != null;
109     }
