5   package org.eolang.jeo.representation.xmir;
6   
7   import org.eolang.jeo.representation.bytecode.BytecodeAttribute;
8   import org.eolang.jeo.representation.bytecode.InnerClass;
9   import org.eolang.jeo.representation.directives.DirectivesAttribute;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.Assertions;
13  import org.junit.jupiter.api.Test;
14  import org.xembly.ImpossibleModificationException;
15  import org.xembly.Xembler;
16  
17  /**
18   * Test case for {@link XmlAttribute}.
19   * @since 0.6
20   */
21  final class XmlAttributeTest {
22  
23      @Test
24      void convertsToDomainInnerClass() throws ImpossibleModificationException {
25          final BytecodeAttribute expected = new InnerClass("name", "outer", "inner", 0);
26          MatcherAssert.assertThat(
27              "We expect the attribute to be converted to a correct  domain attribute (inner class)",
28              new XmlAttribute(new Xembler(expected.directives()).xml()).attribute(),
29              Matchers.equalTo(expected)
30          );
31      }
32  
33      @Test
34      void throwsExceptionIfCannotIdentifyAttribute() {
35          MatcherAssert.assertThat(
36              "We expect an exception message be understandable and clear",
37              Assertions.assertThrows(
38                  IllegalArgumentException.class,
39                  () -> new XmlAttribute(
40                      new Xembler(new DirectivesAttribute("unknown")).xmlQuietly()
41                  ).attribute(),
42                  "We expect an exception to be thrown if the attribute cannot be identified"
43              ).getMessage(),
44              Matchers.equalTo("Unknown attribute base 'Q.jeo.unknown'")
45          );
46      }
47  }
